#!/usr/bin/env bash
set -euo pipefail

script_name="${script_name:-$(basename "${BASH_SOURCE[0]}")}"

rt() {
    local perform_readiness_check=false
    local args=()

    # Parse args and strip our custom flag
    while [[ $# -gt 0 ]]; do
        case "$1" in
            --perform_readiness_check)
            perform_readiness_check=true
            ;;
            *)
            args+=("$1")
            ;;
        esac
        shift
    done

    if (( perform_readiness_check )); then
        command -v gcloud >/dev/null 2>&1 || { echo "ERROR: gcloud not found"; fatal 1; }
    fi

    local service="${SERVICE_NAME:-${STACK}}"

    # Cache key for this Cloud Run service
    local cache_key="${DEPLOYMENT_ID}:${REGION}:${service}"

    # Global cache vars (not declared local on purpose)
    #   __GCS_WP_URL_CACHE      - cached URL
    #   __GCS_WP_URL_CACHE_KEY  - cache key it was derived from
    local wp_url

    if [[ -n "${__GCS_WP_URL_CACHE:-}" && "${__GCS_WP_URL_CACHE_KEY:-}" == "${cache_key}" ]]; then
        # Use cached URL
        wp_url="${__GCS_WP_URL_CACHE}"
    else
        # Resolve Cloud Run URL and cache it
        wp_url="$(gcloud run services describe "${service}" \
        --region="${REGION}" \
        --project="${DEPLOYMENT_ID}" \
        --format='value(status.url)')"

        if [[ -z "${wp_url}" ]]; then
            echo "ERROR: could not resolve Cloud Run URL for ${service} in ${REGION}/${DEPLOYMENT_ID}" >&2
            fatal 1
        fi

        __GCS_WP_URL_CACHE="${wp_url}"
        __GCS_WP_URL_CACHE_KEY="${cache_key}"
    fi

    local -a cmd=(wp --url="${wp_url}" "${args[@]}")
    log_cmd "${cmd[@]}"
    "${cmd[@]}"
}

#!/usr/bin/env bash
set -euo pipefail

script_name="${script_name:-$(basename "${BASH_SOURCE[0]}")}"

rt() {
    local perform_readiness_check=false
    local args=()

    # Parse args and strip our custom flag
    while [[ $# -gt 0 ]]; do
        case "$1" in
            --perform_readiness_check)
            perform_readiness_check=true
            ;;
            *)
            args+=("$1")
            ;;
        esac
        shift
    done

    if $perform_readiness_check == "true"; then
        # Ensure stack is up
        if ! docker compose ps -q wp >/dev/null 2>&1; then
            docker compose up -d
        fi

        # Light readiness guard so compose run doesn't race a cold wp
        log "performing readiness check"
        for i in {1..20}; do
            if docker compose exec -T wp curl -fsS http://localhost/wp-login.php >/dev/null 2>&1; then
                break
            fi
        sleep 1
        done
    fi

    # Kill all pagers; still pipe to cat so we never land in ":"
    docker compose run --rm \
    -e HOME=/var/www \
    -e WP_CLI_CACHE_DIR=/tmp/wp-cli-cache \
    -e PAGER=cat \
    -e MANPAGER=cat \
    -e GIT_PAGER=cat \
    -e LESS= \
    wpcli wp --no-color --allow-root --path=/var/www/html "${args[@]}" | cat
}

#!/usr/bin/env bash
# Purpose: Capture project + Traefik state, archive it, and (optionally) copy to a host path.
# Behavior: Never abort on errors; log to stdout/stderr and continue.

set -u
set -o pipefail || true

script_name="${script_name:-$(basename "${BASH_SOURCE[0]}")}"

ensure_portmason_share() {
  # Only discover if unset/empty
  [[ -n "${PORTMASON_SHARE:-}" ]] && return 0

  local target="pm-helpers"
  local here parent default="/usr/local/share/portmason/bin"

  here="$(pwd -P)"
  PROJECT_OUT="${PROJECT_OUT:-$here}"
  parent="$(cd .. && pwd -P)"

  # 1) current dir
  if [[ -f "${here}/${target}" ]]; then
    PORTMASON_SHARE="${here}"
    export PORTMASON_SHARE
    return 0
  fi

  # 2) parent dir
  if [[ -f "${parent}/${target}" ]]; then
    PORTMASON_SHARE="${parent}"
    export PORTMASON_SHARE
    return 0
  fi

  # 3) adjacent sibling dirs (children of parent, excluding current)
  local d
  for d in "${parent}"/*; do
    [[ -d "$d" ]] || continue
    [[ "$(cd "$d" && pwd -P)" == "$here" ]] && continue
    if [[ -f "${d}/${target}" ]]; then
      PORTMASON_SHARE="${d}"
      export PORTMASON_SHARE
      return 0
    fi
  done

  # 4) default location
  if [[ -f "${default}/${target}" ]]; then
    PORTMASON_SHARE="${default}"
    export PORTMASON_SHARE
    return 0
  fi

  printf 'ERROR: PORTMASON_SHARE unset and could not find %s in %s, %s, or adjacent dirs under %s\n' \
    "$target" "$here" "$parent" "$parent" >&2
  return 2
}
ensure_portmason_share

. "${PORTMASON_SHARE}/pm-helpers"

log "PROJECT_OUT: $PROJECT_OUT"

try() {
  local desc="$1"; shift
  log "$desc"
  "$@"
  local rc=$?
  if (( rc != 0 )); then
    warn "$desc failed with exit code $rc"
  fi
  return 0
}

exists() { [ -e "$1" ]; }
isfile() { [ -f "$1" ]; }
writable_dir() { [ -d "$1" ] && [ -w "$1" ]; }

# -----------------------------
# Args
# -----------------------------
COPY_TARGET="${COPY_TARGET:-}"
SHOW_WHICH=0
COPILOT_MODE="${COPILOT_MODE:-0}"

while (( "$#" )); do
  case "${1:-}" in
    --copy)
      shift || true
      COPY_TARGET="${1:-}"
      ;;
    --copy-public)
      COPY_TARGET="/mnt/c/Users/Public/Desktop"
      ;;
    --which)
      SHOW_WHICH=1
      ;;
    --copy-target)
      shift || true
      COPY_TARGET="${1:-}"
      ;;
    --copilot)
      shift || true
      COPILOT_MODE=1
      ;;
    *)
      warn "Unknown arg: $1 (ignored)"
      ;;
  esac
  shift || true
done

if (( SHOW_WHICH == 1 )); then
  echo "which: $(command -v "$SCRIPT_NAME" 2>/dev/null || true)"
fi

# -----------------------------
# Inputs / Defaults
# -----------------------------
PROJECT_ROOT="${PROJECT_ROOT:-$PWD}"
TRAEFIK_DIR="${TRAEFIK_DIR:-$HOME/code/traefik}"
CURRENT_DIR_NAME="${PWD##*/}"
STACK="${STACK:-$CURRENT_DIR_NAME}"
CAPTURE_ROOT="${PROJECT_ROOT}/capture"

ts="$(date +'%Y-%m-%dT%H-%M-%S%z')"
RUN_DIR="$CAPTURE_ROOT/${STACK}_$ts"
ARCHIVE_PATH="$CAPTURE_ROOT/${STACK}_$ts.tar.gz"
ARCHIVE_BASENAME="$(basename "$ARCHIVE_PATH")"

log "PROJECT_ROOT: $PROJECT_ROOT"
log "STACK:        $STACK"
log "CAPTURE_ROOT:   $CAPTURE_ROOT"
log "RUN_DIR:      $RUN_DIR"
log "PORTMASON_SHARE:      $PORTMASON_SHARE"
log "ARCHIVE:      $ARCHIVE_PATH"
[ -n "${COPY_TARGET:-}" ] && log "COPY_TARGET:  $COPY_TARGET"

mkdir -p "$RUN_DIR" || {
  err "Could not create run dir: $RUN_DIR"
  exit 1
}

# -----------------------------
# Docker/Compose snapshot (best-effort)
# -----------------------------
DOCKER_OK=0
if command -v docker >/dev/null 2>&1; then
  DOCKER_OK=1
elif command -v podman >/dev/null 2>&1; then
  DOCKER_OK=1
  alias docker='podman'
fi

COMPOSE_FILE_FOUND=0
if [ -f "$PROJECT_ROOT/docker-compose.yml" ] || [ -f "$PROJECT_ROOT/compose.yml" ]; then
  COMPOSE_FILE_FOUND=1
fi

if (( DOCKER_OK == 1 && COMPOSE_FILE_FOUND == 1 )); then
  DOCKER_OUT="$RUN_DIR/docker"
  mkdir -p "$DOCKER_OUT" || warn "Could not create $DOCKER_OUT"

  try "docker compose ps"            bash -lc "cd '$PROJECT_ROOT' && docker compose ps > '$DOCKER_OUT/compose-ps.txt' 2>&1"
  try "docker network ls"            bash -lc "docker network ls > '$DOCKER_OUT/network-ls.txt' 2>&1"
  try "docker volume ls"             bash -lc "docker volume ls  > '$DOCKER_OUT/volume-ls.txt' 2>&1"
  try "docker ps --all"              bash -lc "docker ps -a      > '$DOCKER_OUT/ps-a.txt' 2>&1"
  try "docker images"                bash -lc "docker images     > '$DOCKER_OUT/images.txt' 2>&1"
else
  log "Skipping Docker/Compose capture (no docker or compose file)."
fi

# -----------------------------
# Traefik state (best-effort, no 'install' dependency)
# -----------------------------
if [ -d "$TRAEFIK_DIR" ]; then
    TRAEFIK_OUT="$RUN_DIR/traefik"
    mkdir -p "$TRAEFIK_OUT" || warn "Could not create $TRAEFIK_OUT"

    try "Collecting Traefik state from: ${TRAEFIK_DIR:-<TRAEFIK_DIR not set>}" bash -eu -c '
    TRAEFIK_DIR="$1"
    TRAEFIK_OUT="$2"

    # If TRAEFIK_DIR is not set or doesn’t exist, nothing to do.
    if [ -z "${TRAEFIK_DIR:-}" ] || [ ! -d "$TRAEFIK_DIR" ]; then
        exit 0
    fi

    shopt -s nullglob || true
    cd "$TRAEFIK_DIR" || exit 0

    files=( rules/*.yml rules/*.yaml dynamic/*.yml dynamic/*.yaml mgt-scripts/* traefik.yml traefik.yaml )
    if (( ${#files[@]} == 0 )); then
        exit 0
    fi

    for f in "${files[@]}"; do
        dst="${TRAEFIK_OUT}/$f"
        mkdir -p "$(dirname "$dst")" || true
        cp -fL "$f" "$dst" 2>/dev/null || true
    done
    ' _ "${TRAEFIK_DIR:-}" "$TRAEFIK_OUT"
else
  log "Skipping Traefik capture (dir not found: $TRAEFIK_DIR)."
fi

# -----------------------------
# Project files (best-effort)
# -----------------------------
PROJ_OUT="$RUN_DIR/project"
mkdir -p "$PROJ_OUT" || warn "Could not create $PROJ_OUT"
try "Copying project files (no redaction)…" bash -lc "
  rsync -aL --delete \
    --exclude 'state' \
    --exclude 'capture' \
    --exclude '.git' \
    --exclude 'node_modules' \
    --exclude 'dist' \
    --exclude 'build' \
    --exclude '*.log_archive' \
    '$PROJECT_ROOT/' '$PROJ_OUT/'
"

# -----------------------------
# Portmason files (best-effort)
# -----------------------------
PORTMASON_OUT="$RUN_DIR/portmason"
mkdir -p "$PORTMASON_OUT" || warn "Could not create $PORTMASON_OUT"
try "Copying portmason files (no redaction)…" bash -lc "
  rsync -aL --delete \
    --exclude '.git' \
    '$PORTMASON_SHARE/' '$PORTMASON_OUT/'
"

# -----------------------------
# System snapshot (/usr/local/bin → sys/)
# -----------------------------
SYS_OUT="$RUN_DIR/usr/local/bin"
mkdir -p "$SYS_OUT" || warn "Could not create $SYS_OUT"

try "Copying real files from /usr/local/bin (no symlinks) into sys/..." bash -eu -c '
  SYS_OUT="$1"

  dir="/usr/local/bin"
  [ -d "$dir" ] || exit 0

  # Exclude cosign + cosign* from the snapshot
  find "$dir" -maxdepth 1 -type f \
    ! -name "cosign" \
    ! -name "cosign*" \
    -print0 | while IFS= read -r -d "" f; do
      cp -fL "$f" "$SYS_OUT"/ 2>/dev/null || true
    done
' _ "$SYS_OUT"


# -----------------------------
# Git summary (best-effort)
# -----------------------------
if [ -d "$PROJECT_ROOT/.git" ]; then
  GIT_OUT="$RUN_DIR/git"
  mkdir -p "$GIT_OUT" || warn "Could not create $GIT_OUT"
  try "Capturing Git summary…" bash -lc "
    cd '$PROJECT_ROOT' || exit 0
    { git rev-parse --short HEAD     > '$GIT_OUT/commit.txt' 2>&1 || true; }
    { git status --porcelain=v1      > '$GIT_OUT/status.txt'  2>&1 || true; }
    { git log -20 --oneline          > '$GIT_OUT/log.txt'     2>&1 || true; }
    { git diff                        > '$GIT_OUT/diff.txt'   2>&1 || true; }
  "
else
  log "Skipping Git summary (no .git directory)."
fi

# -----------------------------
# Archive
# -----------------------------
if [ -w "$CAPTURE_ROOT" ]; then
  log "Creating archive: $ARCHIVE_PATH"
  (
    cd "$CAPTURE_ROOT" && \
    tar -czf "$ARCHIVE_PATH" "$(basename "$RUN_DIR")"
  ) || {
    warn "Failed to create archive at $ARCHIVE_PATH"
  }
else
  warn "Capture dir not writable; skipping archive at $ARCHIVE_PATH"
fi

# -----------------------------
# Copy (best-effort with fallbacks)
# -----------------------------
if [ -n "${COPY_TARGET:-}" ] && isfile "$ARCHIVE_PATH"; then
  if writable_dir "$COPY_TARGET"; then
    try "Copying archive to $COPY_TARGET" cp -fL "$ARCHIVE_PATH" "$COPY_TARGET/$ARCHIVE_BASENAME"
  else
    warn "Copy target not writable: $COPY_TARGET"
    for alt in "/mnt/c/Users/Public/Desktop" "$HOME/Desktop"; do
      if writable_dir "$alt"; then
        try "Fallback copy to $alt" cp -fL "$ARCHIVE_PATH" "$alt/$ARCHIVE_BASENAME"
        break
      fi
    done
  fi
fi

if (( COPILOT_MODE == 1 )); then
  missing_dependency=""
  command -v a2ps >/dev/null 2>&1 || { log "a2ps not found but can be installed via: sudo apt install -y a2ps"; missing_dependency="1"; }
  command -v ps2pdf >/dev/null 2>&1 || { log "ps2pdf not found but can be installed via: sudo apt install -y ghostscript"; missing_dependency="1"; }

  if [ -n "$missing_dependency" ]; then
    log "Missing dependencies for Copilot snapshot generation. See above."
    exit 1
  fi

  SNAPSHOT_FILE="$RUN_DIR/copilot_snapshot.txt"
  log "Generating Copilot snapshot: $SNAPSHOT_FILE"

  cd "${RUN_DIR}"
  
  # find all files (skip binary by using grep -Iq test)
  find . -type f | while read -r file; do
    # skip self (optional)
    [[ "$file" == "./$SNAPSHOT_FILE" ]] && continue

    # check if file is text (non-binary)
    if grep -Iq . "$file"; then
      {
        echo "============================"
        echo "FILE: $(realpath "$file")"
        echo "============================"
        echo ""
        cat "$file"
        echo ""
        echo ""
      } >> "$SNAPSHOT_FILE"
    fi
  done

  OUT_PDF="${SNAPSHOT_FILE%.*}.pdf"

  a2ps "${SNAPSHOT_FILE}" -o - | ps2pdf - "$OUT_PDF"

  echo "Done. Output written to $OUT_PDF"
fi

log "Capture complete."
echo "Run directory:  $RUN_DIR"
isfile "$ARCHIVE_PATH" && echo "Archive:        $ARCHIVE_PATH"
[ -n "${COPY_TARGET:-}" ] && echo "Copy target:    $COPY_TARGET (best-effort)"

#!/usr/bin/env bash
set -euo pipefail

: "${CUR_SCRIPT:=$(basename "${BASH_SOURCE[0]:-${0:-}}")}"
export CUR_SCRIPT

: "${PORTMASON_SHARE:="$(cd -- "$(dirname -- "${BASH_SOURCE[0]:-${0:-}}")" && pwd -P)"}"
. "${PORTMASON_SHARE}/pm-helpers" "$@"

if [ "$#" -ne 1 ]; then
  echo "Usage: $0 NEW_FOLDER_NAME" >&2
  exit 1
fi

target="$1"

# Create the target directory if it doesn't exist
mkdir -p "$target"

# Copy everything from current dir to target, excluding ID/timestamp files
rsync -a \
  --exclude='.project_uuid' \
  --exclude='.project_timestamp' \
  ./ "$target/"

# Generate an epoch timestamp in SECONDS (integer)
ts=""

# Prefer `date +%s` (works on GNU and BSD/macOS)
if command -v date >/dev/null 2>&1; then
  tmp_ts="$(date +%s 2>/dev/null || true)"
  # Ensure it's all digits
  case "$tmp_ts" in
    ''|*[!0-9]*) ;;   # not a pure integer, ignore
    *) ts="$tmp_ts" ;;
  esac
fi

# Fallback: Python
if [ -z "$ts" ]; then
  if command -v python3 >/dev/null 2>&1; then
    ts="$(python3 - <<'PY'
import time
print(int(time.time()))
PY
)"
  elif command -v python >/dev/null 2>&1; then
    ts="$(python - <<'PY'
import time
print(int(time.time()))
PY
)"
  else
    echo "Unable to generate seconds timestamp: need 'date' or Python" >&2
    exit 1
  fi
fi

# Write the new .project_timestamp in the target folder
printf '%s\n' "$ts" > "$target/.project_timestamp"

env_wildcard="*.env"
example_name=".env.example"
example_wildcard="*${example_name}"

env_wildcard="${target%/}/*.env"
example_name=".env.example"
example_wildcard="${target%/}/*${example_name}"

if compgen -G "${env_wildcard}" > /dev/null || compgen -G "${example_wildcard}" > /dev/null; then
    # At least one matching file exists → do nothing
    :
else
    # No *.env and no *.example files exist so we'll copy the default .env.example
    echo "No env/example files found, priming default..."
    script_dir="$( cd "$( dirname "${BASH_SOURCE[0]}" )" && pwd )"
    cp "${script_dir}/templates/${example_name}" "${target}"
fi

#!/usr/bin/env bash
set -euo pipefail

: "${CUR_SCRIPT:=$(basename "${BASH_SOURCE[0]}")}"
export CUR_SCRIPT

: "${PORTMASON_SHARE:=$(cd -- "$(dirname -- "${BASH_SOURCE[0]}")" && pwd -P)}"
. "${PORTMASON_SHARE}/pm-helpers" "$@"

# pm-container-nuke
# Default: FULLY EXECUTE and remove ONLY artifacts for the current STACK.
# Use --all to nuke ALL Docker artifacts on the machine (dangerous, requires confirmation).
#
# STACK detection:
#   1) If --stack <name> is provided, use it.
#   2) Else if $STACK is set and non-empty, use it.
#   3) Else use basename "$PWD".
#
# What counts as STACK "stuff" (scoped mode):
#   - Containers, networks, volumes that match ANY of:
#       * Label com.docker.compose.project == $STACK
#       * Label stack == $STACK
#       * Name/prefix patterns:
#           - Containers: "${STACK}-*"
#           - Networks/Volumes: "${STACK}_*"
#   - Images selected per --images mode (see below).
#
# Image selection (scoped mode) via --images <mode> (default: smart):
#   smart      -> union of: images used by matched containers ("associated"),
#                           images labeled stack=$STACK ("labeled"),
#                           images with repo or tag starting with $STACK ("tagged").
#   associated -> only images used by matched containers
#   labeled    -> only images labeled stack=$STACK
#   tagged     -> only images with repo or tag starting with $STACK
#   all        -> any of the above (same as smart)
#   none       -> do not remove images
#
# Options:
#   --stack <name>   Override detected STACK
#   --dry-run        Print the plan, do not delete
#   --images <mode>  Image selection mode in scoped mode (default: smart)
#   --all            Nuke ALL Docker artifacts on the machine (containers, networks, volumes, images)
#   -h|--help        Show help
#
# Notes:
#   - Default behavior performs deletions (no dry-run).
#   - We never remove default networks (bridge/host/none).
#
# ----------------------------------------------------------------------------

DO_DRYRUN="false"
IMAGES_MODE="smart"

# Helpers
unique () { awk 'BEGIN{RS="\n";FS=" "} { if (!seen[$1]++) print $1 }'; }

run_hooks () {
    local scope="${1:-local}"   # "local" or "global"
    local rc=0
    local file

    # Adapter-scoped hook: ${PORTMASON_SHARE}/pm-container-nuke-${ADAPTER_CODE}
    file="${PORTMASON_SHARE}/pm-container-nuke-${ADAPTER_CODE}"
#    [[ -x "${file}" ]] && { "${file}" "${scope}" || { [[ "$rc" == 0 ]] && rc=$? }
    [[ -x "${file}" ]] && { "${file}" "${scope}" || { [[ "$rc" == 0 ]] && rc=$?; }; }
    # Runtime-scoped hook: ${PORTMASON_SHARE}/pm-container-nuke-${RUNTIME_CODE}
    file="${PORTMASON_SHARE}/pm-container-nuke-${RUNTIME_CODE}"
#    [[ -x "${file}" ]] && { "${file}" "${scope}" || { [[ "$rc" == 0 ]] && rc=$? }
    [[ -x "${file}" ]] && { "${file}" "${scope}" || { [[ "$rc" == 0 ]] && rc=$?; }; }
    # In global mode, hooks can’t veto the nuke
    [[ "${scope}" == "global" ]] && return 0

    # In local mode, propagate non-zero rc to allow early exit
    return "${rc}"
}

# Resolve STACK according to requested logic
resolve_stack () {
    [[ -n "${STACK_FROM_ARG}" ]] && { echo "${STACK_FROM_ARG}"; return 0; }
    [[ -n "${STACK:-}" ]] && { echo "${STACK}"; return 0; }
    basename "${PWD}"
}

# Return a newline-delimited list of *volume names* that are actually mounted
# by the provided container IDs (from `.Mounts` where Type == "volume").
volumes_from_containers () {
  local cids="${1:-}"
  [[ -z "${cids}" ]] && return 0

  # Using docker inspect template avoids jq dependency.
  # `.Name` is only populated for Type=volume mounts.
  while IFS= read -r cid; do
    [[ -z "${cid}" ]] && continue
    docker inspect -f '{{range .Mounts}}{{if eq .Type "volume"}}{{.Name}}{{"\n"}}{{end}}{{end}}' "${cid}" 2>/dev/null || true
  done <<< "${cids}"
}

print_plan () {
  echo "=== PLAN ==="
  if [[ "${NUKE_ALL}" == 1 ]]; then
    echo "Mode: GLOBAL (--all)"
  else
    echo "Mode: STACK scoped"
    echo "STACK: ${STACK_VAL}"
    echo "Images mode: ${IMAGES_MODE}"
  fi
  echo "Containers:"
  if [[ -n "${containers}" ]]; then
    while IFS= read -r id; do
      [[ -z "$id" ]] && continue
      cname="$(docker inspect -f '{{.Name}}' "$id" 2>/dev/null | sed 's#^/##' || true)"
      echo "  - ${id} (${cname})"
    done <<< "${containers}"
  else
    echo "  (none)"
  fi
  echo "Networks:"
  if [[ -n "${networks}" ]]; then
    while IFS= read -r id; do
      [[ -z "$id" ]] && continue
      nname="$(docker network inspect -f '{{.Name}}' "$id" 2>/dev/null || true)"
      echo "  - ${id} (${nname})"
    done <<< "$(printf "%b" "${networks}")"
  else
    echo "  (none)"
  fi
  echo "Volumes:"
  if [[ -n "${volumes}" ]]; then
    while IFS= read -r name; do
      [[ -z "$name" ]] && continue
      echo "  - ${name}"
    done <<< "${volumes}"
  else
    echo "  (none)"
  fi
  echo "Images:"
  if [[ -n "${images}" ]]; then
    while IFS= read -r id; do
      [[ -z "$id" ]] && continue
      repo="$(docker image inspect -f '{{index .RepoTags 0}}' "$id" 2>/dev/null || true)"
      echo "  - ${id} (${repo})"
    done <<< "${images}"
  else
    echo "  (none)"
  fi
  echo "=== END PLAN ==="
}

usage () {
  cat <<'USAGE'
pm-container-nuke

Default: fully execute and remove ONLY artifacts for the detected STACK.
Use --all to nuke EVERYTHING Docker on this machine (dangerous).

STACK detection (in order):
  1) --stack <name>
  2) $STACK environment variable
  3) basename "$PWD"

Usage:
  pm-container-nuke [--stack <name>] [--dry-run] [--images smart|associated|labeled|tagged|none]
  pm-container-nuke --all [--dry-run]

Options:
  --stack <name>      Override detected STACK
  --dry-run           Print the plan only; do not delete
  --images <mode>     Image selection in scoped mode (smart|associated|labeled|tagged|none). Default: smart
  --all               Nuke all Docker artifacts (containers, networks, volumes, images)
  -h, --help          Show this help
USAGE
}

main() {
  STACK_VAL="$(resolve_stack)"
  [[ -z "${STACK_VAL}" ]] && fatal_error "Could not resolve STACK." 1

  case "${IMAGES_MODE}" in
    smart|associated|labeled|tagged|none|all) : ;;
    *) err "--images must be one of: smart|associated|labeled|tagged|none|all"; exit 1;;
  esac

  if [[ "${NUKE_ALL}" == 1 ]]; then
    warn "GLOBAL NUKE (--all): This will delete ALL Docker artifacts on this machine."
  else
    info "Scoped mode: targeting STACK='${STACK_VAL}' (images mode: ${IMAGES_MODE})"
  fi

  # Decide scope for hooks based on --all
  HOOK_SCOPE="local"
  if [[ "${NUKE_ALL}" == 1 ]]; then
    HOOK_SCOPE="global"
  fi

  # Always run hooks; capture their output and exit code
  hook_output=""
  if ! hook_output="$(run_hooks "${HOOK_SCOPE}" 2>&1)"; then
    hook_rc=$?

    # In global mode, hooks *should* already be non-vetoing, but if run_hooks
    # ever returns non-zero here, we just log and continue.
    if [[ "${HOOK_SCOPE}" == "global" ]]; then
      if [[ -n "${hook_output}" ]]; then
        echo "[pm-container-nuke] (global) hook output (ignored):" >&2
        echo "${hook_output}" >&2
      fi
    else
      case "${hook_rc}" in
        1)
          if [[ -n "${hook_output}" ]]; then
            echo "${hook_output}" >&2
            echo >&2
          fi
          echo "One or more helpers suggested NOT running the local nuke." >&2
          read -r -p "Abort local nuke? [Y/n] " ans
          ans=${ans:-Y}
          if [[ "${ans}" =~ ^[Yy]$ ]]; then
            echo "Aborting local nuke at helper’s request."
            exit 0
          fi
          ;;
        2)
          if [[ -n "${hook_output}" ]]; then
            echo "${hook_output}" >&2
            echo >&2
          fi
          echo "A helper returned a hard-fail (2). Aborting local nuke." >&2
          exit 0
          ;;
        *)
          if [[ -n "${hook_output}" ]]; then
            echo "[pm-container-nuke] Unexpected hook rc=${hook_rc}:" >&2
            echo "${hook_output}" >&2
          fi
          ;;
      esac
    fi
  fi

  # Build the plan
  containers=""
  networks=""
  volumes=""
  images=""

  if [[ "${NUKE_ALL}" == 1 ]]; then
    # Everything
    containers="$(docker ps -aq || true)"
    # all networks except default trio
    all_nets="$(docker network ls -q || true)"
    if [[ -n "${all_nets}" ]]; then
      while IFS= read -r nid; do
        [[ -z "$nid" ]] && continue
        nname="$(docker network inspect -f '{{.Name}}' "$nid" 2>/dev/null || true)"
        if [[ -n "$nname" && "$nname" != "bridge" && "$nname" != "host" && "$nname" != "none" ]]; then
          networks+="${nid}\n"
        fi
      done <<< "${all_nets}"
    fi
    volumes="$(docker volume ls -q || true)"
    images="$(docker images -q || true)"
  else
    # Scoped by STACK
    label1="label=com.docker.compose.project=${STACK_VAL}"
    label2="label=stack=${STACK_VAL}"
    c_name_pfx="${STACK_VAL}-"
    nv_name_pfx="${STACK_VAL}_"

    # Containers
    mapfile -t c_by_label1 < <(docker ps -a -q --filter "${label1}" || true)
    mapfile -t c_by_label2 < <(docker ps -a -q --filter "${label2}" || true)
    mapfile -t c_by_name   < <(docker ps -a --format '{{.ID}} {{.Names}}' | awk -v pfx="${c_name_pfx}" '$2 ~ "^"pfx {print $1}')
    containers="$(printf "%s\n%s\n%s\n" "${c_by_label1[@]:-}" "${c_by_label2[@]:-}" "${c_by_name[@]:-}" | unique)"

    # Networks
    mapfile -t n_by_label1 < <(docker network ls -q --filter "${label1}" || true)
    mapfile -t n_by_label2 < <(docker network ls -q --filter "${label2}" || true)
    mapfile -t n_by_name   < <(docker network ls --format '{{.ID}} {{.Name}}' | awk -v pfx="${nv_name_pfx}" '$2 ~ "^"pfx {print $1}')
    nets_raw="$(printf "%s\n%s\n%s\n" "${n_by_label1[@]:-}" "${n_by_label2[@]:-}" "${n_by_name[@]:-}" | unique)"
    if [[ -n "${nets_raw}" ]]; then
      while IFS= read -r nid; do
        [[ -z "$nid" ]] && continue
        nname="$(docker network inspect -f '{{.Name}}' "$nid" 2>/dev/null || true)"
        if [[ -n "$nname" && "$nname" != "bridge" && "$nname" != "host" && "$nname" != "none" ]]; then
          networks+="${nid}\n"
        fi
      done <<< "${nets_raw}"
    fi

    # Volumes (metadata-based + ACTUAL mounted volumes from matched containers)
    mapfile -t v_by_label1 < <(docker volume ls -q --filter "${label1}" || true)
    mapfile -t v_by_label2 < <(docker volume ls -q --filter "${label2}" || true)
    mapfile -t v_by_name   < <(docker volume ls --format '{{.Name}}' | awk -v pfx="${nv_name_pfx}" '$1 ~ "^"pfx {print $1}')
    v_by_mounts="$(volumes_from_containers "${containers}")"

    volumes="$(printf "%s\n%s\n%s\n%s\n" \
      "${v_by_label1[@]:-}" \
      "${v_by_label2[@]:-}" \
      "${v_by_name[@]:-}" \
      "${v_by_mounts:-}" \
      | unique)"

    # Images (scoped)
    images_sets=()

    if [[ "${IMAGES_MODE}" == "smart" || "${IMAGES_MODE}" == "associated" || "${IMAGES_MODE}" == "all" ]]; then
      # Images used by matched containers
      if [[ -n "${containers}" ]]; then
        while IFS= read -r cid; do
          [[ -z "$cid" ]] && continue
          img_id="$(docker inspect -f '{{.Image}}' "$cid" 2>/dev/null || true)"
          [[ -n "$img_id" ]] && images_sets+=("$img_id")
        done <<< "${containers}"
      fi
    fi

    if [[ "${IMAGES_MODE}" == "smart" || "${IMAGES_MODE}" == "labeled" || "${IMAGES_MODE}" == "all" ]]; then
      mapfile -t i_by_label_stack < <(docker images -q --filter "label=stack=${STACK_VAL}" || true)
      images_sets+=("${i_by_label_stack[@]:-}")
    fi

    if [[ "${IMAGES_MODE}" == "smart" || "${IMAGES_MODE}" == "tagged" || "${IMAGES_MODE}" == "all" ]]; then
      mapfile -t i_repo_pfx < <(docker images --format '{{.Repository}} {{.ID}}' | awk -v pfx="${STACK_VAL}" '$1 ~ "^"pfx {print $2}')
      mapfile -t i_tag_pfx  < <(docker images --format '{{.Tag}} {{.ID}}' | awk -v pfx="${STACK_VAL}" '$1 ~ "^"pfx {print $2}')
      images_sets+=("${i_repo_pfx[@]:-}")
      images_sets+=("${i_tag_pfx[@]:-}")
    fi

    if [[ "${IMAGES_MODE}" == "none" ]]; then
      images=""
    else
      images="$(printf "%s\n" "${images_sets[@]:-}" | unique)"
    fi
  fi

  print_plan

  # Confirm if doing a GLOBAL NUKE (only when not a dry-run)
  if [[ "${NUKE_ALL}" == 1 && "${DO_DRYRUN}" != "true" ]]; then
    read -r -p "Are you sure you want to nuke ALL projects? Type YES to confirm: " confirm
    if [[ "${confirm}" != "YES" ]]; then
      echo "Aborted."
      exit 1
    fi
  fi

  if [[ "${DO_DRYRUN}" == "true" ]]; then
    info "Dry-run selected; no changes made."
    exit 0
  fi

  # Execute deletions
  # 1) Containers
  if [[ -n "${containers}" ]]; then
    info "Removing containers..."
    while IFS= read -r id; do
      [[ -z "$id" ]] && continue
      docker rm -f "$id" || true
    done <<< "${containers}"
  else
    info "No containers to remove."
  fi

  # 2) Networks
  if [[ -n "${networks}" ]]; then
    info "Removing networks..."
    while IFS= read -r id; do
      [[ -z "$id" ]] && continue
      docker network rm "$id" || true
    done <<< "$(printf "%b" "${networks}")"
  else
    info "No networks to remove."
  fi

  # 3) Volumes
  if [[ -n "${volumes}" ]]; then
    info "Removing volumes..."
    while IFS= read -r name; do
      [[ -z "$name" ]] && continue
      docker volume rm -f "$name" || true
    done <<< "${volumes}"
  else
    info "No volumes to remove."
  fi

  # 4) Images
  if [[ -n "${images}" ]]; then
    info "Removing images..."
    while IFS= read -r id; do
      [[ -z "$id" ]] && continue
      docker rmi -f "$id" || true
    done <<< "${images}"
  else
    info "No images to remove."
  fi

    read -r -p "Do you also want to remove ALL unused Docker images (including old builds). Continue? [y/N] " reply
    case "$reply" in
    [yY]|[yY][eE][sS])
        echo "Pruning unused Docker images..."
        docker image prune -a
        ;;
    *)
        echo "No additional Docker images were removed."
        ;;
    esac

  ok "pm-container-nuke complete."
  exit 0
}

pm_setup_arg NUKE_ALL "--all" export
pm_setup_arg STACK_FROM_ARG "-s|--stack" needs_value export
pm_commit_args "$@"

main "$@"

#!/usr/bin/env bash
set -euo pipefail

: "${CUR_SCRIPT:=$(basename "${BASH_SOURCE[0]:-${0:-}}")}"
export CUR_SCRIPT
: "${PORTMASON_SHARE:="$(cd -- "$(dirname -- "${BASH_SOURCE[0]:-${0:-}}")" && pwd -P)"}"
. "${PORTMASON_SHARE}/pm-helpers" "$@"

deploy_images() {
    local compose_file="${1:-docker-compose.yml}"

    env_init_set DOCKER_BUILDKIT "${DOCKER_BUILDKIT:-0}"
    env_init_set COMPOSE_DOCKER_CLI_BUILD "${COMPOSE_DOCKER_CLI_BUILD:-0}"

    . "$SCRIPT_DIR/build-images"

    log "Resolving source images from ${compose_file}..."

    if [[ -z "${PROJECT_SERVICES:-}" ]]; then
        export PROJECT_SERVICES="$(
            pm-util-render-env "${compose_file}" \
            | yq -o=json '
                (.services // {})
                | to_entries
                | map(select(.value.image))
                | map({"service": .key, "image": .value.image})
                ' -
            )"
    fi

    while IFS=$'\t' read -r svc img; do
        [[ "${svc}" == "db" ]] || { deploy_image "${svc}" "${img}"; log "deployed image:${img} for service:${svc} "; }
    done < <(echo "$PROJECT_SERVICES" | jq -r '.[] | "\(.service)\t\(.image)"')

    return 0
}

deploy_image() {
    local svc="${1:?svc required}"
    local src_image="${2:?src_image required}"

    local service_name="${svc}" #"${STACK}-${DEPLOY_ENV}-${svc}"

    if [[ -n "${src_image:-}" ]]; then
        log "Deploying source image '$src_image' for service '$svc' from docker-compose …"

#        docker image inspect "$src_image" >/dev/null 2>&1 \
#            || docker pull "$src_image" \
#            || docker build -t "$src_image" .

        local tag
        tag="$(date -u +%Y%m%d-%H%M%S)"

        local target_image="${REGION}-docker.pkg.dev/${DEPLOYMENT_ID}/${AR_REPO}/${service_name}"
        local tagged_image="${target_image}:${tag}"

        log "Tagging image ${tagged_image}..."
        docker tag "${src_image}" "${tagged_image}"

        log "Pushing image to repository …"
        docker push --quiet "${tagged_image}" >/dev/null 2>&1 || die "docker push failed for ${tagged_image}"

        log "Deploying ${tagged_image}"
        ensure_deployment "$tagged_image" "$service_name" "service"
    fi
}

ensure_deployment() {
  local img="$1"
  local name="$2"
  local type="${3:-service}" # allowed: service, job

  local -a gcp_env_flags=()
  readarray -d '' -t gcp_env_flags < <(
    env_kv_for_injection | env_injection_args__gcp
  )

  if [[ "$type" == "service" ]]; then
    log "Deploying Cloud Run Service '$name'"

    local -a cmd=(
      gcloud run deploy "$name"
        --image "$img"
        --platform managed
        --region "$REGION"
        --allow-unauthenticated
        #--port "$PORT"
        --startup-probe "httpGet.path=/health,initialDelaySeconds=30,periodSeconds=10,timeoutSeconds=10,failureThreshold=30"
      "${gcp_env_flags[@]}"
    )

    log_cmd "${cmd[@]}"
    "${cmd[@]}" >/dev/null

    log "== DONE. Deployed service $name"
    log "Open the printed URL above. Full log at: ${LOG_FILE}"
    return 0
  fi

  if [[ "$type" == "job" ]]; then
    log "Ensuring/Updating Cloud Run Job '$name'"

    local action="update"
    if ! gcloud run jobs describe "$name" --region "$REGION" >/dev/null 2>&1; then
      action="create"
    fi

    local -a cmd=(
      gcloud run jobs "$action" "$name"
      --region "$REGION"
      --image "$img"
      "${gcp_env_flags[@]}"
    )

    log_cmd "${cmd[@]}"
    "${cmd[@]}" >/dev/null

    if [[ -n "${SCHED_SA:-}" ]]; then
      local -a bind_cmd=(
        gcloud run jobs add-iam-policy-binding "$name"
        --region "$REGION"
        --member "serviceAccount:${SCHED_SA}"
        --role "roles/run.developer"
      )

      log_cmd "${bind_cmd[@]}"
      "${bind_cmd[@]}" >/dev/null || true
    fi

    log "== DONE. Ensured job $name"
    log "Full log at: ${LOG_FILE}"
    return 0
  fi

  die "ensure_deployment: invalid type '$type' (expected service|job)"
}

main() {
    log "== ${CUR_SCRIPT} start =="
    log "stack: ${STACK}"
    log "GCS project: ${DEPLOYMENT_ID}"
    log "region: ${REGION}"
    log "log: ${LOG_FILE}"
    log

    : "${GCLOUD_PROJECT_INFO_JSON:="$(gcloud projects describe "${DEPLOYMENT_ID}" --format=json 2>/dev/null || echo '{}')"}"
    export GCLOUD_PROJECT_INFO_JSON

    if [[ -z "${GCLOUD_PROJECT_INFO_JSON:-}" || "${GCLOUD_PROJECT_INFO_JSON}" == "{}" ]]; then
        "${PORTMASON_SHARE}/pm-provision-$ADAPTER_CODE" "$@"
    else
        gcloud config set project "${DEPLOYMENT_ID}"
        log "Using project ${DEPLOYMENT_ID} that was created previously"
    fi

    deploy_images
}

main "$@"

===== pm-deploy-static =====
#!/usr/bin/env bash

# Static runtime deploy tasks.
#
# Responsibilities:
# - Render static HTML partials and executable hooks into deployable pages.
# - Keep deployed output fully static.
# - Preserve Portmason marker comments for repeatable/idempotent rendering.
# - Support page-specific partial tokens, including CASE_SUBTITLE.
# - Roll back the current HTML page if rendering that page fails.
#
# Marker model:
#
#   <!-- PM:HEADER -->
#   rendered content
#   <!-- /PM:HEADER -->
#
# Markers are managed regions. They remain in production HTML.
#
# Renderer resolution order for PM:<NAME>:
#   1. partials/hooks/<lowercase-name> when it is executable
#   2. partials/<lowercase-name>.html as the static fallback
#
# Page-specific token model:
#
#   Preferred page-level attribute:
#     <body data-case-subtitle="Integration Case Study">
#
#   Supported partial token:
#     {{CASE_SUBTITLE}}
#
#   Fallbacks for CASE_SUBTITLE:
#     1. <body data-case-subtitle="...">
#     2. content/case-studies.json entry matching the page URL or ID slug
#     3. "Architecture Case Study"

set -euo pipefail

: "${CUR_SCRIPT:=$(basename "${BASH_SOURCE[0]:-${0:-}}")}" 
export CUR_SCRIPT

: "${PORTMASON_SHARE:="$(cd -- "$(dirname -- "${BASH_SOURCE[0]:-${0:-}}")" && pwd -P)"}"
. "${PORTMASON_SHARE}/pm-helpers" "$@"

require_site_dir() {
    [[ -n "${SITE_DIR:-}" && -d "${SITE_DIR}" ]] || die "SITE_DIR is empty or not a directory: ${SITE_DIR:-<unset>}" 1
}

pm_partial_name_to_file() {
    local __out="$1"
    local partial_name="$2"

    local partial_slug=""
    partial_slug="$(printf '%s' "$partial_name" | tr '[:upper:]' '[:lower:]')"

    printf -v "$__out" '%s/partials/%s.html' "${SITE_DIR}" "$partial_slug"
}

pm_partial_name_to_hook() {
    local __out="$1"
    local partial_name="$2"

    local partial_slug=""
    partial_slug="$(printf '%s' "$partial_name" | tr '[:upper:]' '[:lower:]')"

    printf -v "$__out" '%s/partials/hooks/%s' "${SITE_DIR}" "$partial_slug"
}

pm_find_html_pages() {
    local site_dir="$1"
    local configured_excludes="${PM_STATIC_EXCLUDE_DIRS:-}"
    local exclude_dir=""
    local -a find_args=(
        "${site_dir}"
        -type f
        -name '*.html'
        -not -path '*/partials/*'
        -not -path '*/.git/*'
        -not -path '*/capture/*'
        -not -path '*/archive/*'
        -not -path '*/portmason/*'
        -not -path '*/tests/*'
        -not -path '*/deploy/*'
    )

    # Optional top-level or nested directory names may be added as a
    # comma- or whitespace-delimited list, for example:
    #   PM_STATIC_EXCLUDE_DIRS="capture,generated"
    configured_excludes="${configured_excludes//,/ }"
    for exclude_dir in ${configured_excludes}; do
        [[ -n "${exclude_dir}" ]] || continue

        case "${exclude_dir}" in
            /*|*'..'*|*'\\'*)
                die "Unsafe PM_STATIC_EXCLUDE_DIRS entry: ${exclude_dir}" 2
                ;;
        esac

        find_args+=( -not -path "*/${exclude_dir}/*" )
    done

    find "${find_args[@]}" -print
}

pm_extract_partial_names() {
    local page="$1"

    grep -Eoh '<!--[[:space:]]*PM:[A-Z0-9_-]+[[:space:]]*-->' "$page" 2>/dev/null \
        | sed -E 's/.*PM:([A-Z0-9_-]+).*/\1/' \
        | sort -u \
        || true
}

pm_count_partial_start_markers() {
    local __out="$1"
    local page="$2"
    local partial_name="$3"

    local marker_count=""
    marker_count="$(
        grep -Ec "<!--[[:space:]]*PM:${partial_name}[[:space:]]*-->" "$page" 2>/dev/null \
            || true
    )"

    printf -v "$__out" '%s' "$marker_count"
}

pm_count_partial_end_markers() {
    local __out="$1"
    local page="$2"
    local partial_name="$3"

    local marker_count=""
    marker_count="$(
        grep -Ec "<!--[[:space:]]*/PM:${partial_name}[[:space:]]*-->" "$page" 2>/dev/null \
            || true
    )"

    printf -v "$__out" '%s' "$marker_count"
}

verify_partial_markers_balanced() {
    local page="$1"
    local partial_name="$2"

    local start_count=""
    local end_count=""

    pm_count_partial_start_markers start_count "$page" "$partial_name"
    pm_count_partial_end_markers end_count "$page" "$partial_name"

    if [[ "$start_count" -eq 0 && "$end_count" -eq 0 ]]; then
        return 0
    fi

    if [[ "$start_count" -eq 0 ]]; then
        log ERROR "Missing start marker for ${partial_name} in ${page}"
        return 1
    fi

    if [[ "$end_count" -eq 0 ]]; then
        log ERROR "Missing end marker for ${partial_name} in ${page}"
        return 1
    fi

    if [[ "$start_count" -ne "$end_count" ]]; then
        log ERROR "Unbalanced partial markers for ${partial_name} in ${page}: start=${start_count}, end=${end_count}"
        return 1
    fi

    return 0
}

pm_make_page_backup() {
    local __out="$1"
    local page="$2"

    local tmp_backup_file=""
    local rc=0
    tmp_backup_file="$(mktemp)"

    if cp -p "$page" "$tmp_backup_file"; then
        :
    else
        rc=$?
        rm -f "$tmp_backup_file"
        log ERROR "Failed to create rollback backup for ${page}; rc=${rc}"
        return "$rc"
    fi

    printf -v "$__out" '%s' "$tmp_backup_file"
}

pm_restore_page_backup() {
    local page="$1"
    local backup_file="${2:-}"

    if [[ -n "$backup_file" && -f "$backup_file" ]]; then
        cp -p "$backup_file" "$page"
        log WARN "Rolled back ${page} from ${backup_file}"
    else
        log ERROR "Rollback backup missing for ${page}: ${backup_file:-<unset>}"
    fi
}

pm_discard_page_backup() {
    local backup_file="${1:-}"

    [[ -n "$backup_file" ]] && rm -f "$backup_file"
}

pm_html_unescape_minimal() {
    local __out="$1"
    local input_value="${2:-}"
    local decoded_value="$input_value"

    decoded_value="${decoded_value//&quot;/\"}"
    decoded_value="${decoded_value//&#34;/\"}"
    decoded_value="${decoded_value//&apos;/\'}"
    decoded_value="${decoded_value//&#39;/\'}"
    decoded_value="${decoded_value//&amp;/&}"
    decoded_value="${decoded_value//&lt;/<}"
    decoded_value="${decoded_value//&gt;/>}"

    printf -v "$__out" '%s' "$decoded_value"
}

pm_extract_body_data_attr() {
    local __out="$1"
    local page="$2"
    local attr_name="$3"

    local raw_attr_value=""
    local decoded_attr_value=""

    raw_attr_value="$(
        awk -v attr="${attr_name}" '
            BEGIN { IGNORECASE = 1 }
            /<body([[:space:]>]|$)/ {
                body = $0
                while (body !~ />/ && (getline next_line) > 0) {
                    body = body " " next_line
                }

                pattern_dq = attr "[[:space:]]*=[[:space:]]*\"[^\"]*\""
                pattern_sq = attr "[[:space:]]*=[[:space:]]*'\''[^'\'']*'\''"

                if (match(body, pattern_dq)) {
                    found = substr(body, RSTART, RLENGTH)
                    sub(/^[^=]*=[[:space:]]*\"/, "", found)
                    sub(/\"$/, "", found)
                    print found
                    exit
                }

                if (match(body, pattern_sq)) {
                    found = substr(body, RSTART, RLENGTH)
                    sub(/^[^=]*=[[:space:]]*'\''/, "", found)
                    sub(/'\''$/, "", found)
                    print found
                    exit
                }
            }
        ' "$page" 2>/dev/null || true
    )"

    pm_html_unescape_minimal decoded_attr_value "$raw_attr_value"
    printf -v "$__out" '%s' "$decoded_attr_value"
}

pm_json_case_subtitle_for_page() {
    local __out="$1"
    local page="$2"

    local json_file="${SITE_DIR}/content/case-studies.json"
    local json_subtitle=""

    if [[ ! -f "$json_file" ]]; then
        printf -v "$__out" '%s' ""
        return 0
    fi

    if command -v python3 >/dev/null 2>&1; then
        json_subtitle="$(python3 -c '
import json
import os
import sys

site_dir, page, json_file = sys.argv[1:4]
rel_page = os.path.relpath(page, site_dir).replace(os.sep, "/")
page_slug = os.path.splitext(os.path.basename(page))[0]

try:
    with open(json_file, "r", encoding="utf-8") as fh:
        items = json.load(fh)
except Exception:
    sys.exit(0)

if not isinstance(items, list):
    sys.exit(0)

for item in items:
    if not isinstance(item, dict):
        continue
    url = str(item.get("url", "")).lstrip("/")
    item_id = str(item.get("id", ""))
    tag = str(item.get("tag", ""))

    if tag and (url == rel_page or item_id == page_slug):
        print(tag)
        break
' "$SITE_DIR" "$page" "$json_file" 2>/dev/null || true)"
    fi

    printf -v "$__out" '%s' "$json_subtitle"
}

pm_page_case_subtitle() {
    local __out="$1"
    local page="$2"

    local subtitle_value=""

    pm_extract_body_data_attr subtitle_value "$page" "data-case-subtitle"

    if [[ -z "$subtitle_value" ]]; then
        pm_json_case_subtitle_for_page subtitle_value "$page"
    fi

    if [[ -z "$subtitle_value" ]]; then
        subtitle_value="Architecture Case Study"
    fi

    printf -v "$__out" '%s' "$subtitle_value"
}

render_partials() {
    require_site_dir

    if [[ ! -d "${SITE_DIR}/partials" ]]; then
        warn "No partials directory found: ${SITE_DIR}/partials"
        return 0
    fi

    local page=""
    local rc=0

    while IFS= read -r page; do
        [[ -f "$page" ]] || continue

        if render_page_partials "$page"; then
            continue
        else
            rc=$?
            die "Static partial rendering failed for ${page}; file was rolled back." "$rc"
        fi
    done < <(pm_find_html_pages "${SITE_DIR}")
}

render_page_partials() {
    local page="$1"

    if [[ ! -f "$page" ]]; then
        log ERROR "Page not found while rendering partials: ${page}"
        return 1
    fi

    local backup_file=""
    local partial_name=""
    local partial_file=""
    local partial_hook=""
    local rc=0

    if ! pm_make_page_backup backup_file "$page"; then
        return 1
    fi

    while IFS= read -r partial_name; do
        [[ -n "$partial_name" ]] || continue

        if verify_partial_markers_balanced "$page" "$partial_name"; then
            :
        else
            rc=$?
            break
        fi

        pm_partial_name_to_file partial_file "$partial_name"
        pm_partial_name_to_hook partial_hook "$partial_name"

        if render_partial "$page" "$partial_name" "$partial_file" "$partial_hook"; then
            :
        else
            rc=$?
            break
        fi
    done < <(pm_extract_partial_names "$page")

    if [[ "$rc" -ne 0 ]]; then
        pm_restore_page_backup "$page" "$backup_file"
        pm_discard_page_backup "$backup_file"
        return "$rc"
    fi

    if verify_page_partial_markers_balanced "$page"; then
        :
    else
        rc=$?
        pm_restore_page_backup "$page" "$backup_file"
        pm_discard_page_backup "$backup_file"
        return "$rc"
    fi

    pm_discard_page_backup "$backup_file"
    return 0
}

render_partial() {
    local page="$1"
    local partial_name="$2"
    local partial_file="$3"
    local partial_hook="$4"

    if [[ ! -f "$page" ]]; then
        log ERROR "Page not found while rendering partial: ${page}"
        return 1
    fi

    local partial_source="$partial_file"
    local partial_source_kind="partial"
    local generated_source=""
    local hook_rc=0

    if [[ -f "$partial_hook" && -x "$partial_hook" ]]; then
        generated_source="$(mktemp)"

        if (
            cd "$SITE_DIR"
            SITE_DIR="$SITE_DIR" "$partial_hook"
        ) > "$generated_source"; then
            partial_source="$generated_source"
            partial_source_kind="hook"
        else
            hook_rc=$?
            rm -f "$generated_source"
            log ERROR "Hook failed for ${partial_name}: ${partial_hook}; rc=${hook_rc}"
            return "$hook_rc"
        fi
    elif [[ ! -f "$partial_file" ]]; then
        if [[ -f "$partial_hook" ]]; then
            log ERROR "Hook is not executable and no static partial exists for ${partial_name}: hook=${partial_hook}, partial=${partial_file}"
        else
            log ERROR "No hook or partial found for ${partial_name}: hook=${partial_hook}, partial=${partial_file}"
        fi
        return 1
    fi

    local case_subtitle=""
    pm_page_case_subtitle case_subtitle "$page"

    local tmp_file=""
    tmp_file="$(mktemp)"

    if awk \
        -v partial_name="$partial_name" \
        -v partial_file="$partial_source" \
        -v case_subtitle="$case_subtitle" '
        function render_template(input, output) {
            output = input
            gsub(/\{\{CASE_SUBTITLE\}\}/, case_subtitle, output)
            return output
        }

        BEGIN {
            start_marker = "<!-- PM:" partial_name " -->"
            end_marker = "<!-- /PM:" partial_name " -->"
            in_block = 0
            rendered = 0

            while ((getline partial_line < partial_file) > 0) {
                partial = partial render_template(partial_line) ORS
            }
            close(partial_file)
        }

        $0 ~ "<!--[[:space:]]*PM:" partial_name "[[:space:]]*-->" {
            print start_marker
            printf "%s", partial
            print end_marker
            in_block = 1
            rendered++
            next
        }

        $0 ~ "<!--[[:space:]]*/PM:" partial_name "[[:space:]]*-->" {
            in_block = 0
            next
        }

        in_block == 0 {
            print
        }

        END {
            if (rendered == 0) {
                exit 10
            }
        }
        ' "$page" > "$tmp_file"; then
        :
    else
        local rc=$?
        rm -f "$tmp_file" "$generated_source"

        case "$rc" in
            10)
                log ERROR "Failed to render ${partial_name}; no start marker processed in ${page}"
                ;;
            *)
                log ERROR "Failed to render ${partial_name} into ${page}; rc=${rc}"
                ;;
        esac

        return "$rc"
    fi

    rm -f "$generated_source"

    if mv "$tmp_file" "$page"; then
        chmod 0644 "$page" || true
    else
        local rc=$?
        rm -f "$tmp_file"
        log ERROR "Failed to replace rendered page: ${page}; rc=${rc}"
        return "$rc"
    fi

    log INFO "Rendered ${partial_source_kind} ${partial_name} into ${page}"
    return 0
}

verify_page_partial_markers_balanced() {
    local page="$1"
    local partial_name=""
    local rc=0

            break

}

verify_all_partial_markers_balanced() {

    local page=""
    local rc=0

    while IFS= read -r page; do
        [[ -f "$page" ]] || continue

        if verify_page_partial_markers_balanced "$page"; then

        else

            break

    done < <(pm_find_html_pages "${SITE_DIR}")

    return "$rc"
}

pm_deploy_static_main() {
    require_site_dir

    log INFO "pm-deploy-static: verifying partial markers"
    local rc=0
    if verify_all_partial_markers_balanced; then
        :
    else
        rc=$?
        die "Static partial marker verification failed before rendering." "$rc"

    log INFO "pm-deploy-static: rendering partials"

    log INFO "pm-deploy-static: verifying rendered partial markers"

        :
    else
        rc=$?
        die "Static partial marker verification failed after rendering." "$rc"
    fi

    log INFO "pm-deploy-static: complete"
}

pm_deploy_static_main "$@"

mask_value() {
  local s="${1-}"
  local n="${#s}"
  if (( n <= 6 )); then
    printf '******'
    return 0
  fi
  printf '%s******%s' "${s:0:2}" "${s: -2}"
}

require_cmd() {
  local cmd="$1"
  command -v "$cmd" >/dev/null 2>&1 || die "Missing required command: $cmd"
}

print_title() {
  echo
  echo "================================================================================"
  echo "$*"
  echo "================================================================================"
}

usage() {
  cat <<USAGE
Usage:
  ${CUR_SCRIPT} --image-tag <name> --port <port> [--env-file <path>] [--cloudrun-args "<args>"] [--docker-user <uid:gid>]

Defaults:
  --image-tag     pm-cloudrun-emulator
  --port          8080
  --env-file      .env
  --docker-user   1000:1000

Notes:
  - This script builds a local image and runs it in a way that resembles Cloud Run:
    * read-only filesystem
    * writable /tmp via tmpfs
    * uses PORT env (default 8080)
    * injects env vars from an env file
USAGE
}

# Build env inject args for docker run
build_env_inject_args() {
  local env_file="$1"
  ENV_INJECT_ARGS=()
  if [[ -f "$env_file" ]]; then
    while IFS='=' read -r k v; do
      [[ -z "${k}" ]] && continue
      [[ "${k:0:1}" == "#" ]] && continue
      # allow values containing '=' by re-splitting from original line
      local line
      line="$(grep -E "^${k}=" "$env_file" | head -n1 || true)"
      [[ -z "$line" ]] && continue
      v="${line#*=}"
      # trim possible CRLF
      v="${v%$'\r'}"
      ENV_INJECT_ARGS+=( -e "${k}=${v}" )
    done < <(grep -E '^[A-Za-z_][A-Za-z0-9_]*=' "$env_file" || true)
  fi
}

    # else
    log INFO "Fetching runtime artifact" "src=$RUNTIME_ARTIFACT" "dst=$out_path"

    if [[ "$RUNTIME_ARTIFACT" == gs://* ]]; then
        command -v gcloud >/dev/null 2>&1 || die "gcloud is required to fetch: $RUNTIME_ARTIFACT"
        gcloud storage cp "$RUNTIME_ARTIFACT" "$out_path" >/dev/null
    elif [[ "$RUNTIME_ARTIFACT" == http://* || "$RUNTIME_ARTIFACT" == https://* ]]; then
        command -v curl >/dev/null 2>&1 || die "curl is required to fetch: $RUNTIME_ARTIFACT"
        curl -fsSL "$RUNTIME_ARTIFACT" -o "$out_path"
    else
        [[ "$RUNTIME_ARTIFACT" == /* ]] || die "Local artifact paths must be absolute: $RUNTIME_ARTIFACT"
        [[ -f "$RUNTIME_ARTIFACT" ]] || die "Artifact not found: $RUNTIME_ARTIFACT"
        cp -f "$RUNTIME_ARTIFACT" "$out_path"
    fi
    #fi

    # Integrity + info
    if command -v sha256sum >/dev/null 2>&1; then
        sha256sum "$out_path" | awk '{print $1}' > "${out_path}.sha256"
        sha="$(cat "${out_path}.sha256" 2>/dev/null || true)"
    else
        sha=""
    fi

    if command -v stat >/dev/null 2>&1; then
        size="$(stat -c '%s' "$out_path" 2>/dev/null || stat -f '%z' "$out_path" 2>/dev/null || echo "")"
    else
        size=""
    fi

    log INFO "Artifact ready" "path=$(pwd -P)/${out_path#./}" "sha256=${sha:-unknown}" "bytes=${size:-unknown}"
}

main(){
    # Ensure artifact exists (repeatable)
    fetch_runtime_artifact

    [[ -f "$HOST_ARTIFACT_PATH" ]] || die "Expected artifact missing after fetch: $HOST_ARTIFACT_PATH"

    abs_art_dir="$(cd "$HOST_ARTIFACT_DIR" && pwd -P)"
    emu_tmp_size="${EMU_TMP_SIZE:-512m}"

    # Extra env injection (optional)
    extra_env_args=()
    if [[ -n "${EMU_EXTRA_ENV:-}" ]]; then
    # shellcheck disable=SC2206
    tokens=( ${EMU_EXTRA_ENV} )
    for t in "${tokens[@]}"; do
        extra_env_args+=( -e "$t" )
    done
    fi

        # shellcheck disable=SC2206
        extra_docker_args=( ${EMU_EXTRA_DOCKER_ARGS} )
    fi

    # Prefer Cloud Run-ish logging behavior (stdout/stderr)

    # Override RUNTIME_ARTIFACT directly so container never touches gs:// and never calls gcloud.
    export CONTAINER_ARGS="\
        -v ${abs_art_dir}:${CONTAINER_ARTIFACT_DIR}:ro \

===== pm-extract-files-from-tar =====
#!/usr/bin/env bash
# Purpose: Safely extract a Portmason tar.gz artifact into a repository directory.
# Usage:
#   pm-extract-files-from-tar ARCHIVE.tar.gz [chg.txt] [TARGET_DIR]
#   pm-extract-files-from-tar --all ARCHIVE.tar.gz [TARGET_DIR]
#
# Contract:
# - With an explicit change list, only the listed files are extracted.
# - With only ARCHIVE.tar.gz, ./chg.txt is used automatically when present.
# - With only ARCHIVE.tar.gz and no ./chg.txt, the complete project/ tree is extracted.
# - --all forces complete extraction even when ./chg.txt exists.
# - chg.txt entries are repo-relative paths beginning with project/.
# - blank lines and comments beginning with # are ignored.
# - the leading project/ prefix is stripped when copying into TARGET_DIR.
# - files are staged before they are copied into the target.

set -euo pipefail

: "${CUR_SCRIPT:=$(basename "${BASH_SOURCE[0]:-${0:-}}")}" 
export CUR_SCRIPT

ensure_portmason_share() {
  local script_dir default
  script_dir="$(cd -- "$(dirname -- "${BASH_SOURCE[0]}")" && pwd -P)"
  default="/usr/local/share/portmason/bin"

  if [[ -n "${PORTMASON_SHARE:-}" && -r "${PORTMASON_SHARE}/pm-helpers" ]]; then
    export PORTMASON_SHARE
    return 0
  fi

  if [[ -r "${script_dir}/pm-helpers" ]]; then
    PORTMASON_SHARE="${script_dir}"
    export PORTMASON_SHARE
    return 0
  fi

  if [[ -r "${default}/pm-helpers" ]]; then
    PORTMASON_SHARE="${default}"
    export PORTMASON_SHARE
    return 0
  fi

  printf 'ERROR: unable to locate pm-helpers. Set PORTMASON_SHARE or install this script beside pm-helpers.\n' >&2
  return 2
}

ensure_portmason_share
# shellcheck disable=SC1090
. "${PORTMASON_SHARE}/pm-helpers" --minimal

usage() {
  cat >&2 <<'USAGE'
Usage:
  pm-extract-files-from-tar ARCHIVE.tar.gz [chg.txt] [TARGET_DIR]
  pm-extract-files-from-tar --all ARCHIVE.tar.gz [TARGET_DIR]

Arguments:
  ARCHIVE.tar.gz  Portmason change-only archive or captured-state archive.
  chg.txt         Optional finite list of files to extract. Entries must begin with project/.
  TARGET_DIR      Destination repository directory. Defaults to the current directory.

Default behavior:
  - When only ARCHIVE.tar.gz is provided and ./chg.txt exists, extract only files listed in ./chg.txt.
  - When only ARCHIVE.tar.gz is provided and ./chg.txt does not exist, extract the complete project/ tree.
  - When an explicit chg.txt argument is provided but does not exist, fail loudly.
  - Use --all to force complete extraction even when ./chg.txt exists.

chg.txt format:
  project/index.html
  project/assets/css/styles.css

  # Blank lines and comments are ignored.
USAGE
}

cleanup() {
  if [[ -n "${STAGE_DIR:-}" && -d "${STAGE_DIR}" ]]; then
    rm --recursive --force -- "${STAGE_DIR}"
  fi
}
trap cleanup EXIT INT TERM

normalize_listed_path() {
  local out_var="${1:?output variable required}"
  local raw="${2-}"
  local candidate

  # Strip a trailing CR so lists created on Windows remain usable.
  candidate="${raw%$'\r'}"

  # Ignore blank lines and comments, including comments with leading whitespace.
  if [[ -z "${candidate//[[:space:]]/}" || "${candidate}" =~ ^[[:space:]]*# ]]; then
    printf -v "${out_var}" '%s' ''
    return 0
  fi

  if [[ "${candidate}" =~ ^[[:space:]] || "${candidate}" =~ [[:space:]]$ ]]; then
    fatal_error "Invalid chg.txt entry with leading or trailing whitespace: ${candidate}" 2
  fi

  if [[ "${candidate}" != project/* ]]; then
    fatal_error "Invalid chg.txt entry; expected project/... path: ${candidate}" 2
  fi

  local relative="${candidate#project/}"
  if [[ -z "${relative}" ]]; then
    fatal_error "Invalid chg.txt entry; project/ must identify a file." 2
  fi

  case "/${relative}/" in
    *'/../'*|*'/./'*|*'//'*)
      fatal_error "Unsafe chg.txt entry: ${candidate}" 2
      ;;
  esac

  case "${relative}" in
    /*|*\\*)
      fatal_error "Unsafe chg.txt entry: ${candidate}" 2
      ;;
  esac

  printf -v "${out_var}" '%s' "${candidate}"
}

validate_archive_member_path() {
  local member="${1:?archive member required}"

  case "/${member}/" in
    *'/../'*|*'/./'*|*'//'*)
      fatal_error "Unsafe archive member path: ${member}" 3
      ;;
  esac

  case "${member}" in
    /*|*\\*)
      fatal_error "Unsafe archive member path: ${member}" 3
      ;;
  esac
}

normalize_archive_member() {
  local out_var="${1:?output variable required}"
  local member="${2:?archive member required}"
  local computed

  validate_archive_member_path "${member}"

  case "${member}" in
    project/*)
      computed="${member}"
      ;;
    */project/*)
      computed="project/${member#*/project/}"
      ;;
    *)
      fatal_error "Archive member is outside the expected project/ tree: ${member}" 3
      ;;
  esac

  if [[ "${computed}" == 'project/' || -z "${computed#project/}" ]]; then
    fatal_error "Archive member does not identify a file: ${member}" 3
  fi

  printf -v "${out_var}" '%s' "${computed}"
}

resolve_archive_member() {
  local out_var="${1:?output variable required}"
  local listed_path="${2:?listed path required}"
  local archive_member match_count=0 match=''

  for archive_member in "${ARCHIVE_MEMBERS[@]}"; do
    if [[ "${archive_member}" == "${listed_path}" || "${archive_member}" == */"${listed_path}" ]]; then
      match="${archive_member}"
      ((match_count += 1))
    fi
  done

  if (( match_count == 0 )); then
    fatal_error "Archive does not contain listed file: ${listed_path}" 3
  fi

  if (( match_count > 1 )); then
    fatal_error "Archive contains multiple matches for listed file: ${listed_path}" 3
  fi

  validate_archive_member_path "${match}"
  printf -v "${out_var}" '%s' "${match}"
}

validate_destination() {
  local target_root="${1:?target root required}"
  local destination="${2:?destination required}"
  local resolved

  resolved="$(realpath --canonicalize-missing -- "${destination}")"
  case "${resolved}" in
    "${target_root}"|"${target_root}"/*)
      ;;
    *)
      fatal_error "Destination escapes target directory: ${destination}" 4
      ;;
  esac
}

assert_regular_archive_member() {
  local archive="${1:?archive required}"
  local member="${2:?archive member required}"
  local verbose first_char

  verbose="$(tar --list --verbose --gzip --file "${archive}" -- "${member}")"
  first_char="${verbose:0:1}"

  case "${first_char}" in
    '-')
      ;;
    'd')
      fatal_error "Archive member is a directory, not a file: ${member}" 3
      ;;
    *)
      fatal_error "Refusing non-regular archive member: ${member}" 3
      ;;
  esac
}

copy_archive_member() {
  local archive="${1:?archive required}"
  local target_root="${2:?target root required}"
  local member="${3:?archive member required}"
  local normalized="${4:?normalized path required}"
  local relative staged destination

  relative="${normalized#project/}"
  assert_regular_archive_member "${archive}" "${member}"

  tar --extract --gzip --file "${archive}" --directory "${STAGE_DIR}" -- "${member}"
  staged="${STAGE_DIR}/${member}"

  [[ -f "${staged}" ]] || fatal_error "Selected archive member is not a regular file: ${member}" 3
  [[ ! -L "${staged}" ]] || fatal_error "Refusing symlink archive member: ${member}" 3

  destination="${target_root}/${relative}"
  validate_destination "${target_root}" "${destination}"
  mkdir --parents -- "$(dirname -- "${destination}")"
  cp --preserve=mode,timestamps -- "${staged}" "${destination}"

  log INFO "Extracted: ${normalized} -> ${relative}"
}

extract_selected_files() {
  local archive="${1:?archive required}"
  local change_list="${2:?change list required}"
  local target_root="${3:?target root required}"
  local listed normalized member
  local extracted_count=0
  declare -A seen=()

  while IFS= read -r listed || [[ -n "${listed}" ]]; do
    normalize_listed_path normalized "${listed}"
    [[ -n "${normalized}" ]] || continue

    if [[ -n "${seen[${normalized}]+x}" ]]; then
      warn "Skipping duplicate chg.txt entry: ${normalized}"
      continue
    fi
    seen["${normalized}"]=1

    resolve_archive_member member "${normalized}"
    copy_archive_member "${archive}" "${target_root}" "${member}" "${normalized}"
    ((extracted_count += 1))
  done < "${change_list}"

  (( extracted_count > 0 )) || fatal_error "No extractable files found in ${change_list}." 2
  log INFO "Complete: extracted ${extracted_count} selected file(s) into ${target_root}."
}

extract_all_files() {
  local archive="${1:?archive required}"
  local target_root="${2:?target root required}"
  local member normalized
  local extracted_count=0
  declare -A seen=()

  for member in "${ARCHIVE_MEMBERS[@]}"; do
    # Directory entries are created on demand while copying regular files.
    [[ "${member}" == */ ]] && continue

    normalize_archive_member normalized "${member}"

    if [[ -n "${seen[${normalized}]+x}" ]]; then
      fatal_error "Archive contains multiple members for repository path: ${normalized}" 3
    fi
    seen["${normalized}"]=1

    copy_archive_member "${archive}" "${target_root}" "${member}" "${normalized}"
    ((extracted_count += 1))
  done

  (( extracted_count > 0 )) || fatal_error "Archive contains no extractable files: ${archive}" 3
  log INFO "Complete: extracted ${extracted_count} file(s) into ${target_root}."
}

main() {
  local extract_mode='auto'
  local archive change_list='' target_dir='.' target_root

  if [[ "${1:-}" == '--help' || "${1:-}" == '-h' ]]; then
    usage
    return 0
  fi

  if [[ "${1:-}" == '--all' ]]; then
    extract_mode='all'
    shift
    if (( $# < 1 || $# > 2 )); then
      usage
      return 2
    fi
    archive="${1}"
    target_dir="${2:-.}"
  else
    if (( $# < 1 || $# > 3 )); then
      usage
      return 2
    fi

    archive="${1}"
    if (( $# >= 2 )); then
      extract_mode='selected'
      change_list="${2}"
      target_dir="${3:-.}"
    elif [[ -f './chg.txt' ]]; then
      extract_mode='selected'
      change_list='./chg.txt'
    else
      extract_mode='all'
    fi
  fi

  have tar || fatal_error "tar is required." 2
  have mktemp || fatal_error "mktemp is required." 2
  have realpath || fatal_error "realpath is required." 2

  [[ -f "${archive}" ]] || fatal_error "Archive not found: ${archive}" 2
  if [[ "${extract_mode}" == 'selected' ]]; then
    [[ -f "${change_list}" ]] || fatal_error "Change list not found: ${change_list}" 2
  fi

  mkdir --parents -- "${target_dir}"
  target_root="$(realpath --canonicalize-missing -- "${target_dir}")"
  STAGE_DIR="$(mktemp --directory --tmpdir pm-extract-files-from-tar.XXXXXXXX)"

  mapfile -t ARCHIVE_MEMBERS < <(tar --list --gzip --file "${archive}")
  (( ${#ARCHIVE_MEMBERS[@]} > 0 )) || fatal_error "Archive is empty: ${archive}" 3

  log INFO "Archive:      ${archive}"
  log INFO "Mode:         ${extract_mode}"
  if [[ "${extract_mode}" == 'selected' ]]; then
    log INFO "Change list:  ${change_list}"
  fi
  log INFO "Target:       ${target_root}"

  case "${extract_mode}" in
    selected)
      extract_selected_files "${archive}" "${change_list}" "${target_root}"
      ;;
    all)
      extract_all_files "${archive}" "${target_root}"
      ;;
    *)
      fatal_error "Unsupported extraction mode: ${extract_mode}" 2
      ;;
  esac
}

main "$@"

#!/usr/bin/env bash
set -euo pipefail

: "${CUR_SCRIPT:=$(basename "${BASH_SOURCE[0]:-${0:-}}")}"
export CUR_SCRIPT
: "${PORTMASON_SHARE:="$(cd -- "$(dirname -- "${BASH_SOURCE[0]:-${0:-}}")" && pwd -P)"}"
. "${PORTMASON_SHARE}/pm-helpers" "$@"


# pm-gen-sslcert
# Generates a dev TLS cert/key pair.
# Prefers mkcert when available; falls back to openssl self-signed.
#
# Outputs (in -o OUTDIR):
#   <stem>.fullchain.crt
#   <stem>.key
#
# Example:
#   pm-gen-sslcert -o /path/to/certs --cn movies-dev.localtest.me -f
#   pm-gen-sslcert -o /path/to/certs --cn '*.localtest.me' -f

usage() {
  cat <<USAGE
Usage: pm-gen-sslcert -o OUTDIR -cn CN [-f]

Options:
  -o OUTDIR       Output directory (required)
  -cn CN          Common Name / hostname (required). May be wildcard (*.example.com)
  -f              Force overwrite if files exist
  -h,--help       Show help

Notes:
  - If mkcert is installed, a locally-trusted cert is generated (after 'mkcert -install').
  - If mkcert is not installed, openssl generates a self-signed cert (browser warning).
  - Output filenames are derived from CN:
      '*' -> 'wildcard', spaces -> '_', '/' -> '_'
USAGE
}

OUTDIR=
CN=
FORCE="no"

while [[ $# -gt 0 ]]; do
  case "$1" in
    -o)
      OUTDIR="${2:-}"; shift 2;;
    -cn)
      CN="${2:-}"; shift 2;;
    -f)
      FORCE="yes"; shift;;
    -h|--help)
      usage; exit 0;;
    *)
      die "Unknown arg: $1 (use --help)";;
  esac
done

[[ -n "$OUTDIR" ]] || die "-o OUTDIR is required"
[[ -n "$CN" ]] || die "--cn CN is required"

mkdir -p "$OUTDIR"

STEM="$(echo "$CN" | sed 's/\*/wildcard/g; s/[[:space:]]\+/_/g; s#/#_#g')"
CRT="$OUTDIR/$STEM.fullchain.crt"
KEY="$OUTDIR/$STEM.key"

if [[ -f "$CRT" || -f "$KEY" ]]; then
  if [[ "$FORCE" != "yes" ]]; then
    die "Output files exist. Use -f to overwrite:
  $CRT
  $KEY"
  fi
fi

# If mkcert is available, use it (best dev experience)
if command -v mkcert >/dev/null 2>&1; then
  # Ensure local CA installed (idempotent). Don't fail hard if it prints warnings.
  mkcert -install >/dev/null 2>&1 || true

  # mkcert requires hostname args; support wildcard CN:
  # If CN starts with '*.', generate for both base + wildcard so the cert is broadly useful.
  if [[ "$CN" == \*.* ]]; then
    BASE="${CN#*.}"
    mkcert -cert-file "$CRT" -key-file "$KEY" "$BASE" "$CN" >/dev/null
  else
    mkcert -cert-file "$CRT" -key-file "$KEY" "$CN" >/dev/null
  fi

  [[ -s "$CRT" && -s "$KEY" ]] || die "mkcert did not produce expected outputs:
  $CRT
  $KEY"

  echo "Generated mkcert cert:"
  echo "  cert: $CRT"
  echo "  key : $KEY"
  exit 0
fi

# Fallback: openssl self-signed
if ! command -v openssl >/dev/null 2>&1; then
  die "Neither mkcert nor openssl found. Install one of them and rerun."
fi

# Build SAN list:
# - Always include CN
# - If wildcard, include base domain too (handy in browsers)
SAN_ITEMS="DNS:$CN"
if [[ "$CN" == \*.* ]]; then
  BASE="${CN#*.}"
  SAN_ITEMS="DNS:$BASE,DNS:$CN"
fi

TMPDIR="$(mktemp -d)"
trap 'rm -rf "$TMPDIR"' EXIT

cat > "$TMPDIR/openssl.cnf" <<CONF
[ req ]
default_bits       = 2048
prompt             = no
default_md         = sha256
distinguished_name = dn
x509_extensions    = v3_req

[ dn ]
CN = ${CN}

[ v3_req ]
subjectAltName = ${SAN_ITEMS}
keyUsage = digitalSignature, keyEncipherment
extendedKeyUsage = serverAuth
CONF

openssl req -x509 -nodes -newkey rsa:2048 \
  -days 825 \
  -keyout "$KEY" \
  -out "$CRT" \
  -config "$TMPDIR/openssl.cnf" >/dev/null 2>&1

[[ -s "$CRT" && -s "$KEY" ]] || die "openssl did not produce expected outputs:
  $CRT
  $KEY"

echo "Generated self-signed openssl cert (browser will warn unless you trust it manually):"
echo "  cert: $CRT"
echo "  key : $KEY"

#!/usr/bin/env bash

main() {
  local compose_file="${1:-docker-compose.yml}"
  local env_file=".env.generated"

  have docker || fatal_error "docker is not installed or not on PATH"
  have jq     || fatal_error "jq is not installed or not on PATH"
  [[ -f "${compose_file}" ]] || fatal_error "${compose_file} not found in $(pwd)"

  local -a base_cmd=(docker compose -f "${compose_file}" config)
  local -a env_flag=()

  # If compose supports --env-file and .env.generated exists, use it for interpolation.
  if [[ -f "${env_file}" ]] && docker compose --help 2>/dev/null | grep -q -- '--env-file'; then
    env_flag=(--env-file "${env_file}")
  fi

  # Prefer JSON output if supported.
  if docker compose "${env_flag[@]}" -f "${compose_file}" config --help 2>/dev/null | grep -q -- '--format'; then
    # Some versions support --format but not json; try and fall back if it fails.
    if docker compose "${env_flag[@]}" -f "${compose_file}" config --format json >/dev/null 2>&1; then
      docker compose "${env_flag[@]}" -f "${compose_file}" config --format json \
        | jq -c '
            .services
            | to_entries
            | map(select(.value.image))
            | map({"service": .key, "image": .value.image})
          '
      return 0
    fi
  fi

  # Fallback: YAML output + yq parsing
  have yq || fatal_error "yq is required for fallback parsing (docker compose config --format json not available)"

  docker compose "${env_flag[@]}" -f "${compose_file}" config \
    | yq eval -o=json '
        (.services // {})
        | to_entries
        | map(select(.value.image))
        | map({"service": .key, "image": .value.image})
      ' -
}

main "$@"

archive_log_on_session_start() {
  # Usage: archive_log_on_session_start "$LOG_FILE"
  #
  # Behavior:
  # - If LOG_FILE exists and non-empty: append to LOG_FILE._archive, then truncate LOG_FILE.
  # - If LOG_FILE._archive exceeds PM_LOG_ARCHIVE_MAX_BYTES: rotate it to numbered suffixes.
  #
  # Knobs:
  #   PM_LOG_ARCHIVE_MAX_BYTES (default 10 MiB)
  #   PM_LOG_ARCHIVE_KEEP      (default 5)

  local log_file="${1:?log_file required}"
  local archive="${log_file}_archive"

  local max_bytes="${PM_LOG_ARCHIVE_MAX_BYTES:-10485760}" # 10 MiB
  local keep="${PM_LOG_ARCHIVE_KEEP:-5}"

  [[ -e "$log_file" ]] || return 0
  [[ -s "$log_file" ]] || return 0

  # Ensure archive exists (best-effort)
  : > "$archive" 2>/dev/null || true

  # Append current into archive (best-effort)
  cat "$log_file" >> "$archive" 2>/dev/null || true

  # Truncate current (best-effort)
  : > "$log_file" 2>/dev/null || true

  rotate_archive_if_needed "$archive" "$max_bytes" "$keep" || true
}

ensure_env_file() {
    # DEPLOY_ENV is required for all runs.
    # It may come from runtime-injected env vars (e.g., container) OR from ENV_FILE (if provided+exists).
    # ENV_FILE is optional and is only sourced when it exists. Missing/empty ENV_FILE is OK if DEPLOY_ENV is already set.
    [[ "${IN_CONTAINER}" == true ]] && return 0
    [[ "${__ENVFILE_LOADED:-}" == true ]] && return 0

    local injected_deploy_env_is_set=false
    local injected_deploy_env=""
    if [[ -v DEPLOY_ENV ]]; then
        injected_deploy_env_is_set=true
        injected_deploy_env="${DEPLOY_ENV}"
    fi

    local env_path="${ENV_FILE:-}"
    local env_example=".env.example"

    # If no .env exists, but .env.example does exist, copy it to .env
    if [[ ! -f ".env" && -f "$env_example" ]]; then
        cp "$env_example" ".env"
        echo "ensure_env_file: copied $env_example -> .env" >&2
    fi

    # If no .env exists, and no .env.example exist, copy .env.example from $PORTMASON_SHARE/templates--copy it.
    if [[ ! -f ".env" && -f "$ENV_EXAMPLE_SRC" ]]; then
        cp "$ENV_EXAMPLE_SRC" ".env"
        echo "ensure_env_file: copied $ENV_EXAMPLE_SRC -> .env" >&2
    fi

    # If ENV_FILE exists, allow it to define/override DEPLOY_ENV.
    if [[ -n "$env_path" && -f "$env_path" ]]; then
        # Source in current shell so exports persist.
        # shellcheck disable=SC1090
        export ENV_FILE="$env_path"

        set -a
        . "$ENV_FILE"
        set +a

        echo "ensure_env_file: sourced ENV_FILE='$ENV_FILE'" >&2
    else
        # Not an error — just informational if caller provided a path.
        [[ -n "$env_path" ]] && echo "ensure_env_file: ENV_FILE not found, skipping: '$env_path'" >&2
    fi

    # If env file did not set DEPLOY_ENV, fall back to injected value, even when
    # that value is intentionally empty. Undefined DEPLOY_ENV is an error; empty
    # DEPLOY_ENV is allowed and means production.
    if [[ ! -v DEPLOY_ENV && "$injected_deploy_env_is_set" == true ]]; then
        DEPLOY_ENV="$injected_deploy_env"
        export DEPLOY_ENV
    fi

    if [[ ! -v DEPLOY_ENV ]]; then
        die "ensure_env_file: DEPLOY_ENV is required (set it in ENV_FILE or inject it in the runtime). Use empty DEPLOY_ENV or 'prd' for production."
    fi


    local runtime_adapter_effective="${RUNTIME_ADAPTER_CODE:--}"
    : "${RUNTIME_CODE:=${runtime_adapter_effective%%-*}}"
    : "${ADAPTER_CODE:=${runtime_adapter_effective##*-}}"

    env_init_set ADAPTER_CODE         "${ADAPTER_CODE:-}"
    env_init_set RUNTIME_CODE         "${RUNTIME_CODE:-}"

    export __ENVFILE_LOADED=true

    echo "ensure_env_file: DEPLOY_ENV='$DEPLOY_ENV'" >&2
}

ensure_jq(){
    if have jq; then
        log "jq already installed: $(jq --version)"
        return 0;
    fi

    log "jq not found. Installing..."

    if [[ "$OSTYPE" == "darwin"* ]]; then
        if ! have brew; then
            fatal_error "Error: Homebrew is required on macOS but not found."
        fi
        brew install jq
    elif have apt-get; then
        as_root apt-get update -y
        as_root apt-get install -y jq
    elif have dnf; then
        as_root dnf install -y jq
    elif have yum; then
        as_root yum install -y jq
    elif have apk; then
        # Alpine Linux
        as_root apk add --no-cache jq
    else
        fatal_error "Error: Unsupported OS. Please install jq manually."
    fi

    echo "jq installed successfully: $(jq --version)" >&2
}

ensure_path() {
    local path="${1:-}"
    local silent=0
    [[ "${2:-}" == "-s" ]] && silent=1

    [[ -n "$path" ]] || { die "ensure_path: path is required" 2; }
    [[ -d "$path" ]] && { (( silent )) || log "ensure_path: exists '$path'"; return 0; }
    [[ -e "$path" ]] && { die "ensure_path: '$path' exists but is not a directory" 2; }

    if can_mkdir_p "$path"; then
        mkdir -p "$path"
        (( silent )) || log "ensure_path: created '$path'"
        return 0
    fi

    (( silent )) || log "ensure_path: no permission; attempting sudo mkdir -p '$path'"
    if have sudo; then
        sudo mkdir -p "$path" || { (( silent )) || log "ensure_path: sudo mkdir failed; continuing without creating '$path'"; return 0; }
        (( silent )) || log "ensure_path: created (sudo) '$path'"
        return 0
    fi

    # no sudo available (common in containers) — don't die; just continue
    (( silent )) || log "ensure_path: sudo not available; continuing without creating '$path'"
    return 0
}

# Ensure PROJECT_ID or DEPLOYMENT_ID:
# - accepts: kind (PROJECT|DEPLOYMENT), file (.env default), value (optional)
# - if provided or existing env var is valid -> export + persist
# - else derive from STACK + .project_timestamp -> validate -> export + persist
ensure_project_identifier() {
    local kind="${1:?must be PROJECT or DEPLOYMENT}"
    local file="${2:-.env}"
    local provided="${3-}"

    kind="${kind^^}"
    if [[ "$kind" != "PROJECT" && "$kind" != "DEPLOYMENT" ]]; then
        die "ERROR: kind must be PROJECT or DEPLOYMENT (got: $kind)"
        return 2
    fi

    local var_name="${kind}_ID"
    local var_value

    log "Ensuring $var_name..."

    # Candidate: explicit arg > existing env var
    var_value="${provided:-${!var_name-}}"

    # If we already have a valid value, keep it
    if [[ -n "${var_value:-}" ]] && is_valid_project_identifier "$var_value"; then
        printf -v "$var_name" '%s' "$var_value"
        export "$var_name"
#        touch "$file"
        persist_kv "$var_name" "$var_value" "$file"
        return 0
    fi

    # Derive a value
    if [[ -z "${STACK:-}" ]]; then
        if [[ "${DEPLOY_ENV:-}" == "sys" ]]; then
            log "STACK is not set to derive ${var_name}, but we can continue anyway due to DEPLOY_ENV=sys"
            return 0
        else
            die "STACK is not set, but is required to derive ${var_name}."
            return 2
        fi
    fi

    local ts_file="${DEPLOYMENT_TS_FILE:-$REPO_ROOT/.project_timestamp}"
    if [[ ! -r "$ts_file"  ]]; then
        if [[ "${DEPLOY_ENV:-}" == "sys" ]]; then
            log "No timestamp file found to derive ${var_name}, but we can continue anyway due to DEPLOY_ENV=sys"
            return 0
        else
            die "ERROR: timestamp file '$ts_file' not found or not readable" >&2
            return 2
        fi
    fi

    local raw_ts ts base suffix
    raw_ts="$(< "$ts_file")"
    ts="${raw_ts//[^0-9]/}"

    if [[ -z "$ts" ]]; then
        log "ERROR: timestamp in '$ts_file' is empty or non-numeric" >&2
        return 2
    fi

    base="$(normalize_token "$STACK" --max 19 --replace -)"
    suffix="${ts:(-10)}"
    var_value="${base:0:19}-${suffix}"

    if ! is_valid_project_identifier "$var_value"; then
        log "ERROR: derived ${var_name} is invalid: $var_value" >&2
        return 2
    fi

    printf -v "$var_name" '%s' "$var_value"
    export "$var_name"
    persist_kv "$var_name" "$var_value" "$file"
    return 0
}

ensure_stack() {
    [[ -n "${STACK:-}" ]] && { log "STACK already set from environment or files: ${STACK}"; return 0; }

    if [[ -n "${APP_SLUG:-}" ]]; then
        STACK="${APP_SLUG}${DEPLOY_ENV:+-${DEPLOY_ENV}}"
        env_init_set STACK "${STACK}"         # sets + exports + registers for injection
        persist_kv "STACK" "${STACK}"
        log "Derived STACK: ${STACK}"
        return 0
    fi

    [[ "${DEPLOY_ENV:-}" == "sys" ]] && { log "Skipping STACK requirement due to DEPLOY_ENV==sys"; return 0; }

    log "ERROR: STACK is not set." >&2
    log "Define APP_SLUG or STACK in ${ENV_FILE:-<unset>} or inject it to the environment and rerun." >&2
    return 1
}

#   1) ENV_FILE   — authoritative bulk config
#   2) init vars  — explicitly registered via env_init_set

#   - Secrets (*_TOKEN, *_SECRET, *_KEY, *_PASSWORD, ENV_FILE, DEBUG, VERBOSE)
#   - Runtime noise (PATH, HOME, etc.)

ENV_INIT_VARS=()

ENV_EXCLUDE_GLOBS=(

  '*_SECRET'
  '*_KEY'
  '*_PASSWORD'
  'ENV_FILE'
  'DEBUG'
  'VERBOSE'
  'CUR_SCRIPT'

env_file_to_kv_nul() {
  local env_file="${1:?env file required}"
  [[ -r "$env_file" ]] || return 0

  (
    set -euo pipefail
    set -a
    # shellcheck disable=SC1090
    . "$env_file"
    set +a

    awk '
      function ltrim(s){ sub(/^[ \t\r\n]+/, "", s); return s }
      function rtrim(s){ sub(/[ \t\r\n]+$/, "", s); return s }
      function trim(s){ return rtrim(ltrim(s)) }

      {
        line = trim($0)
        if (line == "" || line ~ /^#/) next

        # allow "export KEY=VAL"
        if (line ~ /^export[ \t]+/) {
          sub(/^export[ \t]+/, "", line)
          line = trim(line)
        }

        # only accept KEY=... (no spaces before '=')
        if (line !~ /^[A-Za-z_][A-Za-z0-9_]*=/) next

        key = line
        sub(/=.*/, "", key)

        val = ENVIRON[key]
        printf "%s\0%s\0", key, val
      }
    ' "$env_file"
  )
}

env_filter_secrets() {
  local k v pat
  while IFS= read -r -d '' k; do
    IFS= read -r -d '' v || v=""

    for pat in "${ENV_EXCLUDE_GLOBS[@]}"; do
      [[ "$k" == $pat ]] && continue 2
    done

    printf '%s\0%s\0' "$k" "$v"
  done
}

env_init_set() {
  local k="${1:?key required}"
  local v="${2-}"

  printf -v "$k" '%s' "$v"
  export "$k"

  for existing in "${ENV_INIT_VARS[@]}"; do
    [[ "$existing" == "$k" ]] && return 0
  done
  ENV_INIT_VARS+=("$k")
}

env_init_to_kv_nul() {
  local k
  for k in "${ENV_INIT_VARS[@]}"; do
    printf '%s\0%s\0' "$k" "${!k-}"
  done
}

env_kv_for_injection() {
  (
    [[ -n "${ENV_FILE-}" ]] && env_file_to_kv_nul "${ENV_FILE}.generated"
    env_init_to_kv_nul
  ) | env_filter_secrets | env_merge_first_wins
}

env_merge_first_wins() {
  declare -A seen=()
  local k v
  while IFS= read -r -d '' k; do
    IFS= read -r -d '' v || v=""
    [[ -n "${seen[$k]+x}" ]] && continue
    seen["$k"]=1
    printf '%s\0%s\0' "$k" "$v"
  done
}

get_deploy_env_value() {
  local short="${1:?pm_env_get requires: <short> <field> [map_file]}"
  local field="${2:?pm_env_get requires: <short> <field> [map_file]}"
  local map_file="${3:-${PORTMASON_SHARE}/pm-deploy-env-mapping.json}"

  have jq || { printf 'ERROR: jq is required\n' >&2; return 1; }
  [[ -f "$map_file" ]] || { printf 'ERROR: mapping file not found: %s\n' "$map_file" >&2; return 1; }

  short="$(printf '%s' "$short" | tr '[:upper:]' '[:lower:]')"
  field="$(printf '%s' "$field" | tr '[:upper:]' '[:lower:]')"

  case "$field" in
    short|long|gcp_tag)
      ;;
    *)
      printf 'ERROR: unknown field "%s" (allowed: short,long,gcp_tag)\n' "$field" >&2
      return 1
      ;;
  esac

  local val
  val="$(jq -r --arg s "$short" --arg f "$field" '
    .envs[$s][$f] // empty
  ' "$map_file")"

  if [[ -z "$val" ]]; then
    printf 'ERROR: unknown env "%s" or missing field "%s"\n' "$short" "$field" >&2
    return 1
  fi

  printf '%s\n' "$val"
}

secret_spec_parts() {
    # Usage: secret_spec_parts SPEC PROVIDER_OUT SECRET_ID_OUT
    # SPEC may be either "provider:path" or bare "path". Bare paths default to
    # GCP so callers can simply use get_secret without knowing the adapter's
    # backing store. "gcs:" aliases to "gcp:" for historical compatibility.
    local spec="${1:?secret spec required}"
    local provider_out="${2:?provider output variable required}"
    local secret_id_out="${3:?secret id output variable required}"

    local provider secret_id
    if [[ "$spec" == *":"* ]]; then
        provider="${spec%%:*}"
        secret_id="${spec#*:}"
    else
        provider="${DEFAULT_SECRET_PROVIDER:-gcp}"
        secret_id="$spec"
    fi

    [[ -n "$provider" ]] || provider="${DEFAULT_SECRET_PROVIDER:-gcp}"
    [[ "$provider" == "gcs" ]] && provider="gcp"

    printf -v "$provider_out" '%s' "$provider"
    printf -v "$secret_id_out" '%s' "$secret_id"
}

get_secret() {
    local spec="${1:?secret spec required}"
    local provider secret_id fn
    secret_spec_parts "$spec" provider secret_id
    fn="get_${provider}_secret"
    declare -F "$fn" >/dev/null 2>&1 || die "get_secret: unsupported secret provider '$provider' for spec '$spec'"
    "$fn" "${secret_id}"
}

get_secret_name_from_path(){
    local spec="${1:?secret path required}"
    local provider secret_id fn
    secret_spec_parts "$spec" provider secret_id
    fn="get_${provider}_secret_name_from_path"
    declare -F "$fn" >/dev/null 2>&1 || die "get_secret_name_from_path: unsupported secret provider '$provider' for spec '$spec'"
    "$fn" "${secret_id}"
}

get_secret_name_by_ref() {
  local path
  path="$(get_secret_path_by_ref "$1")" || return 1
  get_secret_name_from_path "$path" #secret_name is the provider's secret_name
}

get_secret_path_by_ref() {
  local want="${1:-}"
  [[ -z "$want" ]] && bail "get_secret_path_by_ref: missing ref key" 2
  : "${SECRET_PATHS:=}" #secret_path consists of [secret_provider (GCS, AZURE, etc)]:[secret_id]

  local line ref path
  while IFS= read -r line || [[ -n "$line" ]]; do
    [[ -z "$line" ]] && continue
    [[ "$line" != *"|"* ]] && bail "get_secret_path_by_ref: invalid entry (expected 'REF|PATH'): '$line'" 2

    ref="${line%%|*}"
    path="${line#*|}"

    [[ -n "$ref" && -n "$path" ]] || bail "get_secret_path_by_ref: invalid entry (empty ref/path): '$line'" 2

    if [[ "$ref" == "$want" ]]; then
      printf '%s\n' "$path"
      return 0
    fi
  done <<< "$SECRET_PATHS"

  return 1
}

get_secret_names_list() {
  local spec secret_path secret_name
  for spec in "${SECRET_PATHS[@]}"; do

    [[ -n "$secret_name" ]] || die "secret_names_list: empty secret name derived from '$secret_path'"

    printf '%s\n' "$secret_name"
  done
}

live_tail_autostart() {
    : "${LIVE_TAIL:=1}"
    [[ "${LIVE_TAIL}" == "1" ]] || return 0
    [[ -n "${LOG_FILE:-}" ]] || return 0
    [[ "${__HAS_TTY_BEFORE_REDIRECT:-0}" == "1" ]] || return 0
    [[ -w /dev/tty ]] || return 0

    if [[ -n "${__LIVE_TAIL_GUARD_PID:-}" ]] &&
       kill -0 "${__LIVE_TAIL_GUARD_PID}" 2>/dev/null; then
        return 0
    fi

    unset __LIVE_TAIL_PID __LIVE_TAIL_GUARD_PID

    export __LIVE_TAIL_OWNER_BASHPID="${BASHPID}"

    local parent_pid="$$"

    (
        set +e

        tail -n 200 -F "$LOG_FILE" > /dev/tty 2>&1 &
        tail_pid=$!

        pidfile="/tmp/pm-live-tail.${parent_pid}.pid"
        printf '%s\n' "$tail_pid" > "$pidfile" 2>/dev/null || true

        while kill -0 "$parent_pid" 2>/dev/null; do
            sleep 0.25
        done

        kill -- "-$tail_pid" 2>/dev/null || true
        kill "$tail_pid" 2>/dev/null || true
        wait "$tail_pid" 2>/dev/null || true

        rm -f "$pidfile" 2>/dev/null || true
        exit 0
    ) &

    guard_pid=$!
    export __LIVE_TAIL_GUARD_PID="$guard_pid"

    pidfile="/tmp/pm-live-tail.${parent_pid}.pid"

    for _ in 1 2 3 4 5; do
        if [[ -s "$pidfile" ]]; then
            __LIVE_TAIL_PID="$(cat "$pidfile" 2>/dev/null || true)"
            export __LIVE_TAIL_PID
            break
        fi

        sleep 0.05
    done

    return 0
}

live_tail_stop() {
    local owner="${__LIVE_TAIL_OWNER_BASHPID:-}"

    if [[ -n "$owner" && "$owner" != "${BASHPID}" ]]; then
        return 0
    fi

    local guard="${__LIVE_TAIL_GUARD_PID:-}"
    local tailp="${__LIVE_TAIL_PID:-}"

    unset \
        __LIVE_TAIL_OWNER_BASHPID \
        __LIVE_TAIL_GUARD_PID \
        __LIVE_TAIL_PID

    if [[ -n "$guard" ]] && kill -0 "$guard" 2>/dev/null; then
        kill "$guard" 2>/dev/null || true
        wait "$guard" 2>/dev/null || true
    fi

    if [[ -n "$tailp" ]] && kill -0 "$tailp" 2>/dev/null; then
        kill -- "-$tailp" 2>/dev/null || true
        kill "$tailp" 2>/dev/null || true
        wait "$tailp" 2>/dev/null || true
    fi

    return 0
}

log() {
    local severity
    local msg
    case "$#" in
        0)
            severity="INFO"
            msg=""
            ;;
        1)
            severity="INFO"
            msg="${1:-}"
            ;;
        *)
            severity="${1:-}"
            msg="${2:-}"
            ;;
    esac

    should_log "${severity}" || return 0

    # Timestamp (RFC3339 UTC)
    local ts
    ts="$(date -u +'%Y-%m-%dT%H:%M:%S.%3NZ')"

    if [[ "${LOG_FORMAT:-text}" == "json" ]]; then
        local service="${K_SERVICE:-${CLOUD_RUN_SERVICE:-}}"
        local revision="${K_REVISION:-}"
        local execution="${CLOUD_RUN_EXECUTION:-}"
        local task_index="${CLOUD_RUN_TASK_INDEX:-}"
        local script_name="${CUR_SCRIPT:-}"

        local esc_msg="${msg//\\/\\\\}"
        esc_msg="${esc_msg//\"/\\\"}"

        local extras_json=""
        local k v kv

        # Only treat args 3+ as extras
        if (( $# >= 3 )); then
          shift 2
          for kv in "$@"; do
            if [[ "$kv" == *"="* ]]; then
              k="${kv%%=*}"
              v="${kv#*=}"
              v="${v//\\/\\\\}"; v="${v//\"/\\\"}"
              extras_json+=",\"${k}\":\"${v}\""
            fi
          done
        fi

        local json
        json="{\"severity\":\"${severity}\",\"timestamp\":\"${ts}\",\"message\":\"${esc_msg}\""
        [[ -n "${service:-}"    ]] && json+=",\"service\":\"${service}\""
        [[ -n "${script_name:-}" ]] && json+=",\"script\":\"${script_name}\""
        [[ -n "${revision:-}"   ]] && json+=",\"revision\":\"${revision}\""
        [[ -n "${execution:-}"  ]] && json+=",\"execution\":\"${execution}\""
        [[ -n "${task_index:-}" ]] && json+=",\"task_index\":\"${task_index}\""
        json+="${extras_json}"
        json+="}"

        printf '%s\n' "$json" >&2
    else
        # Text mode: readable local/dev output; extras appended.
        local extras=""
        if (( "$#" > 0 )); then
            extras=" $*"
        fi

    #        printf '[%s][%s][%s] %s%s\n' "$ts" "$severity" "${script_name:-}" "${msg}" "$extras" >&2
        printf '[%s][%s][%s] %s\n' "$ts" "$severity" "${CUR_SCRIPT:-}" "${msg}" >&2
    fi
}

normalize_token() {
  # normalize_token <value> (--max N | --fixed N) [--min N] [--replace ""|"-"] [--ensure-last-numeric]
  #
  # Defaults:
  #   --min 1
  #   --replace ""   (remove invalid chars)
  #
  # Determinism:
  #   Padding digits are derived from a SHA-256 hash of the inputs/options (NOT wall-clock time).
  #   Requires sha256sum to ensure identical output across environments.
  #
  # --max:
  #   <= N, but if result shorter than --min, pads deterministically up to --min
  #   empty => "pstack"
  # --fixed:
  #   == N, pads deterministically
  #
  # Ensure first char alpha: prefix "p" if needed (then re-trim if needed)

  local in="${1:-}"; shift || true

  local mode="" n="" min="1" replace="" ensure_last_numeric="false"
  while (( $# )); do
    case "$1" in
      --max|--fixed)
        [[ -n "$mode" ]] && { echo "normalize_token: choose one of --max or --fixed" >&2; return 2; }
        mode="${1#--}"
        shift; n="${1:-}"; [[ -z "$n" ]] && { echo "normalize_token: $mode requires N" >&2; return 2; }
        ;;
      --min)
        shift; min="${1:-}"; [[ -z "$min" ]] && { echo "normalize_token: --min requires N" >&2; return 2; }
        ;;
      --replace)
        shift; replace="${1-}"
        ;;
      --ensure-last-numeric)
        ensure_last_numeric="true"
        ;;
      *) echo "normalize_token: bad arg: $1" >&2; return 2 ;;
    esac
    shift || true
  done

  [[ -z "$mode" ]] && { echo "normalize_token: one of --max or --fixed is required" >&2; return 2; }
  [[ "$n"   =~ ^[0-9]+$ ]] && (( n   > 0 )) || { echo "normalize_token: N must be positive int" >&2; return 2; }
  [[ "$min" =~ ^[0-9]+$ ]] && (( min > 0 )) || { echo "normalize_token: --min must be positive int" >&2; return 2; }
  (( min <= n )) || { echo "normalize_token: --min must be <= $mode length" >&2; return 2; }

  # Lock down replace to safe, deterministic values (matches allow logic below)
  case "$replace" in
    ""|"-") ;;
    *) echo "normalize_token: --replace only supports '' or '-'" >&2; return 2 ;;
  esac

  local allow='a-z0-9'
  [[ "$replace" == "-" ]] && allow='a-z0-9-'

  local s
  s="$(printf '%s' "$in" | tr '[:upper:]' '[:lower:]')"

  if [[ -z "$replace" ]]; then
    s="$(printf '%s' "$s" | tr -cd "$allow")"
  else
    # replace == "-" (only other allowed value)
    s="$(printf '%s' "$s" | sed "s/[^$allow]\\+/-/g; s/-\\{2,\\}/-/g")"
    s="$(printf '%s' "$s" | sed 's/^-//; s/-$//')"
  fi

  # Deterministic padding digits derived from inputs/options (NOT wall-clock time).
  # Require sha256sum so output is identical across environments.
  have sha256sum \
    || { echo "normalize_token: sha256sum is required for deterministic padding" >&2; return 2; }

  local seed digits
  seed="in=${in}|mode=${mode}|n=${n}|min=${min}|replace=${replace}|ensure_last_numeric=${ensure_last_numeric}"
  digits="$(printf '%s' "$seed" | sha256sum | awk '{print $1}' | tr -cd '0-9')"
  [[ -n "$digits" ]] || digits="0"

  if [[ "$mode" == "max" ]]; then
    (( ${#s} > n )) && s="${s:0:n}"
    [[ -z "$s" ]] && s="pstack"
    [[ "$s" =~ ^[a-z] ]] || s="p$s"
    (( ${#s} > n )) && s="${s:0:n}"

    # pad up to min (but never exceed max)
    if (( ${#s} < min )); then
      local need=$(( min - ${#s} )) pad=""
      while (( ${#pad} < need )); do pad+="$digits"; done
      s+="${pad:0:need}"
    fi
    (( ${#s} > n )) && s="${s:0:n}"

    if [[ "$ensure_last_numeric" == "true" && $n -gt 1 && ! "$s" =~ [0-9]$ ]]; then
      s="${s}0"
      (( ${#s} > n )) && s="${s:0:n}"
    fi

    printf '%s' "$s"
    return 0
  fi

  # fixed mode
  [[ -z "$s" ]] && s="p"
  [[ "$s" =~ ^[a-z] ]] || s="p$s"
  (( ${#s} > n )) && s="${s:0:n}"

  if (( ${#s} < n )); then
    local need=$(( n - ${#s} )) pad=""
    while (( ${#pad} < need )); do pad+="$digits"; done
    s+="${pad:0:need}"
  fi

  if [[ "$ensure_last_numeric" == "true" && $n -gt 1 && ! "$s" =~ [0-9]$ ]]; then
    s="${s:0:n-1}${digits:0:1}"
  fi

  printf '%s' "$s"
}

pm_setup_arg() {
  # Usage:
  #   pm_setup_arg VAR_NAME ALIASES [required] [needs_value] [export] [default=...]
  #
  # Examples:
  #   pm_setup_arg LOG_ROW_LIMIT "-r|--row-limit" required needs_value default=2000 export
  #   pm_setup_arg VERBOSE "-v|--verbose"                      # flag (needs_value=no)
  #
  # Notes:
  # - required / needs_value / export are single-word flags
  # - default is provided as default=...
  # - default does NOT imply needs_value=yes, but it CAN satisfy needs_value=yes in pm_commit_args (Rule B)
  # - required is satisfied if explicitly seen OR default is non-empty

  if (( $# < 2 )); then
    echo >&2 "pm_setup_arg: requires 2 positional args: VAR_NAME ALIASES"
    return 2
  fi

  local var_name="$1"
  local aliases="$2"
  shift 2

  if [[ -z "$var_name" || -z "$aliases" ]]; then
    echo >&2 "pm_setup_arg: VAR_NAME and ALIASES must be non-empty"
    return 2
  fi

  local required="no"
  local needs_value="no"
  local default_value=""
  local do_export="no"

  local arg
  while (( $# > 0 )); do
    arg="$1"
    shift

    case "$arg" in
      required)     required="yes" ;;
      needs_value)  needs_value="yes" ;;
      export)       do_export="yes" ;;
      default=*)    default_value="${arg#*=}" ;;
      *)
        echo >&2 "pm_setup_arg: unknown token: $arg"
        return 2
        ;;
    esac
  done

  # Init registries once
  if ! declare -p PM_ARGS &>/dev/null; then
    declare -gA PM_ARGS=()             # key_spec (aliases string) -> var_name
    declare -ga PM_ARGS_KEYS=()        # ordered list of key_specs
    declare -gA PM_ARGS_REQUIRED=()    # var_name -> yes|no
    declare -gA PM_ARGS_NEEDS_VALUE=() # var_name -> yes|no
    declare -gA PM_ARGS_DEFAULT=()     # var_name -> default string (possibly empty)
  fi

  # Declare target var if missing (global scalar)
  if ! declare -p "$var_name" &>/dev/null; then
    declare -g "$var_name"=""
  fi

  # Apply default at setup time if provided AND current var is empty/unset
  if [[ -n "$default_value" ]]; then
    local cur="${!var_name-}"
    if [[ -z "$cur" ]]; then
      printf -v "$var_name" '%s' "$default_value"
    fi
  fi

  # Export now if requested (not persisted)
  if [[ "$do_export" == "yes" ]]; then
    export "$var_name"
  fi

  # Register
  PM_ARGS["$aliases"]="$var_name"
  PM_ARGS_KEYS+=( "$aliases" )
  PM_ARGS_REQUIRED["$var_name"]="$required"
  PM_ARGS_NEEDS_VALUE["$var_name"]="$needs_value"
  PM_ARGS_DEFAULT["$var_name"]="$default_value"
}

pm_commit_args() {
  # Parses ONLY option tokens (no positional args support).
  #
  # Contract:
  # - Values must be passed as key=value (e.g., --threshold=-5). No "--key value".
  # - If needs_value=yes and bare --key appears:
  #     - use default if present (Rule B)
  #     - else error
  # - needs_value=no => presence sets var to 1.
  # - required=yes satisfied if explicitly seen OR default non-empty.

  if ! declare -p PM_ARGS &>/dev/null; then
    return 0
  fi

  local -a argv=( "$@" )

  # Build alias -> var lookup
  local -A alias_to_var=()
  local key_spec var_name alias
  local -a _aliases

  for key_spec in "${PM_ARGS_KEYS[@]}"; do
    var_name="${PM_ARGS[$key_spec]}"
    IFS='|' read -r -a _aliases <<< "$key_spec"
    for alias in "${_aliases[@]}"; do
      [[ -n "$alias" ]] || continue
      alias_to_var["$alias"]="$var_name"
    done
  done

  local -A seen=()

  local arg key val target nv def
  for arg in "${argv[@]}"; do
    [[ -n "$arg" ]] || continue

    # token with '='
    if [[ "$arg" == *=* ]]; then
      key="${arg%%=*}"
      val="${arg#*=}"

      [[ -n "$key" ]] || continue

      target="${alias_to_var[$key]:-}"
      if [[ -n "$target" ]]; then
        nv="${PM_ARGS_NEEDS_VALUE[$target]:-yes}"
        if [[ "$nv" == "no" ]]; then
          echo >&2 "Unexpected value for flag: $key (use '$key' without '=')"
          return 2
        fi
        printf -v "$target" '%s' "$val"
        seen["$target"]=1
      fi
      continue
    fi

    # bare token
    target="${alias_to_var[$arg]:-}"
    if [[ -z "$target" ]]; then
      continue
    fi

    nv="${PM_ARGS_NEEDS_VALUE[$target]:-yes}"
    def="${PM_ARGS_DEFAULT[$target]:-}"

    if [[ "$nv" == "no" ]]; then
      printf -v "$target" '%s' "1"
      seen["$target"]=1
      continue
    fi

    # needs_value == yes, but no '=value'
    if [[ -n "$def" ]]; then
      # default already applied (or could be reapplied); counts as satisfied
      seen["$target"]=1
      continue
    fi

    echo >&2 "Missing required value for option: $arg (use $arg=value)"
    return 2
  done

  # Enforce requireds: explicit or default non-empty
  for key_spec in "${PM_ARGS_KEYS[@]}"; do
    var_name="${PM_ARGS[$key_spec]}"
    if [[ "${PM_ARGS_REQUIRED[$var_name]:-no}" != "yes" ]]; then
      continue
    fi
    if [[ -n "${seen[$var_name]:-}" ]]; then
      continue
    fi
    if [[ -n "${PM_ARGS_DEFAULT[$var_name]:-}" ]]; then
      continue
    fi
    echo >&2 "Missing required option for variable: $var_name"
    return 2
  done
}

rotate_archive_if_needed() {
  # Usage: rotate_archive_if_needed "$archive_file" "$max_bytes" "$keep"
  local archive="${1:?archive required}"
  local max_bytes="${2:?max_bytes required}"
  local keep="${3:?keep required}"

  [[ -e "$archive" ]] || return 0

  local size=0
  if have stat; then
    size="$(stat -c '%s' "$archive" 2>/dev/null || echo 0)"
  else
    # fallback
    size="$(wc -c < "$archive" 2>/dev/null || echo 0)"
  fi

  (( size > max_bytes )) || return 0

  # Shift older archives: .(keep-1) -> .keep
  local i
  for (( i=keep; i>=2; i-- )); do
    local prev=$((i-1))
    [[ -e "${archive}.${prev}" ]] && mv -f "${archive}.${prev}" "${archive}.${i}" 2>/dev/null || true
  done

  # Move current archive -> .1, then start a new empty archive
  mv -f "$archive" "${archive}.1" 2>/dev/null || true
  : > "$archive" 2>/dev/null || true
}

setup_logging() {
    # Safe runtime-friendly logging setup.
    #
    # Goals:
    # - Never require sudo.
    # - Never break if log dir/file can’t be created/opened.
    # - Only redirect stdout/stderr to a file if we successfully opened the file.
    # - Default to container behavior (stdout/stderr) when container detected, unless forced.
    # - If archiving occurs, ALWAYS truncate current log afterwards (fresh session file).
    #
    # Knobs:
    #   PM_NO_REDIRECT=1        -> never redirect (always leave stdout/stderr unchanged)
    #   PM_FORCE_REDIRECT=1     -> redirect even in container (not recommended for Cloud Run)
    #   LOG_DIR=/path           -> preferred log dir
    #   LOG_FILE_CONFIGURE=/p   -> explicit log file path
    #   LOG_TRUNCATE=1          -> truncate LOG_FILE at start (even if no archive)
    #
    # Exports:
    #   LOG_DIR, LOG_FILE, __HAS_TTY_BEFORE_REDIRECT, IN_CONTAINER
    #
    # Notes:
    # - This function must never hard-fail. It always returns 0.
    #
    # Functions:
    # log [<severity: INFO, WARN, etc. (defaults to INFO)>] <message> -> logs message based on log level
    # ok <message>                                            -> calls log() then returns code 0
    # bail [<return code (defaults to 1)>]                    -> returns/exits with return code
    # warn <message>                                          -> calls log() with severity WARN
    # error <message>                                         -> calls log() with severity ERROR
    # fatal [<return code (defaults to 1)>]                   -> exits with return code
    # fatal_error <message> [<return code (defaults to 1)>]   -> calls error() then exits with return code

    # ---- small internal helpers (must not depend on your log()/warn()) ----
    _sl_stderr() { printf '%s\n' "$*" >&2 || true; }
    _sl_can_write_dir() { [[ -d "$1" && -w "$1" ]]; }
    _sl_can_mkdir() { mkdir -p "$1" >/dev/null 2>&1; }
    _sl_file_open_append() { # sets FD 3 if succeeds
        exec 3>>"$1" 2>/dev/null
    }
    _sl_is_ro_rootfs() {
        # Read-only rootfs detection WITHOUT writing to /
        # Returns 0 if root appears read-only, 1 otherwise.
        #
        # /proc/mounts format: <src> <target> <fstype> <opts> ...
        # We find the line where target == "/" and check if opts contains "ro".
        local line opts
        line="$(awk '$2=="/"{print; exit}' /proc/mounts 2>/dev/null || true)"
        [[ -n "$line" ]] || return 1
        opts="$(awk '$2=="/"{print $4; exit}' /proc/mounts 2>/dev/null || true)"
        echo "$opts" | grep -qE '(^|,)ro(,|$)' >&2
    }
    _sl_detect_container() {
        # Return 0 if container-ish, 1 otherwise.
        # Priority: explicit runtime vars -> /.dockerenv -> cgroup -> ro rootfs -> pid1 comm
        # informed directly from var
        case "${IN_CONTAINER:-false}" in
            1|true|TRUE|yes|YES|y|Y|on|ON) return 0 ;;
        esac
        # Cloud Run / Knative / K8s hints
        if [[ -n "${K_REVISION:-}" || -n "${K_SERVICE:-}" || -n "${K_CONFIGURATION:-}" ]]; then return 0; fi
        if [[ -n "${CLOUD_RUN_JOB:-}" || -n "${CLOUD_RUN_EXECUTION:-}" || -n "${CLOUD_RUN_TASK_INDEX:-}" ]]; then return 0; fi
        if [[ -n "${KUBERNETES_SERVICE_HOST:-}" ]]; then return 0; fi

        # Classic Docker marker
        if [[ -f "/.dockerenv" ]]; then return 0; fi

        # cgroup markers (works for docker/containerd/k8s)
        local cg=""
        if [[ -r /proc/1/cgroup ]]; then
            cg="$(cat /proc/1/cgroup 2>/dev/null || true)"
            if echo "$cg" | grep -qiE 'docker|containerd|kubepods|crio' >&2; then return 0; fi
        elif [[ -r /proc/self/cgroup ]]; then
            cg="$(cat /proc/self/cgroup 2>/dev/null || true)"
            if echo "$cg" | grep -qiE 'docker|containerd|kubepods|crio' >&2; then return 0; fi
        fi

        # Read-only rootfs is a strong signal in managed container runtimes
        if _sl_is_ro_rootfs; then return 0; fi

        # PID 1 name (weak hint)
        if [[ -r /proc/1/comm ]]; then
            local comm
            comm="$(cat /proc/1/comm 2>/dev/null || true)"
            if echo "$comm" | grep -qiE 'tini|docker-init|pause' >&2; then return 0; fi
        fi

        return 1
    }

    # ---- determine caller for default log name ----
    local top_caller="<interactive>"
    for (( i=${#BASH_SOURCE[@]}-1; i>=0; i-- )); do
        [[ -n "${BASH_SOURCE[$i]:-}" ]] && { top_caller="${BASH_SOURCE[$i]}"; break; }
    done
    top_caller="$(basename "$top_caller")"

    # ---- tty state ----
    __HAS_TTY_BEFORE_REDIRECT=0
    [[ -t 0 && -t 1 ]] && __HAS_TTY_BEFORE_REDIRECT=1
    export __HAS_TTY_BEFORE_REDIRECT

    # ---- container detection ----
    if _sl_detect_container; then IN_CONTAINER=true; fi
    export IN_CONTAINER

    # ---- explicit overrides ----
    if [[ "${PM_NO_REDIRECT:-0}" == "1" ]]; then
        : "${LOG_FILE:=${LOG_FILE_CONFIGURE:-}}"
        _sl_stderr "[setup_logging] no-redirect (PM_NO_REDIRECT=1); leaving stdout/stderr unchanged"
        return 0
    fi

    # Container behavior:
    # - Default: do NOT redirect in container (platform captures stdout/stderr).
    # - Local docker (ADAPTER_CODE=local): prefix every line with "[container]" (text logs only),
    #   but STILL do not redirect unless PM_FORCE_REDIRECT=1.
    if [[ "${IN_CONTAINER}" == true ]]; then
        local adapter="${ADAPTER_CODE:-}"

        # Forced redirect is allowed (but usually not recommended in managed runtimes).
        if [[ "${PM_FORCE_REDIRECT:-0}" == "1" ]]; then
            _sl_stderr "[setup_logging] WARNING: PM_FORCE_REDIRECT=1 inside container may prevent platform logging from capturing stdout/stderr"
            # fall through to file-redirect logic below
        else
            : "${LOG_FILE:=${LOG_FILE_CONFIGURE:-}}"

            # Prefix only in local docker and only for text logs (prefix would break JSON).
            if [[ "${adapter}" == "local" && "${LOG_FORMAT:-text}" != "json" ]]; then
                if [[ "${__PM_CONTAINER_PREFIXED:-0}" != "1" ]]; then
                export __PM_CONTAINER_PREFIXED=1
                exec > >(sed -u 's/^/[container] /')
                exec 2> >(sed -u 's/^/[container] /' >&2)
                fi
                _sl_stderr "[setup_logging] local container detected (ADAPTER_CODE=local); prefixing stdout/stderr with [container]"
            else
                _sl_stderr "[setup_logging] container detected; leaving stdout/stderr unchanged (set PM_FORCE_REDIRECT=1 to redirect)"
            fi

            # IMPORTANT: Preserve original behavior — no file redirect in containers unless forced.
            return 0
        fi
    fi

    # ---- choose log dir (must never fail) ----
    # Precedence:
    # 1) Existing LOG_DIR, if specified
    # 2) REPO_ROOT/logs
    # 3) Runtime fallback

    if [[ -z "${LOG_DIR:-}" ]]; then
        LOG_DIR="${REPO_ROOT:-$(pwd -P)}/logs"
    fi

    if ! _sl_can_mkdir "$LOG_DIR"; then
        local fallback=""
        if [[ -n "${XDG_RUNTIME_DIR:-}" ]]; then
            fallback="${XDG_RUNTIME_DIR%/}/logs"
        else
            fallback="/tmp/logs"
        fi

        _sl_can_mkdir "$fallback" || true
        LOG_DIR="$fallback"
    fi
    export LOG_DIR

    LOG_FILE="${LOG_FILE_CONFIGURE:-$LOG_DIR/$top_caller.log}"
    export LOG_FILE

    # ---- optional archive (must never fail) ----
    local did_archive=0
    if declare -F archive_log_on_session_start >/dev/null 2>&1; then
        if [[ -s "$LOG_FILE" ]]; then
            archive_log_on_session_start "$LOG_FILE" >/dev/null 2>&1 || true
            did_archive=1
        else
            archive_log_on_session_start "$LOG_FILE" >/dev/null 2>&1 || true
        fi
    fi

    # ---- truncate (must never fail) ----
    if [[ "${LOG_TRUNCATE:-0}" == "1" || "$did_archive" == "1" ]]; then
        : >"$LOG_FILE" 2>/dev/null || true
    fi

    # ---- open file; redirect only if open succeeded ----
    if ! _sl_file_open_append "$LOG_FILE"; then
        _sl_stderr "[setup_logging] could not open log file; leaving stdout/stderr unchanged: $LOG_FILE"
        return 0
    fi

    # Redirect stdout/stderr to FD 3
    exec 1>&3 2>&3

    # xtrace to file if enabled
    if [[ "${DEBUG:-0}" == "1" ]]; then
        export BASH_XTRACEFD=3
        export PS4='+ ${BASH_SOURCE##*/}:${LINENO}:${FUNCNAME[0]:-MAIN}(): '
        set -x
    fi

    # Now it’s safe to use your structured logger if it exists.
    if declare -F log >/dev/null 2>&1; then
        log INFO "SESSION START" \
        "caller=${top_caller}" "pid=$$" "ppid=${PPID:-}" "user=${USER:-}" \
        "host=$(hostname 2>/dev/null || echo unknown >&2)" "cwd=$(pwd -P)" \
        "redirect=1" "log_file=${LOG_FILE}" "archived=${did_archive}" \
        "debug=${DEBUG:-0}" "verbose=${VERBOSE:-0}" "IN_CONTAINER=${IN_CONTAINER}"
    fi

    # Optional live tail (must tolerate no /dev/tty)
    if declare -F live_tail_autostart >/dev/null 2>&1; then
        live_tail_autostart >/dev/null 2>&1 || true
    fi

    return 0
}

# Internal: should we emit this severity at current LOG_LEVEL?
should_log() {
  local severity="${1:-INFO}"
  local sev_n=20 thr_n=20

  case "${severity}" in
    DEBUG)   sev_n=10 ;;
    INFO)    sev_n=20 ;;
    WARNING|WARN) sev_n=30 ;;
    ERROR)   sev_n=40 ;;
    *)       sev_n=20 ;;
  esac

  case "${LOG_LEVEL:-INFO}" in
    DEBUG)   thr_n=10 ;;
    INFO)    thr_n=20 ;;
    WARNING|WARN) thr_n=30 ;;
    ERROR)   thr_n=40 ;;
    *)       thr_n=20 ;;
  esac

  [[ "$sev_n" -ge "$thr_n" ]]
}

stage_portmason() {
    # Usage:
    #   stage_portmason [exclude_pattern...]
    #
    # Exclude patterns are rsync-style, relative to the source root:
    #   "cache/"          (exclude subdir)
    #   "secrets.env"     (exclude file)
    #   "*.log"           (exclude glob)
    #
    # Example:
    #   stage_portmason "cache/" "*.tmp" "node_modules/"

    local src="${PORTMASON_SHARE}"
#    local dest="$(determine_dest_root)/portmason"
    local dest="${REPO_ROOT}/portmason"

    log "Staging portmason -> ${dest} (from ${PORTMASON_SHARE})"

    [[ -d "$src" ]] || die "Missing directory: ${src}"
    mkdir -p "$dest"

    # -a = archive; -L = copy the symlink targets as real files/dirs
    local -a rsync_cmd=(rsync -aL --delete)

    # Add excludes (if any)
    local pat
    for pat in "$@"; do
        [[ -n "$pat" ]] || continue
        rsync_cmd+=(--exclude "$pat")
    done

    rsync_cmd+=("${src}/" "${dest}/")

    # Optional: log the exact command
    local _print
    printf -v _print '%q ' "${rsync_cmd[@]}"
    log DEBUG "CMD: ${_print}"

    "${rsync_cmd[@]}"
}

pm_init() {
    case "${__PM_HELPERS_STATE:-}" in
        loaded)  return 0 2>/dev/null || exit 0 ;;
        loading) return 0 2>/dev/null || exit 0 ;;
    esac

    __PM_HELPERS_STATE="loading"
    export __PM_HELPERS_STATE

    # If init fails, clear the state so a later retry is allowed.
    # (This also avoids leaving "loading" stuck in the caller shell.)
    trap '__PM_HELPERS_STATE=""; unset __PM_HELPERS_STATE' ERR

    env_init_set IN_CONTAINER   "${IN_CONTAINER:-false}"
    env_init_set ORIG_PWD       "$(pwd -P)"

    if [[ -z "${REPO_ROOT:-}" ]]; then
        if [[ "$ORIG_PWD" =~ ^(.+)/deploy/[^/]+$ ]]; then
            env_init_set REPO_ROOT "${BASH_REMATCH[1]}"
        else
            env_init_set REPO_ROOT "$ORIG_PWD"
        fi
    fi

    env_init_set REPO_NAME "${REPO_NAME:-$(basename -- "$REPO_ROOT")}"

    if [[ -z "${SITE_DIR:-}" ]]; then
        if [[ -f "$ORIG_PWD/index.html" ]]; then
            env_init_set SITE_DIR "$ORIG_PWD"
        elif [[ -f "$ORIG_PWD/www/index.html" ]]; then
            env_init_set SITE_DIR "$ORIG_PWD/www"
        else
            env_init_set SITE_DIR "$ORIG_PWD"
        fi
    fi

    env_init_set SITE_ROOT          "${SITE_ROOT:-$SITE_DIR}"
    env_init_set SCRIPT_DIR         "${SCRIPT_DIR:-${SITE_DIR}/mgt-scripts}"
    env_init_set PORTMASON_SHARE    "${PORTMASON_SHARE:-"$(cd -- "$(dirname -- "${BASH_SOURCE[0]:-${0:-}}")" && pwd -P)"}"

    env_init_set ENV_FILE           "${ENV_FILE:-${ORIG_PWD}/.env}"
    env_init_set ENV_EXAMPLE_SRC    "${ENV_EXAMPLE_SRC:-${PORTMASON_SHARE}/templates/.env.example}"

    ensure_env_file "$@"

    setup_logging "$@"

    # Optional: now that logging is configured, dump env if you still want it.
    if [[ "${PM_DUMP_ENV_ON_INIT:-0}" == "1" ]]; then
        log "Environment variables:"
        env | sort | sed 's/^/  /'
    fi

    # shellcheck disable=SC1090
    . "${PORTMASON_SHARE}/pm-helpers-base"

    sys_bootstrap "$@"

    __PM_HELPERS_STATE="loaded"
    export __PM_HELPERS_STATE

    # Init succeeded; no need for ERR cleanup anymore.
    trap - ERR
}

trap_add() {
  local trap_name="$1"; shift
  local new_cmd="$*"
  local existing
  existing="$(trap -p "$trap_name" | sed -E "s/^trap -- '(.*)' ${trap_name}$/\1/")"

  if [[ -n "$existing" ]]; then
    trap "${existing}; ${new_cmd}" "$trap_name"
  else
    trap "${new_cmd}" "$trap_name"
  fi
}

begin_var_trace() {
  # If already set, reuse same file; otherwise create one.
  VAR_TRACE_START="${VAR_TRACE_START:-$(mktemp)}"
  declare -p | sort > "$VAR_TRACE_START"
}

canon_slug_19() {
  local raw="$1"
  local slug
  slug="$(printf '%s\n' "$raw" \
    | tr '[:upper:]' '[:lower:]' \
    | sed 's/[^a-z0-9-]/-/g' \
    | sed 's/-\{2,\}/-/g' \
    | sed 's/^-//; s/-$//')"
  slug="${slug:0:19}"
  if [[ ! "$slug" =~ ^[a-z] ]]; then
    slug="p${slug}"
  fi
  printf '%s\n' "$slug"
}

end_var_trace() {
  # If no trace was started, do nothing (avoid set -u death)
  if [[ -z "${VAR_TRACE_START:-}" ]]; then
    log "variable changes: (no var trace started)" >&2
    return 0
  fi

  local end
  end="$(mktemp)"
  declare -p | sort > "$end"

  log "variable changes:"
  diff -u "$VAR_TRACE_START" "$end" || true

  rm -f "$VAR_TRACE_START" "$end"
  unset VAR_TRACE_START
}

is_env_dir() {
  local deploy_env="${DEPLOY_ENV:-}"
  local parent

  parent="$(basename "$(dirname "$PWD")")"

  # Empty DEPLOY_ENV and dev do not require deploy/ parent.
  if [[ -z "$deploy_env" || "$deploy_env" == "dev" ]]; then
    return 0
  fi

  # Anything else must run from deploy/<env>.
  [[ "$parent" == "deploy" ]]
}

# Identifier validation (applies to both PROJECT_ID and DEPLOYMENT_ID)
# - 6–30 chars
# - starts with [a-z]
# - only [a-z0-9-]
# - ends with [a-z0-9]
is_valid_project_identifier() {
  local id="${1-}"
  [[ ${#id} -ge 6 && ${#id} -le 30 ]] || return 1
  [[ "$id" =~ ^[a-z][a-z0-9-]*[a-z0-9]$ ]] || return 1
  return 0
}

persist_kv() {
  local key="$1"
  local value="$2"
  local file="${3:-${ENV_FILE}}"

  if [[ -z "${file}" || ! -f "${file}" ]]; then
    ok "Unable to find env file (file: '${file:-}')"
    return 0
  fi

  touch "$file"

  # Validate key (POSIX-ish env key)
  if [[ ! "$key" =~ ^[A-Za-z_][A-Za-z0-9_]*$ ]]; then
    log "ERROR: invalid env key: $key" >&2
    return 2
  fi

  # Escape value for sed replacement (handles \, &, and delimiter |)
  local esc_value
  esc_value="$(printf '%s' "$value" | sed -e 's/[\\&|]/\\&/g')"

  # If key exists, replace; else append.
  if grep -qE "^${key}=" "$file" 2>/dev/null; then
    # Use | as delimiter to avoid escaping / in values
    sed -i.bak -E "s|^${key}=.*|${key}=${esc_value}|" "$file"
  else
    cp -p "$file" "$file.bak" 2>/dev/null || true
    printf '%s=%s\n' "$key" "$value" >> "$file"
  fi
}

set-docker-env-file() {
    local file="${1:-docker-compose.yml}"

    if [[ ! -f "$file" ]]; then
        log "ERROR: $file not found." >&2
        return 1
    fi

    if ! command -v yq >/dev/null 2>&1; then
        log "ERROR: 'yq' (mikefarah Go version) is required but not found in PATH." >&2
        return 1
    fi

    if [[ -z "$ENV_FILE" || ! -f "$ENV_FILE" ]]; then
        log "ERROR: unabel to find file 'ENV_FILE' of the value isn't set." >&2
        return 1
    fi

    log "Using env file: $ENV_FILE"

    # Backup original
    cp "$file" "${file}.bak"

    # For every service:
    # - overwrite env_file with the detected ENV_FILE
    # This uses Go yq v4 syntax.
    yq eval -i ".services[].env_file = \"$ENV_FILE\"" "$file"

    log "Updated $file -> env_file: \"$ENV_FILE\" for all services."
    log "Backup saved as ${file}.bak."
}

env_injection_args__gcp() {
  # Reads NUL-separated KEY\0VALUE\0... from stdin and prints NUL-separated
  # argv tokens for gcloud:
  #   --set-env-vars\0KEY=VALUE\0...

  local key="" val=""

  while IFS= read -r -d '' key; do
    # If key contains '=', warn and strip '=' chars from key
    if [[ "$key" == *"="* ]]; then
      warn "env_injection_args__gcp: key contains '='; stripping and continuing" "key=${key}"
      key="${key//=}"
    fi

    # If key becomes empty/invalid, warn+skip (and skip the subsequent value too, if present)
    if [[ -z "$key" ]]; then
      warn "env_injection_args__gcp: empty key after normalization; skipping pair"
      IFS= read -r -d '' _ 2>/dev/null || true
      continue
    fi

    # Read value; if missing (dangling key), treat as empty value
    if IFS= read -r -d '' val 2>/dev/null; then
      :
    else
      val=""
    fi

    # Emit one argv token (NUL-separated) for gcloud
    printf '%s\0%s\0' "--set-env-vars" "${key}=${val}"
  done
}

resolve_gcp_secret_path() {
    local in_path="$1"
    in_path="${in_path#gcs:}"
    in_path="${in_path#gcp:}"

  # If already a full Secret Manager resource path, pass through.
  if [[ "$in_path" =~ ^projects/[^/]+/secrets/[^/:]+(:[^/]+)?$ ]]; then
    printf '%s' "$in_path"
    return 0
  fi

  # Expect "PROJECT_ID:SECRET_NAME" (e.g. github-archive-480721:github-backup-token)
  local project_id="${in_path%%:*}"
  local secret_name="${in_path#*:}"

  [[ -n "$project_id" && -n "$secret_name" && "$secret_name" != "$in_path" ]] \
    || { die "resolve_gcp_secret_path: invalid input '$1' (expected gcp:project:secret or projects/.../secrets/...)" 2; }

  local project_number
  project_number="$(get_gcp_project_number "$project_id")" || return 1

  printf 'projects/%s/secrets/%s' "$project_number" "$secret_name"
}

#!/usr/bin/env bash
set -euo pipefail

# Live-tail implementation that survives caller `exec` and always cleans up.
# Strategy: background guard process monitors original parent PID; kills tail when parent exits.

live_tail_autostart() {
  : "${LIVE_TAIL:=1}"
  [[ "${LIVE_TAIL}" == "1" ]] || return 0
  [[ -n "${LOG_FILE:-}" ]] || return 0

  [[ "${__HAS_TTY_BEFORE_REDIRECT:-0}" == "1" ]] || return 0
  [[ -w /dev/tty ]] || return 0

  # Already running?
  if [[ -n "${__LIVE_TAIL_GUARD_PID:-}" ]] && kill -0 "${__LIVE_TAIL_GUARD_PID}" 2>/dev/null; then
    return 0
  fi
  unset __LIVE_TAIL_PID __LIVE_TAIL_GUARD_PID

  local parent_pid="$$"

  (
    set +e

    # Start tail
    tail -n 200 -F "$LOG_FILE" > /dev/tty 2>&1 &
    tail_pid=$!

    # Publish PIDs back into environment via a temp file (cannot export back to parent directly)
    pidfile="/tmp/pm-live-tail.${parent_pid}.pid"
    printf '%s\n' "$tail_pid" > "$pidfile" 2>/dev/null || true

    # Watch parent; when it dies, stop tail
    while kill -0 "$parent_pid" 2>/dev/null; do
      sleep 0.25
    done

    kill -- "-$tail_pid" 2>/dev/null || true
    kill "$tail_pid" 2>/dev/null || true
    wait "$tail_pid" 2>/dev/null || true

    rm -f "$pidfile" 2>/dev/null || true
    exit 0
  ) &

  guard_pid=$!
  export __LIVE_TAIL_GUARD_PID="$guard_pid"

  # Best-effort: read tail pid
  pidfile="/tmp/pm-live-tail.${parent_pid}.pid"
  for _ in 1 2 3 4 5; do
    if [[ -s "$pidfile" ]]; then
      __LIVE_TAIL_PID="$(cat "$pidfile" 2>/dev/null || true)"
      export __LIVE_TAIL_PID
      break
    fi
    sleep 0.05
  done

  return 0
}

live_tail_stop() {
  # Stop guard (which stops tail), plus best-effort direct kill.
  local guard="${__LIVE_TAIL_GUARD_PID:-}"
  local tailp="${__LIVE_TAIL_PID:-}"

  unset __LIVE_TAIL_GUARD_PID __LIVE_TAIL_PID

  if [[ -n "$guard" ]] && kill -0 "$guard" 2>/dev/null; then
    kill "$guard" 2>/dev/null || true
    wait "$guard" 2>/dev/null || true
  fi

  if [[ -n "$tailp" ]] && kill -0 "$tailp" 2>/dev/null; then
    kill -- "-$tailp" 2>/dev/null || true
    kill "$tailp" 2>/dev/null || true
    wait "$tailp" 2>/dev/null || true
  fi

  return 0
}

install_core() {
  if ! rt core is-installed --perform_readiness_check >/dev/null 2>&1; then
    echo "[${script_name}] Installing WordPress core via WP-CLI…"
    rt core install \
      --url="${WP_SITE_URL:?WP_SITE_URL must be set in .env}" \
      --title="${WP_SITE_TITLE:-WordPress Site}" \
      --admin_user="${WP_ADMIN_USER:?WP_ADMIN_USER must be set in .env}" \
      --admin_password="${WP_ADMIN_PASS:?WP_ADMIN_PASS must be set in .env}" \
      --admin_email="${WP_ADMIN_EMAIL:?WP_ADMIN_EMAIL must be set in .env}" \
      --skip-email

    echo "[${script_name}] Configuring basic WordPress site options…"
    rt option update home "$WP_SITE_URL" --perform_readiness_check >/dev/null || true
    rt option update siteurl "$WP_SITE_URL" >/dev/null || true

    nuke_content

    # Rely on baked-in theme files; only activate (do not install/download).
    if rt theme is-installed etal-core >/dev/null 2>&1; then

    script_name="${script_name:-$(basename "${BASH_SOURCE[0]}")}"
    ok
  fi
}

      printf '%s\n' "$loc"
      ok
    fi
  fi
  die
}

#!/usr/bin/env bash
# Purpose: Migrate Portmason deployment-directory terminology from cm/ to deploy/.
# Usage:
#   pm-migrate-cm-to-deploy [PORTMASON_SHARE] [REPO_ROOT ...]
#
# Behavior:
# - Renames Portmason artifacts:
#     pm-sync-cm-from-root      -> pm-sync-deploy-from-root
#     pm-cm-env-mapping.json    -> pm-deploy-env-mapping.json
# - Updates Portmason script references and terminology.
# - Adds the root-level DEPLOY_ENV=dev exception to project identity initialization.
# - Renames an existing REPO_ROOT/cm directory to REPO_ROOT/deploy when repo roots are supplied.
# - Is safe to rerun. Conflicting old/new paths fail loudly.

set -euo pipefail

usage() {
  cat >&2 <<'USAGE'
Usage:
  pm-migrate-cm-to-deploy [PORTMASON_SHARE] [REPO_ROOT ...]

Defaults:
  PORTMASON_SHARE defaults to $PORTMASON_SHARE when set, otherwise:
    /usr/local/share/portmason/bin

Examples:
  pm-migrate-cm-to-deploy
  pm-migrate-cm-to-deploy /usr/local/share/portmason/bin
  pm-migrate-cm-to-deploy /usr/local/share/portmason/bin ~/code/project-a ~/code/project-b
USAGE
}

log() {
  printf '%s\n' "$*" >&2
}

fatal() {
  log "ERROR: $*"
  exit 1
}

move_if_needed() {
  local old_path="${1:?old path required}"
  local new_path="${2:?new path required}"

  if [[ -e "${old_path}" && -e "${new_path}" ]]; then
    fatal "Both old and new paths exist; resolve manually: ${old_path} and ${new_path}"
  fi

  if [[ -e "${old_path}" ]]; then
    mv -- "${old_path}" "${new_path}"
    log "Renamed: ${old_path} -> ${new_path}"
  elif [[ -e "${new_path}" ]]; then
    log "Already renamed: ${new_path}"
  else
    log "Not present; skipping rename: ${old_path}"
  fi
}

replace_literal_in_file() {
  local file="${1:?file required}"
  local old="${2:?old text required}"
  local new="${3:?new text required}"

  [[ -f "${file}" ]] || return 0
  grep --quiet --fixed-strings -- "${old}" "${file}" || return 0

  OLD_TEXT="${old}" NEW_TEXT="${new}" perl -0pi -e '
    BEGIN {
      $old = $ENV{"OLD_TEXT"};
      $new = $ENV{"NEW_TEXT"};
    }
    s/\Q$old\E/$new/g;
  ' -- "${file}"

  log "Updated: ${file}: ${old} -> ${new}"
}

replace_gate_in_pm_helpers() {
  local file="${1:?pm-helpers path required}"
  local old='elif [[ "${IN_CONTAINER:-false}" != "true" ]] && ! is_env_dir; then'
  local new='elif [[ "${IN_CONTAINER:-false}" != "true" ]] \
      && [[ "${DEPLOY_ENV:-}" != "dev" ]] \
      && ! is_env_dir; then'

  [[ -f "${file}" ]] || fatal "Required file not found: ${file}"

  if grep --quiet --fixed-strings -- '&& [[ "${DEPLOY_ENV:-}" != "dev" ]]' "${file}"; then
    log "DEV exception already present: ${file}"
    return 0
  fi

  grep --quiet --fixed-strings -- "${old}" "${file}" \
    || fatal "Unable to locate expected identity gate in ${file}"

  replace_literal_in_file "${file}" "${old}" "${new}"
}

migrate_repo_dir() {
  local repo_root="${1:?repo root required}"
  local old_dir new_dir

  [[ -d "${repo_root}" ]] || fatal "Repository root not found: ${repo_root}"

  old_dir="${repo_root%/}/cm"
  new_dir="${repo_root%/}/deploy"
  move_if_needed "${old_dir}" "${new_dir}"
}

main() {
  if [[ "${1:-}" == '--help' || "${1:-}" == '-h' ]]; then
    usage
    return 0
  fi

  local portmason_share
  if (( $# > 0 )) && [[ -d "${1}" ]] && [[ -e "${1}/pm-helpers" || -e "${1}/pm-sync-cm-from-root" || -e "${1}/pm-sync-deploy-from-root" ]]; then
    portmason_share="${1}"
    shift
  else
    portmason_share="${PORTMASON_SHARE:-/usr/local/share/portmason/bin}"
  fi

  [[ -d "${portmason_share}" ]] || fatal "Portmason share directory not found: ${portmason_share}"
  command -v perl >/dev/null 2>&1 || fatal "perl is required."

  move_if_needed \
    "${portmason_share}/pm-sync-cm-from-root" \
    "${portmason_share}/pm-sync-deploy-from-root"

  move_if_needed \
    "${portmason_share}/pm-cm-env-mapping.json" \
    "${portmason_share}/pm-deploy-env-mapping.json"

  local -a files=(
    "${portmason_share}/pm-setup"
    "${portmason_share}/pm-helpers"
    "${portmason_share}/pm-helpers-base"
    "${portmason_share}/pm-sync-deploy-from-root"
    "${portmason_share}/pm-provision-gcp"
    "${portmason_share}/pm-ensure-sqlinstance-gcp"
  )

  local file
  for file in "${files[@]}"; do
    [[ -f "${file}" ]] || continue

    replace_literal_in_file "${file}" 'pm-sync-cm-from-root' 'pm-sync-deploy-from-root'
    replace_literal_in_file "${file}" 'pm-cm-env-mapping.json' 'pm-deploy-env-mapping.json'
    replace_literal_in_file "${file}" 'get_cm_env_value' 'get_deploy_env_value'
    replace_literal_in_file "${file}" 'injected_cm_env_is_set' 'injected_deploy_env_is_set'
    replace_literal_in_file "${file}" 'injected_cm_env' 'injected_deploy_env'
    replace_literal_in_file "${file}" 'cm_sync_from_root' 'deploy_sync_from_root'
    replace_literal_in_file "${file}" 'cm/qas' 'deploy/qas'
    replace_literal_in_file "${file}" 'cm/prd' 'deploy/prd'
    replace_literal_in_file "${file}" "parent is 'cm'" "parent is 'deploy'"
    replace_literal_in_file "${file}" "parent must be 'cm'" "parent must be 'deploy'"
    replace_literal_in_file "${file}" 'logs|capture|cm)' 'logs|capture|deploy)'
    replace_literal_in_file "${file}" 'logs, capture, cm, .git' 'logs, capture, deploy, .git'
    replace_literal_in_file "${file}" 'per cm_env' 'per deploy_env'
    replace_literal_in_file "${file}" '"cm"' '"deploy"'
    replace_literal_in_file "${file}" '/cm/' '/deploy/'
  done

  replace_gate_in_pm_helpers "${portmason_share}/pm-helpers"

  for file in "${files[@]}"; do
    [[ -f "${file}" ]] || continue
    bash -n "${file}" || fatal "Bash syntax validation failed: ${file}"
  done

  local repo_root
  for repo_root in "$@"; do
    migrate_repo_dir "${repo_root}"
  done

  log "Complete: Portmason deployment-directory convention is deploy/<environment>."
  log "DEV remains the root-level exception through DEPLOY_ENV=dev."
}

main "$@"

===== pm-process-site-partials =====
#!/usr/bin/env python3
import os
import re
import stat
import subprocess
import sys
from pathlib import Path

SITE_DIR = Path(os.environ.get("SITE_DIR", Path.cwd())).resolve()
INDEX_FILE = SITE_DIR / "index.html"
PARTIALS_DIR = SITE_DIR / "partials"
HOOKS_DIR = PARTIALS_DIR / "hooks"

BLOCK_RE = re.compile(
    r"<!--\s*PM:([A-Za-z0-9_-]+)\s*-->(.*?)<!--\s*/PM:\1\s*-->",
    re.DOTALL,
)

def normalize_tag(tag: str) -> str:
    return tag.strip().lower()

def is_executable(path: Path) -> bool:
    return path.is_file() and bool(path.stat().st_mode & stat.S_IXUSR)

def render_tag(tag: str) -> str:
    normalized = normalize_tag(tag)
    hook = HOOKS_DIR / normalized
    partial = PARTIALS_DIR / f"{normalized}.html"

    if is_executable(hook):
        result = subprocess.run(
            [str(hook)],
            cwd=str(SITE_DIR),
            text=True,
            stdout=subprocess.PIPE,
            stderr=None,
            check=False,
        )
        if result.returncode != 0:
            raise RuntimeError(f"Hook failed for PM:{tag}: {hook} exited {result.returncode}")
        return result.stdout.rstrip() + "\n"

    if partial.is_file():
        return partial.read_text(encoding="utf-8").rstrip() + "\n"

    raise FileNotFoundError(f"No hook or partial found for PM:{tag}; expected {hook} or {partial}")

def replace_block(match: re.Match) -> str:
    tag = match.group(1)
    rendered = render_tag(tag)
    return f"<!-- PM:{tag} -->\n{rendered}<!-- /PM:{tag} -->"

def main() -> int:
    if not INDEX_FILE.is_file():
        print(f"ERROR: index.html not found: {INDEX_FILE}", file=sys.stderr)
        return 1

    source = INDEX_FILE.read_text(encoding="utf-8")
    rendered, count = BLOCK_RE.subn(replace_block, source)

    if count == 0:
        print(f"ERROR: no PM partial blocks found in {INDEX_FILE}", file=sys.stderr)
        return 1

    INDEX_FILE.write_text(rendered, encoding="utf-8")
    os.chmod(INDEX_FILE, 0o644)

    print(f"Processed {count} partial block(s) in {INDEX_FILE}")
    return 0

if __name__ == "__main__":
    raise SystemExit(main())

bind_service_account(){
    local sa_email="${1:?sa email required}"

    local role

    # storage bucket
    if [[ -n "${GCS_BUCKET:-}" ]]; then
        log "Ensure GCS_BUCKET access for '${sa_email}'"
        role="storage.objectAdmin"
        local resource=="//storage.googleapis.com/projects/_/buckets/${GCS_BUCKET}"
        if ! iam_cache_has_binding "${resource}" "${sa_email}" "${role}"; then
            gcloud storage buckets add-iam-policy-binding "gs://${GCS_BUCKET}" \
                --member="serviceAccount:${sa_email}" \
                --role="roles/${role}" >/dev/null || true
            iam_cache_add_binding "${resource}" "${sa_email}" "${role}"
        fi
    fi

    # secrets
    log "Ensure secret access for '${sa_email}'"
    role="secretmanager.secretAccessor"
    while IFS= read -r secret_name; do
        local resource="//secretmanager.googleapis.com/projects/${DEPLOYMENT_ID}/secrets/${secret_name}"
        if ! iam_cache_has_binding "${resource}" "${sa_email}" "${role}"; then
            gcloud secrets add-iam-policy-binding "$secret_name" \
                --member="serviceAccount:${sa_email}" \
                --role="roles/secretmanager.secretAccessor" >/dev/null || true
            iam_cache_add_binding "${resource}" "${sa_email}" "${role}"
        fi
    done < <(get_secret_names_list)
}

ensure_apis() {
    log "Verify/Enable required APIs"

    if [[ -z "${GCLOUD_ENABLED_SERVICE_NAMES:-}" ]]; then

        fi
        export GCLOUD_ENABLED_SERVICE_NAMES
        log "GCLOUD_ENABLED_SERVICE_NAMES=${GCLOUD_ENABLED_SERVICE_NAMES}"
    fi

    for api in \
        run.googleapis.com \
        cloudbuild.googleapis.com \
        artifactregistry.googleapis.com \
        sqladmin.googleapis.com \
        secretmanager.googleapis.com \
        iamcredentials.googleapis.com \
        cloudresourcemanager.googleapis.com
    do
        if ! printf '%s\n' "$GCLOUD_ENABLED_SERVICE_NAMES" | grep -Fxq "$api"; then
            log "Enabling API: $api"

                    GCLOUD_ENABLED_SERVICE_NAMES+=$'\n'"$api"
                fi
            fi
        else
            log "API already enabled: $api"
        fi
    done
}

ensure_env_tag() {
    local base_project_id=github-archive-480721
    local base_parent="projects/$base_project_id"
    local keyname='environment'
    local keyvalue="$(get_deploy_env_value "$DEPLOY_ENV" long)" || die "get_deploy_env_value failed with '$DEPLOY_ENV'"
    keyvalue="${keyvalue^}"

    local target_project_number="$(jq -er '.projectNumber' <<<"$GCLOUD_PROJECT_INFO_JSON")"
    local target_parent="//cloudresourcemanager.googleapis.com/projects/$target_project_number"

    log "Ensuring project has an '$keyname' tag"

    # ---- Find Tag Key NAME (tagKeys/<id>) scoped to base project ----
    # IMPORTANT: Use the base project scope to avoid collisions.
    local tag_key_name="$(
    gcloud resource-manager tags keys list \
        --parent="$base_parent" \
        --filter="shortName=$keyname" \
        --format='value(name)' \
        --limit=1
    )"

    [[ -n "$tag_key_name" ]] || { error "Tag Key '${keyname}' not found under ${base_parent}"; return 1; }
    [[ "$(printf '%s\n' "$tag_key_name" | grep -c .)" -le 1 ]] || { error "Multiple Tag Keys matched '${keyname}' under ${base_parent}"; return 2; }

    # ---- Find Tag Value NAME (tagValues/<id>) under the key ----
    local tag_value_name="$(
    gcloud resource-manager tags values list     \
        --parent="$tag_key_name" \
        --filter="shortName=$keyvalue" \
        --format='value(name)' \
        --limit=1
    )"

    if [[ -z "$tag_value_name" ]]; then
        tag_value_name="$(
            gcloud resource-manager tags values create "${keyvalue}" \
                --parent="${tag_key_name}" \
                --format="value(name)"
        )" || { error "Failed to create tag value ${keyvalue}"; return 3; }
    fi

    [[ -n "$tag_value_name" ]] || { error "Tag Value '${keyvalue}' not found under Tag Key '${keyname}'"; return 3; }
    [[ "$(printf '%s\n' "$tag_value_name" | grep -c .)" -le 1 ]] || { error "Multiple Tag Values matched '${keyvalue}' under ${tag_key_name}"; return 4; }

    # Bindings list returns tagValue (tagValues/...). To figure out if that value belongs to TAG_KEY_NAME,
    # we have to look up each value's parent (which is tagKeys/<id>).

    local existing_tag_values="$(
        gcloud resource-manager tags bindings list \
        --parent="$target_parent" \
        --format='value(tagValue)' 2>/dev/null || true
    )"

    if printf '%s\n' "$existing_tag_values" | grep -Fxq "$tag_value_name"; then
        log "Tag already bound: $tag_value_name"
        return 0
    fi

    local existing_parent_key
    if [[ -n "$existing_tag_values" ]]; then
        while IFS= read -r existing_tv; do
            [[ -z "$existing_tv" ]] && continue

            # Determine which key this tag value belongs to
            existing_parent_key="$(
                gcloud resource-manager tags values describe "$existing_tv" --format='value(parent)' 2>/dev/null || true
            )"

            if [[ "$existing_parent_key" = "$tag_key_name" && "$existing_tv" != "$tag_value_name" ]]; then
                log "Unbinding old value for key '$keyname': $existing_tv"
                # Delete needs both parent + tag-value
                gcloud resource-manager tags bindings delete \
                --parent="$target_parent" \
                --tag-value="$existing_tv" >/dev/null
            fi
        done <<< "$existing_tag_values"
    fi

    # ---- Create binding (treat ALREADY_EXISTS as success) ----
    set +e
    local out="$(gcloud resource-manager tags bindings create \
        --tag-value="$tag_value_name" \
        --parent="$target_parent" 2>&1
    )"
    local rc=$?
    set -e

    if [[ $rc -eq 0 ]] || printf '%s' "$out" | grep -qiE 'ALREADY_EXISTS|already exists'; then
      log "Tag ensured: ${keyname}=${keyvalue}"
      return 0
    fi

    error "failed to create tag binding (rc=$rc): $out" >&2
    return "$rc"
}

        log "linking to ${BILLING_ACCOUNT}"

        export GCLOUD_PROJECT_INFO_JSON
        log "GCLOUD_PROJECT_INFO_JSON=${GCLOUD_PROJECT_INFO_JSON}"

        export GCLOUD_BILLING_JSON
#    else
#        die "unknown (auth/perm/config issue)" 2
    fi
    log "GCLOUD_BILLING_JSON=${GCLOUD_BILLING_JSON}"

ensure_project_role() {
    local role="${1:?role required}"
    local member="user:${GCLOUD_ACTIVE_USER:?GCLOUD_ACTIVE_USER required}"

    # Normalize role to avoid INVALID_ARGUMENT from stray whitespace/CR
    role="${role//$'\r'/}"
    role="$(xargs <<<"$role")"

    if [[ ! "$role" =~ ^(roles|projects/[a-z0-9-]+/roles|organizations/[0-9]+/roles)/[A-Za-z0-9_.-]+$ ]]; then
        die "Bad role string (expected roles/foo or projects/*/roles/foo or organizations/*/roles/foo): '$role'"
    fi

    # Fetch IAM policy once for performance (JSON, stable for parsing)
    if [[ -z "${GCLOUD_POLICY_JSON:-}" ]]; then
        if ! GCLOUD_POLICY_JSON="$(gcloud projects get-iam-policy "${DEPLOYMENT_ID}" --format=json 2>/dev/null)"; then
            GCLOUD_POLICY_JSON='{}'
        fi
        export GCLOUD_POLICY_JSON
        log "GCLOUD_POLICY_JSON loaded"
    fi

    # Exact membership check
    if jq -e --arg r "$role" --arg m "$member" \
        'any(.bindings[]?; .role == $r and any(.members[]?; . == $m))' \
        >/dev/null 2>&1 <<<"$GCLOUD_POLICY_JSON"; then
        log "  = already has ${role}"
        return 0
    fi

    # Fast path: try add-iam-policy-binding
    if gcloud projects add-iam-policy-binding "${DEPLOYMENT_ID}" \
        --member="$member" \
        --role="$role" >/dev/null 2>&1; then
        log "  + granted ${role} (add-iam-policy-binding)"
    else
        # Slow-but-reliable path for conditional policies:
        # Update cached policy JSON and set it back.
        log "  ! add-iam-policy-binding failed; applying via set-iam-policy"

        GCLOUD_POLICY_JSON="$(
        jq -c --arg r "$role" --arg m "$member" '
            .bindings = (.bindings // [])
            | (if any(.bindings[]?; .role == $r) then .
            else .bindings += [{role:$r, members:[]}] | .
            end)
            | .bindings = (
                .bindings
                | map(
                    if .role != $r then .
                    else .members = ((.members // []) | (if index($m) then . else . + [$m] end))
                    end
                )
            )
        ' <<<"$GCLOUD_POLICY_JSON" 2>/dev/null || printf '%s' '{}'
        )"

        local tmp_policy
        tmp_policy="$(mktemp)"
        printf '%s\n' "$GCLOUD_POLICY_JSON" >"$tmp_policy"

        gcloud projects set-iam-policy "${DEPLOYMENT_ID}" "$tmp_policy" >/dev/null \
        || { rm -f "$tmp_policy"; die "Failed to set IAM policy for ${DEPLOYMENT_ID}"; }

        rm -f "$tmp_policy"
        export GCLOUD_POLICY_JSON
        log "  + granted ${role} (set-iam-policy)"
    fi

    # Keep cache accurate for the rest of this run (manual patch is harmless even on fast path)
    GCLOUD_POLICY_JSON="$(
        jq -c --arg r "$role" --arg m "$member" '
        .bindings = (.bindings // [])
        | (if any(.bindings[]?; .role == $r) then .
            else .bindings += [{role:$r, members:[]}] | .
            end)
        | .bindings = (
            .bindings
            | map(
                if .role != $r then .
                else .members = ((.members // []) | (if index($m) then . else . + [$m] end))
                end
                )
            )
        ' <<<"$GCLOUD_POLICY_JSON" 2>/dev/null || printf '%s' '{}'
    )"
    export GCLOUD_POLICY_JSON

    return 0
}

ensure_scheduler() {
  if [[ -z "${SCHED_NAME:-}" ]]; then
     log "SCHED_NAME is undefined; skipping Cloud Scheduler creation"
    return 0
  fi

  local run_url="https://run.googleapis.com/v2/projects/${DEPLOYMENT_ID}/locations/${REGION}/jobs/${RUN_JOB}:run"
  local body='{}'

  log "Ensuring Cloud Scheduler job exists: ${SCHED_NAME}"
  if gcloud scheduler jobs describe "$SCHED_NAME" --location "$REGION" >/dev/null 2>&1; then
    log "Updating Cloud Scheduler job: ${SCHED_NAME}"
    gcloud scheduler jobs update http "$SCHED_NAME" \
      --location "$REGION" \
      --schedule "$SCHEDULE_CRON" \
      --time-zone "$SCHED_TIMEZONE" \
      --uri "$run_url" \
      --http-method POST \
      --update-headers "Content-Type=application/json" \
      --message-body "$body" \
      --oauth-service-account-email "$SCHED_SA" \
      >/dev/null
  else
    log "Creating Cloud Scheduler job: ${SCHED_NAME}"
    gcloud scheduler jobs create http "$SCHED_NAME" \
      --location "$REGION" \
      --schedule "$SCHEDULE_CRON" \
      --time-zone "$SCHED_TIMEZONE" \
      --uri "$run_url" \
      --http-method POST \
      --headers "Content-Type=application/json" \
      --message-body "$body" \
      --oauth-service-account-email "$SCHED_SA" \
      >/dev/null
  fi

    return 0
}

iam_cache_add_binding() {
  local role="${1:?role required}"       # e.g. roles/storage.objectAdmin
  local member="${2:?member required}"   # e.g. user:... serviceAccount:...

  # Ensure cache is at least an object
  if [[ -z "${GCLOUD_POLICY_JSON:-}" ]]; then
    GCLOUD_POLICY_JSON='{}'
  fi

  # Normalize to valid JSON object; if invalid, reset
  if ! jq -e 'type=="object"' >/dev/null 2>&1 <<<"$GCLOUD_POLICY_JSON"; then
    GCLOUD_POLICY_JSON='{}'
  fi

  GCLOUD_POLICY_JSON="$(
    jq -c --arg r "$role" --arg m "$member" '
      .bindings = (.bindings // [])

      # ensure role binding exists
      | (if any(.bindings[]?; .role == $r) then .
         else .bindings += [{role:$r, members:[]}] | .
         end)

      # ensure member exists in that role binding
      | .bindings = (
          .bindings
          | map(
              if .role != $r then .
              else .members = ((.members // []) | (if index($m) then . else . + [$m] end)) | .
              end
            )
        )
    ' <<<"$GCLOUD_POLICY_JSON" 2>/dev/null || printf '%s' '{}'
  )"

#!/usr/bin/env bash
set -euo pipefail

: "${CUR_SCRIPT:=$(basename "${BASH_SOURCE[0]}")}"
export CUR_SCRIPT
: "${PORTMASON_SHARE:=$(cd -- "$(dirname -- "${BASH_SOURCE[0]}")" && pwd -P)}"

# Helpers
is_localtest_me_cn() {
  local cn="$1"
  cn="${cn,,}"
  cn="${cn%:}"
  [[ "$cn" == "localtest.me" ]] || [[ "$cn" == *.localtest.me ]]
}

ensure_mkcert_installed() {
  if command -v mkcert >/dev/null 2>&1; then
    return 0
  fi

  log "mkcert not found; attempting install via apt-get (requires sudo)..."

  if ! command -v apt-get >/dev/null 2>&1; then
    die "mkcert not found and apt-get is not available. Install mkcert manually, then rerun."
  fi
  if ! command -v sudo >/dev/null 2>&1; then
    die "mkcert not found and sudo is not available. Install mkcert manually, then rerun."
  fi

  sudo apt-get update
  sudo apt-get install -y mkcert libnss3-tools
}

write_site_cert_yml() {
  local yml_path="$1"
  local cert_basename="$2"
  local key_basename="$3"

  cat > "$yml_path" <<YAML
tls:
  certificates:
    - certFile: /certs/${cert_basename}
      keyFile:  /certs/${key_basename}
YAML
}

main() {
    log "Start provisioning local ..."

    # If URI_SCHEME isn't https (case-insensitive), SSL is not required.
    local scheme="${URI_SCHEME:-http}"
    scheme="${scheme,,}"
    scheme="${scheme%:}"   # allow "https:" too

    local val="${FORCE_SSL_CERT:-}"
    val="${val,,}"  # normalize to lowercase
    if [[ $val == 0 || $val == no ]]; then
        log "End provisioning local"
        return 0
    fi

    if [[ "$scheme" != "https" ]]; then
        log "End provisioning local"
        return 0
    fi

    local dir="${TRAEFIK_DIR:-${SSL_CERT_DIR:-}}"
    : "${dir?TRAEFIK_DIR or SSL_CERT_DIR must be set.}"

    local cert_dir="${dir}/certs"
    local dyn_dir="${dir}/dynamic"
    local certs_out="${dir}/out-certs"

    # Prefer an explicit cn override; otherwise use APP_HOST.
    local cn="${CN:-${APP_HOST:-}}"
    : "${cn?CN or APP_HOST must be set.}"

    mkdir -p "$cert_dir" "$certs_out" "$dyn_dir"

    pushd "$SITE_DIR" >/dev/null

    log "HTTPS enabled (URI_SCHEME=$scheme). cn=$cn"

    if is_localtest_me_cn "$cn"; then
        # --- Preferred dev path: mkcert + trusted local CA + wildcard certs for localtest.me ---
        ensure_mkcert_installed

        # Install mkcert root CA into host trust stores (idempotent).
        # This is what makes Chrome stop showing ERR_CERT_AUTHORITY_INVALID.
        log "Ensuring mkcert local CA is installed/trusted (mkcert -install)..."
        if ! mkcert -install >/dev/null 2>&1; then
            warn "mkcert -install failed (trust store may not be updated). If you're using Windows Chrome, import rootCA.pem into Windows Trusted Roots."
        fi

        local crt_path="$cert_dir/localtest.me.crt"
        local key_path="$cert_dir/localtest.me.key"
        local site_cert_yml="$certs_out/localtest.me.yml"

        if [[ ! -s "$crt_path" || ! -s "$key_path" || ! -s "$site_cert_yml" ]]; then
            log "Generating/refreshing wildcard cert for *.localtest.me"
            mkcert -cert-file "$crt_path" -key-file "$key_path" "localtest.me" "*.localtest.me" >/dev/null
        else
            log "Wildcard cert for *.localtest.me is already installed"
            popd >/dev/null
            log "End provisioning local"
            return 0
        fi

        if [[ ! -f "$crt_path" || ! -f "$key_path" ]]; then
            die "Expected mkcert outputs not found:" \
            "   $crt_path" \
            "   $key_path"
        fi

        if [[ ! -f "$site_cert_yml" ]]; then
            write_site_cert_yml "$site_cert_yml" "$(basename "$crt_path")" "$(basename "$key_path")"
        fi

        # Trigger Traefik reload (touch an existing dyn file if present; otherwise touch the site cert file)
        touch "$dyn_dir/certificates.yml" 2>/dev/null || touch "$site_cert_yml"

        log "TLS ready (mkcert):"
        log "  cert: $crt_path"
        log "  key : $key_path"
        log "  dyn : $site_cert_yml"
        log "NOTE: If Chrome was already open, restart Chrome to pick up the newly trusted CA."
    else
        # --- Fallback path: pm-gen-sslcert (if you use non-localtest.me domains) ---
        if ! command -v pm-gen-sslcert >/dev/null 2>&1; then
            die "pm-gen-sslcert not found in PATH."
        fi

        local -a sslcert_args=()
        if [[ "$val" == 1 || "$val" == yes ]]; then
            sslcert_args+=( -f )
        fi

        log "Using pm-gen-sslcert for cn=$cn"
        pm-gen-sslcert -o "$cert_dir" -cn "$cn" "${sslcert_args[@]}"

        STEM="$(echo "$cn" | sed 's/\*/wildcard/g; s/[[:space:]]\+/_/g; s#/#_#g')"
        FULLCHAIN="$cert_dir/$STEM.fullchain.crt"
        KEY="$cert_dir/$STEM.key"

        if [[ ! -f "$FULLCHAIN" || ! -f "$KEY" ]]; then
            die "Expected cert outputs not found:" \
            "   $FULLCHAIN" \
            "   $KEY"
        fi

        site_cert_yml="$certs_out/$STEM.yml"
        if [[ ! -f "$site_cert_yml" ]]; then
            write_site_cert_yml "$site_cert_yml" "$(basename "$FULLCHAIN")" "$(basename "$KEY")"
        fi

        touch "$dyn_dir/certificates.yml" 2>/dev/null || touch "$site_cert_yml"

        log "TLS ready:"
        log "  cert: $FULLCHAIN"
        log "  key : $KEY"
        log "  dyn : $site_cert_yml"
    fi

    popd >/dev/null
    log "End provisioning local"
}

main "$@"

#!/usr/bin/env bash
set -euo pipefail

: "${CUR_SCRIPT:=$(basename "${BASH_SOURCE[0]}")}"
export CUR_SCRIPT
: "${PORTMASON_SHARE:="$(cd -- "$(dirname -- "${BASH_SOURCE[0]}")" && pwd -P)"}"
. "${PORTMASON_SHARE}/pm-helpers" "$@"

artifacts_dir__prepare() {
  local d="${1:?}"
  rm -rf "$d"
  mkdir -p "$d"
}

artifacts_id__compute() {
  # Prefer explicit BUILD_ID, else timestamp + short git sha if available.
  if [[ -n "${BUILD_ID:-}" ]]; then
    printf '%s' "${BUILD_ID}"
    return 0
  fi

  local ts sha
  ts="$(date -u +%Y%m%dT%H%M%SZ)"
  if command -v git >/dev/null 2>&1 && git rev-parse --is-inside-work-tree >/dev/null 2>&1; then
    sha="$(git rev-parse --short HEAD 2>/dev/null || true)"
  fi
  sha="${sha:-nogit}"

  printf '%s-%s' "$ts" "$sha"
}

artifacts_paths__derive() {
    : "${GCS_BUCKET:?}"
    GCS_PREFIX="artifacts"

    ARTIFACTS_LOCAL_DIR="${ARTIFACTS_LOCAL_DIR:-$(pwd -P)/artifacts}"
    ARTIFACTS_BUILD_ID="$(artifacts_id__compute)"
    ARTIFACTS_GCS_BASE="gs://${GCS_BUCKET}/${GCS_PREFIX}"
    ARTIFACTS_GCS_BUILD="${ARTIFACTS_GCS_BASE}/${ARTIFACTS_BUILD_ID}"
    ARTIFACTS_GCS_CURRENT="${ARTIFACTS_GCS_BASE}/current"

    export GCS_PREFIX GCS_BUCKET ARTIFACTS_LOCAL_DIR ARTIFACTS_BUILD_ID ARTIFACTS_GCS_BASE ARTIFACTS_GCS_BUILD ARTIFACTS_GCS_CURRENT
}

artifacts_stage__copy_app() {
  local dest_root="${1:?}"
  local src="mgt-scripts"
  local dest="${dest_root}/app"

  [[ -d "$src" ]] || die "Missing directory: ${src}"
  mkdir -p "$dest"
  cp -a "${src}/." "$dest/"
}

artifacts_stage__copy_portmason() {
    # Usage:
    #   artifacts_stage__copy_portmason <dest_root> [exclude_pattern...]
    #
    # Exclude patterns are rsync-style, relative to the source root:
    #   "cache/"          (exclude subdir)
    #   "secrets.env"     (exclude file)
    #   "*.log"           (exclude glob)
    #
    # Example:
    #   artifacts_stage__copy_portmason "$dest_root" "cache/" "*.tmp" "node_modules/"

    local dest_root="${1:?}"
    shift || true

    local src="${PORTMASON_SHARE}"
    local dest="${dest_root}/portmason"

    [[ -d "$src" ]] || die "Missing directory: ${src}"
    mkdir -p "$dest"

    # -a = archive; -L = copy the symlink targets as real files/dirs
    local -a rsync_cmd=(rsync -aL --delete)

    # Add excludes (if any)
    local pat
    for pat in "$@"; do
    [[ -n "$pat" ]] || continue
    rsync_cmd+=(--exclude "$pat")
    done

    rsync_cmd+=("${src}/" "${dest}/")

    # Optional: log the exact command
    local _print
    printf -v _print '%q ' "${rsync_cmd[@]}"
    log "CMD: ${_print}"

    "${rsync_cmd[@]}"
}

artifacts_upload__gcs() {
  local artifact_tgz="${1:?}"

  have gcloud || die "Missing required command: gcloud"

  # Upload immutable build folder
  log "Uploading build artifacts -> ${ARTIFACTS_GCS_BUILD}/"
  gcloud storage cp "${artifact_tgz}" "${artifact_tgz}.sha256" "${ARTIFACTS_GCS_BUILD}/" >/dev/null

  # Update current/ pointer (replace)
  log "Updating current pointer -> ${ARTIFACTS_GCS_CURRENT}/"
  gcloud storage cp "${artifact_tgz}" "${artifact_tgz}.sha256" "${ARTIFACTS_GCS_CURRENT}/" >/dev/null
}

artifacts_verify__print() {
  local artifacts_root="${1:?}"
  local runtime_tgz="./runtime.tgz"

  log "Local verification:"
  log "  $(wc -c < "${runtime_tgz}") bytes  sha256=$(cut -d' ' -f1 "${runtime_tgz}.sha256")  ${runtime_tgz}"

  log "Published:"
  log "  Build ID: ${ARTIFACTS_BUILD_ID}"
  log "  Build:    ${ARTIFACTS_GCS_BUILD}/"
  log "  Current:  ${ARTIFACTS_GCS_CURRENT}/"
}

main() {
  artifacts_paths__derive

  have cp  || die "Missing required command: cp"
  have rm  || die "Missing required command: rm"
  have tar || die "Missing required command: tar"
  have sha256sum || die "Missing required command: sha256sum"

  log "Preparing local artifacts directory: ${ARTIFACTS_LOCAL_DIR}"
  artifacts_dir__prepare "${ARTIFACTS_LOCAL_DIR}"

  log "Staging portmason -> ${ARTIFACTS_LOCAL_DIR}/portmason (from /usr/local/share/portmason/bin/)"
  artifacts_stage__copy_portmason "${ARTIFACTS_LOCAL_DIR}" "archive" "cagent" "*cosign*" "*log" # Exclude archive/, cagent, cosign files, log files

  log "Staging app -> ${ARTIFACTS_LOCAL_DIR}/app (from mgt-scripts/)"
  artifacts_stage__copy_app "${ARTIFACTS_LOCAL_DIR}"

  log "Packaging tgz builds"
  runtime_tgz__build "${ARTIFACTS_LOCAL_DIR}"

  log "Publishing to GCS"
  artifacts_upload__gcs "./runtime.tgz"

  artifacts_verify__print "${ARTIFACTS_LOCAL_DIR}"
}

runtime_tgz__build() {
  local artifacts_root="${1:?}"

  local runtime_src_dir="${artifacts_root}"

  [[ -d "$runtime_src_dir" ]] || die "Missing staged dir: $runtime_src_dir"

  # Build outputs in cwd (next to artifacts/), but sourced from artifacts/
  local runtime_tgz="./runtime.tgz"

  # Ensure expected executables are executable if present
  [[ -f "${runtime_src_dir}/portmason" ]] && chmod +x "${runtime_src_dir}/portmason" || true

  # runtime.tgz should contain top folders "portmason/" and "app/"
  tar -C "${runtime_src_dir}" -czf "${runtime_tgz}" .

  sha256sum "${runtime_tgz}" > "${runtime_tgz}.sha256"

  log "Built: ${runtime_tgz}"
}

main "$@"

#!/usr/bin/env bash
set -euo pipefail

: "${CUR_SCRIPT:=$(basename "${BASH_SOURCE[0]:-${0:-}}")}"
export CUR_SCRIPT

: "${PORTMASON_SHARE:="$(cd -- "$(dirname -- "${BASH_SOURCE[0]:-${0:-}}")" && pwd -P)"}"

# Purpose: Recover a missing static partial from one managed HTML region.
# Usage:   pm-recover-static-partial-from-page PAGE MARKER [OUTPUT_FILE]
# Example: pm-recover-static-partial-from-page index.html CATALOG-BRANDS-FULL-LIST

usage() {
  cat >&2 <<'USAGE'
Usage:
  pm-recover-static-partial-from-page PAGE MARKER [OUTPUT_FILE]

Arguments:
  PAGE         HTML page containing one balanced Portmason managed region.
  MARKER       Marker name without PM: prefix, for example CATALOG-BRANDS-FULL-LIST.
  OUTPUT_FILE  Optional output path. Defaults to partials/<lowercase-marker>.html.
USAGE
}

if [[ "${1:-}" == '--help' || "${1:-}" == '-h' ]]; then
  usage
  exit 0
fi

if (( $# < 2 || $# > 3 )); then
  usage
  exit 2
fi

page="$1"
marker="$2"
output_file="${3:-}"

[[ -f "$page" ]] || { printf 'ERROR: page not found: %s\n' "$page" >&2; exit 2; }
[[ "$marker" =~ ^[A-Z0-9_-]+$ ]] || { printf 'ERROR: invalid marker: %s\n' "$marker" >&2; exit 2; }

if [[ -z "$output_file" ]]; then
  slug="$(printf '%s' "$marker" | tr '[:upper:]' '[:lower:]')"
  output_file="partials/${slug}.html"
fi

mkdir -p -- "$(dirname -- "$output_file")"
tmp_file="$(mktemp)"
trap 'rm -f -- "$tmp_file"' EXIT INT TERM

awk -v marker="$marker" '
  BEGIN {
    start = "<!--[[:space:]]*PM:" marker "[[:space:]]*-->"
    finish = "<!--[[:space:]]*/PM:" marker "[[:space:]]*-->"
    in_region = 0
    starts = 0
    ends = 0
  }

  $0 ~ start {
    starts++
    if (in_region) {
      exit 4
    }
    in_region = 1
    next
  }

  $0 ~ finish {
    ends++
    if (!in_region) {
      exit 5
    }
    in_region = 0
    next
  }

  in_region {
    print
  }

  END {
    if (starts != 1 || ends != 1 || in_region) {
      exit 6
    }
  }
' "$page" > "$tmp_file"

[[ -s "$tmp_file" ]] || { printf 'ERROR: recovered partial is empty for marker: %s\n' "$marker" >&2; exit 3; }

if [[ -f "$output_file" ]] && cmp -s -- "$tmp_file" "$output_file"; then
  printf 'OK: partial already matches managed region: %s\n' "$output_file"
  exit 0
fi

cp -- "$tmp_file" "$output_file"
printf 'RECOVERED: %s -> %s\n' "$marker" "$output_file"

    local file="${PORTMASON_SHARE}/pm-provision-${ADAPTER_CODE}"
    [[ -r "$file" ]] && . "$file" "$@"

    # run the optional adapter setup script
    file="${PORTMASON_SHARE}/pm-setup-${ADAPTER_CODE}"
    [[ -r "$file" ]] && . "$file" "$@"

    # run the optional runtime setup script
    file="${PORTMASON_SHARE}/pm-setup-${RUNTIME_CODE}"
    [[ -r "$file" ]] && . "$file" "$@"

    # if a custom setup file exists, it must handle all deployment tasks for the project,
    # Otherwise, run the standard runtime deploy followed by the standard adapter deploy
    file="${SCRIPT_DIR}/setup"
    if [[ -r "$file" ]]; then
        . "$file" "$@"
        local rc=$?

        case "$rc" in
            0)
                log "Success."
                ;;
            1)
                die "General error:" "$rc"
                ;;
            2)
                die "Misuse of shell builtins or command-specific code." "$rc"
                ;;

                ;;
        esac
    else
        # run the optional runtime deploy script

        # run the optional adapter deploy script

        PM_DEPLOY_SYNC_OWNER_PID="$$" \
        PM_DEPLOY_SYNC_OWNER_HOST="$sync_owner_host" \
            "${PORTMASON_SHARE}/pm-sync-deploy-from-root" --finalize
    fi
}

#!/usr/bin/env bash
set -euo pipefail

: "${CUR_SCRIPT:=$(basename "${BASH_SOURCE[0]:-${0:-}}")}"
export CUR_SCRIPT

: "${PORTMASON_SHARE:="$(cd -- "$(dirname -- "${BASH_SOURCE[0]:-${0:-}}")" && pwd -P)"}"
. "${PORTMASON_SHARE}/pm-helpers" "$@"

STATE_VERSION="1"
STATE_FILE_NAME=".pm-deploy-sync.state"
MANIFEST_FILE_NAME=".pm-deploy-sync.manifest"
IN_PROGRESS_FILE_NAME=".pm-deploy-sync.in-progress"

usage() {
  cat <<'USAGE'
Usage: pm-sync-deploy-from-root [--prepare|--finalize|--status] [--force]

Safely refresh a deploy/<env> snapshot from the authoritative repository root.

Lifecycle:
  --prepare   Validate deploy drift, refresh from root when safe or required,
              and mark setup in progress. This is the default action.
  --finalize  Record the root and deploy baselines after setup succeeds.
  --status    Report synchronization state without changing files.

Refresh rules:
  - No state file: refresh from root.
  - --force explicitly reconstructs the managed deploy surface from root
    without requiring synchronization state to be deleted first.
  - State exists and deploy matches its baseline: refresh automatically when
    the authoritative root changed.
  - State exists and deploy differs from its baseline: stop before copying and
    report added, modified, and removed managed paths.
  - A stale in-progress marker means the previous setup did not finish; refresh
    from root and retry.

Compatibility:
  -o, --overwrite, -f, and --force explicitly authorize a refresh from root.

Environment-local files, generated configuration, logs, staged Portmason files,
and synchronization state are excluded from both copying and drift hashes.
USAGE
}

log_info() {
  printf '[pm-sync-deploy-from-root] %s\n' "$*"
}

log_error() {
  printf '[pm-sync-deploy-from-root] ERROR: %s\n' "$*" >&2
}

is_deploy_snapshot_dir() {
  [[ "$(basename -- "$(dirname -- "$PWD")")" == "deploy" ]]
}

detect_repo_root() {
  local env_dir="$PWD"

  if git -C "$env_dir" rev-parse --show-toplevel >/dev/null 2>&1; then
    git -C "$env_dir" rev-parse --show-toplevel
    return 0
  fi

  (cd -- "$env_dir/../.." && pwd -P)
}

state_path() {
  printf '%s/%s\n' "$PWD" "$STATE_FILE_NAME"
}

manifest_path() {
  printf '%s/%s\n' "$PWD" "$MANIFEST_FILE_NAME"
}

in_progress_path() {
  printf '%s/%s\n' "$PWD" "$IN_PROGRESS_FILE_NAME"
}

owner_pid() {
  printf '%s\n' "${PM_DEPLOY_SYNC_OWNER_PID:-$$}"
}

owner_host() {
  if [[ -n "${PM_DEPLOY_SYNC_OWNER_HOST:-}" ]]; then
    printf '%s\n' "$PM_DEPLOY_SYNC_OWNER_HOST"
  else
    hostname 2>/dev/null || printf 'unknown\n'
  fi
}

iso_utc_now() {
  date -u +'%Y-%m-%dT%H:%M:%SZ'
}

ensure_git_local_ignore() {
  local root_dir="$1"
  local exclude_file pattern

  if ! git -C "$root_dir" rev-parse --git-dir >/dev/null 2>&1; then
    return 0
  fi

  exclude_file="$(git -C "$root_dir" rev-parse --git-path info/exclude)"
  if [[ "$exclude_file" != /* ]]; then
    exclude_file="$root_dir/$exclude_file"
  fi
  pattern='/deploy/*/.pm-deploy-sync.*'

  mkdir -p -- "$(dirname -- "$exclude_file")"
  touch -- "$exclude_file"

  if ! grep -Fqx -- "$pattern" "$exclude_file"; then
    {
      printf '\n# Portmason deploy synchronization state (local only).\n'
      printf '%s\n' "$pattern"
    } >> "$exclude_file"
  fi
}

is_excluded_directory_name() {
  case "$1" in
    .git|.github|deploy|logs|capture|archive|portmason|tests|.artifacts|__pycache__)
      return 0
      ;;
    *)
      return 1
      ;;
  esac
}

is_excluded_file_basename() {
  case "$1" in
    "$STATE_FILE_NAME"|"$MANIFEST_FILE_NAME"|"$IN_PROGRESS_FILE_NAME"|.pm-deploy-sync.*)
      return 0
      ;;
    .project_timestamp|.htaccess|.env|.env.example|.env.generated|*.env|*.env.example|*.env.generated)
      return 0
      ;;
    docker-compose.yml|config.generated.json|Dockerfile*|*.pyc|*.bak|*:Zone.Identifier|.DS_Store|Thumbs.db)
      return 0
      ;;
    *)
      return 1
      ;;
  esac
}

rsync_exclude_args() {
  cat <<'EOF_ARGS'
--exclude=.git/
--exclude=.github/
--exclude=deploy/
--exclude=logs/
--exclude=capture/
--exclude=archive/
--exclude=portmason/
--exclude=tests/
--exclude=.artifacts/
--exclude=__pycache__/
--exclude=.pm-deploy-sync.*
--exclude=.project_timestamp
--exclude=.htaccess
--exclude=.env
--exclude=.env.example
--exclude=.env.generated
--exclude=*.env
--exclude=*.env.example
--exclude=*.env.generated
--exclude=docker-compose.yml
--exclude=config.generated.json
--exclude=Dockerfile*
--exclude=*.pyc
--exclude=*.bak
--exclude=*:Zone.Identifier
--exclude=.DS_Store
--exclude=Thumbs.db
EOF_ARGS
}

build_rsync_args() {
  local -n output_ref="$1"
  local arg

  output_ref=()
  while IFS= read -r arg; do
    [[ -n "$arg" ]] && output_ref+=("$arg")
  done < <(rsync_exclude_args)
}

generate_manifest() {
  local tree_root="$1"
  local output_file="$2"
  local temp_file

  if [[ ! -d "$tree_root" ]]; then
    log_error "manifest root does not exist: $tree_root"
    return 1
  fi

  temp_file="$(mktemp "${output_file}.tmp.XXXXXX")"

  (
    cd -- "$tree_root"

    find . -mindepth 1 \
      \( -type d \( \
          -name .git -o \
          -name .github -o \
          -name deploy -o \
          -name logs -o \
          -name capture -o \
          -name archive -o \
          -name portmason -o \
          -name tests -o \
          -name .artifacts -o \
          -name __pycache__ \
        \) -prune \) -o \
      -print0 \
    | LC_ALL=C sort -z \
    | while IFS= read -r -d '' entry; do
        local_rel="${entry#./}"
        base_name="$(basename -- "$local_rel")"

        if is_excluded_file_basename "$base_name"; then
          continue
        fi

        mode="$(stat -c '%a' -- "$entry")"

        if [[ -L "$entry" ]]; then
          target="$(readlink -- "$entry")"
          digest="$(printf '%s' "$target" | sha256sum | awk '{print $1}')"
          printf '%s\tl\t%s\t%s\n' "$local_rel" "$mode" "$digest"
        elif [[ -f "$entry" ]]; then
          digest="$(sha256sum -- "$entry" | awk '{print $1}')"
          printf '%s\tf\t%s\t%s\n' "$local_rel" "$mode" "$digest"
        elif [[ -d "$entry" ]]; then
          printf '%s\td\t%s\t-\n' "$local_rel" "$mode"
        fi
      done
  ) > "$temp_file"

  mv -f -- "$temp_file" "$output_file"
}

manifest_hash() {
  sha256sum -- "$1" | awk '{print $1}'
}

read_state_value() {
  local state_file="$1"
  local key="$2"

  awk -F= -v wanted="$key" '$1 == wanted { sub(/^[^=]*=/, ""); print; exit }' "$state_file"
}

write_in_progress_marker() {
  local root_dir="$1"
  local marker temp

  marker="$(in_progress_path)"
  temp="$(mktemp "${marker}.tmp.XXXXXX")"

  {
    printf 'version=%s\n' "$STATE_VERSION"
    printf 'owner_pid=%s\n' "$(owner_pid)"
    printf 'owner_host=%s\n' "$(owner_host)"
    printf 'started_at=%s\n' "$(iso_utc_now)"
    printf 'root_dir=%s\n' "$root_dir"
    printf 'env_dir=%s\n' "$PWD"
  } > "$temp"

  mv -f -- "$temp" "$marker"
}

marker_owner_is_active() {
  local marker pid host current_host

  marker="$(in_progress_path)"
  [[ -f "$marker" ]] || return 1

  pid="$(read_state_value "$marker" owner_pid)"
  host="$(read_state_value "$marker" owner_host)"
  current_host="$(owner_host)"

  [[ -n "$pid" && "$pid" =~ ^[0-9]+$ ]] || return 1
  [[ -z "$host" || "$host" == "$current_host" ]] || return 1

  kill -0 "$pid" 2>/dev/null
}

sync_root_to_deploy() {
  local root_dir="$1"
  local env_dir="$PWD"
  local -a excludes

  build_rsync_args excludes

  log_info "Refreshing deploy snapshot from authoritative root"
  log_info "Repo root:  $root_dir"
  log_info "Env folder: $env_dir"

  rsync -a --checksum --delete \
    "${excludes[@]}" \
    "$root_dir/" \
    "$env_dir/"
}

report_manifest_drift() {
  local baseline="$1"
  local current="$2"
  local work_dir added modified removed

  work_dir="$(mktemp -d)"
  added="$work_dir/added"
  modified="$work_dir/modified"
  removed="$work_dir/removed"
  : > "$added"
  : > "$modified"
  : > "$removed"

  awk -F '\t' -v added="$added" -v modified="$modified" -v removed="$removed" '
    NR == FNR {
      old[$1] = $0
      next
    }
    {
      current[$1] = $0
      if (!($1 in old)) {
        print $1 >> added
      } else if (old[$1] != $0) {
        print $1 >> modified
      }
    }
    END {
      for (path in old) {
        if (!(path in current)) {
          print path >> removed
        }
      }
    }
  ' "$baseline" "$current"

  LC_ALL=C sort -o "$added" "$added"
  LC_ALL=C sort -o "$modified" "$modified"
  LC_ALL=C sort -o "$removed" "$removed"

  printf '\nDeployment drift detected in %s.\n' "$PWD" >&2

  if [[ -s "$modified" ]]; then
    printf '\nModified:\n' >&2
    sed 's/^/  /' "$modified" >&2
  fi

  if [[ -s "$added" ]]; then
    printf '\nAdded:\n' >&2
    sed 's/^/  /' "$added" >&2
  fi

  if [[ -s "$removed" ]]; then
    printf '\nRemoved:\n' >&2
    sed 's/^/  /' "$removed" >&2
  fi

  printf '\nRoot synchronization was not performed.\n' >&2
  printf 'Reconcile the managed deploy changes, or run pm-setup --refresh-from-root\n' >&2
  printf 'to explicitly reconstruct the managed deploy surface from root.\n' >&2

  rm -rf -- "$work_dir"
}

prepare_sync() {
  local force_refresh="$1"
  local root_dir state_file baseline_manifest marker_file
  local current_manifest current_deploy_hash baseline_deploy_hash
  local current_root_manifest current_root_hash baseline_root_hash
  local refresh_reason=""

  if ! is_deploy_snapshot_dir; then
    log_error "must run from deploy/<env>; current directory is '$PWD'"
    return 1
  fi

  root_dir="$(detect_repo_root)"
  state_file="$(state_path)"
  baseline_manifest="$(manifest_path)"
  marker_file="$(in_progress_path)"

  if [[ ! -d "$root_dir" ]]; then
    log_error "repo root '$root_dir' not found"
    return 1
  fi

  ensure_git_local_ignore "$root_dir"

  if [[ -f "$marker_file" ]] && marker_owner_is_active; then
    log_error "another pm-setup process is already active for '$PWD'"
    return 4
  fi

  if [[ "$force_refresh" -eq 1 ]]; then
    refresh_reason="explicit force/overwrite request"
  elif [[ ! -f "$state_file" ]]; then
    refresh_reason="state file is absent"
  elif [[ -f "$marker_file" ]]; then
    refresh_reason="previous setup did not finalize"
  else
    current_manifest="$(mktemp)"
    generate_manifest "$PWD" "$current_manifest"
    current_deploy_hash="$(manifest_hash "$current_manifest")"
    baseline_deploy_hash="$(read_state_value "$state_file" deploy_hash)"

    if [[ -z "$baseline_deploy_hash" ]]; then
      rm -f -- "$current_manifest"
      log_error "state file is invalid or missing deploy_hash: $state_file"
      return 2
    fi

    if [[ "$current_deploy_hash" != "$baseline_deploy_hash" ]]; then
      if [[ -f "$baseline_manifest" ]]; then
        report_manifest_drift "$baseline_manifest" "$current_manifest"
      else
        log_error "deployment drift detected, but the baseline manifest is missing"
        log_error "run pm-setup --refresh-from-root to reconstruct from root"
      fi
      rm -f -- "$current_manifest"
      return 3
    fi

    rm -f -- "$current_manifest"

    current_root_manifest="$(mktemp)"
    generate_manifest "$root_dir" "$current_root_manifest"
    current_root_hash="$(manifest_hash "$current_root_manifest")"
    baseline_root_hash="$(read_state_value "$state_file" root_hash)"
    rm -f -- "$current_root_manifest"

    if [[ -z "$baseline_root_hash" ]]; then
      log_error "state file is invalid or missing root_hash: $state_file"
      return 2
    fi

    if [[ "$current_root_hash" != "$baseline_root_hash" ]]; then
      refresh_reason="authoritative root changed"
    fi
  fi

  if [[ -n "$refresh_reason" ]]; then
    log_info "Refresh authorized: $refresh_reason"
    sync_root_to_deploy "$root_dir"
  else
    log_info "Deploy snapshot matches its baseline and the authoritative root is unchanged"
  fi

  write_in_progress_marker "$root_dir"
  log_info "Deploy synchronization preflight complete"
}

finalize_sync() {
  local root_dir state_file baseline_manifest marker_file
  local root_manifest deploy_manifest root_hash deploy_hash temp_state

  if ! is_deploy_snapshot_dir; then
    log_error "must run from deploy/<env>; current directory is '$PWD'"
    return 1
  fi

  root_dir="$(detect_repo_root)"
  state_file="$(state_path)"
  baseline_manifest="$(manifest_path)"
  marker_file="$(in_progress_path)"

  if [[ ! -f "$marker_file" ]]; then
    log_error "cannot finalize because no in-progress marker exists"
    return 2
  fi

  root_manifest="$(mktemp)"
  deploy_manifest="$(mktemp)"
  generate_manifest "$root_dir" "$root_manifest"
  generate_manifest "$PWD" "$deploy_manifest"
  root_hash="$(manifest_hash "$root_manifest")"
  deploy_hash="$(manifest_hash "$deploy_manifest")"

  temp_state="$(mktemp "${state_file}.tmp.XXXXXX")"
  {
    printf 'version=%s\n' "$STATE_VERSION"
    printf 'root_hash=%s\n' "$root_hash"
    printf 'deploy_hash=%s\n' "$deploy_hash"
    printf 'recorded_at=%s\n' "$(iso_utc_now)"
    printf 'root_dir=%s\n' "$root_dir"
    printf 'env_dir=%s\n' "$PWD"
  } > "$temp_state"

  mv -f -- "$deploy_manifest" "$baseline_manifest"
  mv -f -- "$temp_state" "$state_file"
  rm -f -- "$root_manifest" "$marker_file"

  log_info "Recorded successful deploy baseline"
  log_info "Root hash:   $root_hash"
  log_info "Deploy hash: $deploy_hash"
}

show_status() {
  local state_file baseline_manifest marker_file current_manifest current_hash baseline_hash

  if ! is_deploy_snapshot_dir; then
    log_error "must run from deploy/<env>; current directory is '$PWD'"
    return 1
  fi

  state_file="$(state_path)"
  baseline_manifest="$(manifest_path)"
  marker_file="$(in_progress_path)"

  printf 'Environment: %s\n' "$PWD"
  printf 'State:       %s\n' "$([[ -f "$state_file" ]] && echo present || echo absent)"
  printf 'Manifest:    %s\n' "$([[ -f "$baseline_manifest" ]] && echo present || echo absent)"
  printf 'In progress: %s\n' "$([[ -f "$marker_file" ]] && echo yes || echo no)"

  if [[ ! -f "$state_file" ]]; then
    printf 'Action:      next prepare will refresh from root\n'
    return 0
  fi

  current_manifest="$(mktemp)"
  generate_manifest "$PWD" "$current_manifest"
  current_hash="$(manifest_hash "$current_manifest")"
  baseline_hash="$(read_state_value "$state_file" deploy_hash)"

  if [[ "$current_hash" == "$baseline_hash" ]]; then
    printf 'Drift:       none\n'
  else
    printf 'Drift:       detected\n'
    if [[ -f "$baseline_manifest" ]]; then
      report_manifest_drift "$baseline_manifest" "$current_manifest"
    fi
  fi

  rm -f -- "$current_manifest"
}

main() {
  local action="prepare"
  local force_refresh=0
  local arg

  for arg in "$@"; do
    case "$arg" in
      --prepare)
        action="prepare"
        ;;
      --finalize)
        action="finalize"
        ;;
      --status)
        action="status"
        ;;
      -o|--overwrite|-f|--force)
        force_refresh=1
        ;;
      -h|--help)
        usage
        return 0
        ;;
      *)
        log_error "unknown argument: $arg"
        usage >&2
        return 2
        ;;
    esac
  done

  case "$action" in
    prepare)
      prepare_sync "$force_refresh"
      ;;
    finalize)
      if [[ "$force_refresh" -eq 1 ]]; then
        log_error "--force is not valid with --finalize"
        return 2
      fi
      finalize_sync
      ;;
    status)
      if [[ "$force_refresh" -eq 1 ]]; then
        log_error "--force is not valid with --status"
        return 2
      fi
      show_status
      ;;
  esac
}

if [[ "${BASH_SOURCE[0]}" == "$0" ]]; then
  main "$@"
fi

#!/usr/bin/env bash
set -euo pipefail

script_name="${script_name:-$(basename "${BASH_SOURCE[0]}")}"

# shellcheck source=/dev/null
. "${PORTMASON_SHARE}/pm-helpers"

# -----------------------------------------------------------------------------
# NOTE (2026-02-10):
# This script intentionally avoids Docker Buildx/BuildKit imagetools so it does
# not create or depend on a buildx builder container. Tradeoffs:
# - Only a SINGLE platform image is built/pushed (your current Docker platform).
# - Multi-arch manifest mirroring + OCI index annotations are skipped.
# -----------------------------------------------------------------------------

# --- Preflight checks ---------------------------------------------------------

preflight_checks() {
    if ! command -v jq >/dev/null 2>&1; then
        echo "[${script_name}] FATAL: 'jq' is required but not found on PATH." >&2
        echo "[${script_name}] Install jq (e.g., sudo apt-get install jq) and retry." >&2
        return 1
    fi

    if ! command -v docker >/dev/null 2>&1; then
        echo "[${script_name}] FATAL: 'docker' is required but not found on PATH." >&2
        return 1
    fi

    if [[ -z "${IMAGE_DESCRIPTION:-}" ]]; then
        echo "[${script_name}] FATAL: IMAGE_DESCRIPTION is required for labeling." >&2
        echo "[${script_name}] Set IMAGE_DESCRIPTION in your .env before running pm-tools-wrap-image." >&2
        return 1
    fi
}

# --- Platform detection (single-platform only) --------------------------------
# Usage: detect_current_platform OUT_VAR
detect_current_platform() {
  local __outvar="$1"
  local os arch
  os="$(docker info --format '{{.OSType}}' 2>/dev/null || echo linux)"
  arch="$(docker info --format '{{.Architecture}}' 2>/dev/null || echo amd64)"

  # Docker reports "aarch64" sometimes; normalize to "arm64"
  case "${arch}" in
    aarch64) arch="arm64" ;;
  esac

  printf -v "$__outvar" '%s/%s' "${os}" "${arch}"
}

# --- Digest helpers -----------------------------------------------------------

# Usage: get_local_digest OUT_VAR IMAGE_TAG
# Attempts to resolve a digest from the local image metadata (platform-specific).
get_local_digest() {
  local __outvar="$1"
  local image="$2"
  local ref=""

  ref="$(docker inspect "${image}" --format '{{index .RepoDigests 0}}' 2>/dev/null || true)"
  if [[ -z "${ref}" ]]; then
    printf -v "$__outvar" ''
    return 0
  fi

  printf -v "$__outvar" '%s' "${ref#*@}"
}

# Usage: push_and_capture_digest OUT_VAR IMAGE_TAG
# Pushes an image tag and extracts the sha256 digest from the push output.
push_and_capture_digest() {
  local __outvar="$1"
  local tag="$2"
  local out digest

  out="$(docker push "${tag}" 2>&1)"
  echo "${out}" >&2

  digest="$(printf '%s\n' "${out}" | awk '/digest: sha256:/ {print $2; exit}')"
  if [[ -z "${digest}" ]]; then
    digest="$(printf '%s\n' "${out}" | sed -n 's/.*digest: \(sha256:[0-9a-f]\{64\}\).*/\1/p' | head -n1)"
  fi

  if [[ -z "${digest}" ]]; then
    echo "[${script_name}] FATAL: Could not capture digest from 'docker push' output for ${tag}" >&2
    return 1
  fi

  printf -v "$__outvar" '%s' "${digest}"
}

# --- Signing helpers (uses get_secret from pm-helpers) ------------------------

sign_with_retry() {
  local image_digest="$1"
  local pem="$2"
  local max_attempts="${3:-3}"

  local attempt=1
  while (( attempt <= max_attempts )); do
    local pass="${signing_pass:-}"
    [[ -z "${signing_pass}" ]] && {
        read -s -p "Enter cosign key password (attempt ${attempt}/${max_attempts}): " pass
        echo
    }

    echo "[${script_name}] Verifying if image is already signed: ${image_digest}" >&2
    if env -i PATH="$PATH" COSIGN_PASSWORD="${pass}" cosign verify --key <(printf '%s\n' "$pem") "${image_digest}" >/dev/null 2>&1; then
        signing_pass="$pass"
        echo "[${script_name}] Image already signed with this key: ${image_digest}" >&2
        printf -v pass '%*s' "${#pass}" ''
        unset pass
        return 0
    fi

    echo "[${script_name}] Signing image ${image_digest}" >&2
    if env -i PATH="$PATH" COSIGN_PASSWORD="$pass" cosign sign --key <(printf '%s\n' "$pem") "$image_digest" >/dev/null 2>&1; then
        signing_pass="$pass"
        echo "[${script_name}] Signing succeeded."
        printf -v pass '%*s' "${#pass}" ''
        unset pass
        return 0
    else
        local rc=$?
    fi

    echo "[${script_name}] Signing failed (rc=${rc})."
    (( attempt++ ))
  done

  echo "[${script_name}] Giving up after ${max_attempts} attempts."
  return 1
}

sign_image() {
    local image_digest="$1"

    if [[ -z "${COSIGN_SECRET_PATH:-}" ]]; then
        echo "[${script_name}] COSIGN_SECRET_PATH not set; skipping signing (this is okay for now)." >&2
        return 0
    fi

    if ! command -v cosign >/dev/null 2>&1; then
        echo "[${script_name}] COSIGN_SECRET_PATH set but 'cosign' not found; skipping signing." >&2
        return 0
    fi

    cosign_key="$(get_secret "${COSIGN_SECRET_PATH}")" || {
        echo "[${script_name}] Failed to retrieve signing key from ${COSIGN_SECRET_PATH}" >&2
        exit 1
    }

    sign_with_retry "$image_digest" "$cosign_key" 3
}

ensure_oci_defaults() {
  local publish_tag="${1:-${PUBLISH_TAG:-}}"

  local repo_name
  repo_name="${WRAPPED_REPO##*/}"

  : "${IMAGE_TITLE:=Et al WordPress Core (${repo_name})}"
  : "${IMAGE_VENDOR:=Et al Solutions LLC}"

  : "${IMAGE_SOURCE_BASE:=https://github.com/solutions-etal}"
  : "${IMAGE_URL_BASE:=${IMAGE_SOURCE_BASE}}"

  : "${IMAGE_SOURCE:=${IMAGE_SOURCE_BASE}/${repo_name}}"
  : "${IMAGE_URL:=${IMAGE_URL_BASE}/${repo_name}}"
  : "${IMAGE_DOCUMENTATION:=${IMAGE_SOURCE}#readme}"

  if [[ -z "${IMAGE_REVISION:-}" ]] && command -v git >/dev/null 2>&1; then
    IMAGE_REVISION="$(git rev-parse HEAD 2>/dev/null || true)"
    export IMAGE_REVISION
  fi

  if [[ -z "${IMAGE_VERSION:-}" ]] && [[ -n "${publish_tag}" ]]; then
    local tag_part build_id
    tag_part="${publish_tag##*:}"
    build_id="${tag_part##*-}"
    if [[ "${build_id}" =~ ^[0-9]{8}T[0-9]{6}Z$ ]]; then
      IMAGE_VERSION="${build_id}"
      export IMAGE_VERSION
    fi
  fi
}

# --- Main script logic --------------------------------------------------------

main() {
  preflight_checks

  VENDOR_IMAGE_TAG="${1:-${VENDOR_IMAGE_TAG:-}}"
  WRAPPED_REPO="${2:-${WRAPPED_REPO:-}}"

  if [[ -z "${VENDOR_IMAGE_TAG}" ]]; then
    echo "[${script_name}] ERROR: VENDOR_IMAGE_TAG must be provided (arg1 or env VENDOR_IMAGE_TAG)." >&2
    return 1
  fi

  if [[ -z "${WRAPPED_REPO}" ]]; then
    echo "[${script_name}] ERROR: WRAPPED_REPO must be provided (arg2 or env WRAPPED_REPO)." >&2
    return 1
  fi

  WRAP_DOCKERFILE="${WRAP_DOCKERFILE:-Dockerfile}"
  SITE_DIR="${SITE_DIR:-.}"

  if [[ ! -f "${WRAP_DOCKERFILE}" ]]; then
    echo "[${script_name}] ERROR: WRAP_DOCKERFILE not found at ${WRAP_DOCKERFILE}" >&2
    return 1
  fi

  IMAGE_DESCRIPTION="${IMAGE_DESCRIPTION:-unknown}"

  WRAP_DOCKERFILE_FULLPATH="$(realpath "${WRAP_DOCKERFILE}")"
  SITE_DIR="$(realpath "${SITE_DIR}")"
  DATE_TAG="${DATE_TAG:-$(date -u +%Y%m%dT%H%M%SZ)}"

  local vendor_tag
  vendor_tag="${VENDOR_IMAGE_TAG#*:}"
  if [[ "${vendor_tag}" == "${VENDOR_IMAGE_TAG}" ]]; then
    vendor_tag="latest"
  fi

  FLAVOR_TAG="${FLAVOR_TAG:-${vendor_tag}}"

  WRAPPED_TAG_VERSIONED="${WRAPPED_REPO}:${FLAVOR_TAG}-${DATE_TAG}"
  WRAPPED_TAG_STABLE="${WRAPPED_REPO}:${FLAVOR_TAG}"

  local platform=""
  detect_current_platform platform

  echo "[${script_name}] Vendor image        : ${VENDOR_IMAGE_TAG}" >&2
  echo "[${script_name}] Wrapped repo        : ${WRAPPED_REPO}" >&2
  echo "[${script_name}] Image description   : ${IMAGE_DESCRIPTION}" >&2
  echo "[${script_name}] Flavor tag          : ${FLAVOR_TAG}" >&2
  echo "[${script_name}] Date tag            : ${DATE_TAG}" >&2
  echo "[${script_name}] Versioned tag       : ${WRAPPED_TAG_VERSIONED}" >&2
  echo "[${script_name}] Stable tag          : ${WRAPPED_TAG_STABLE}" >&2
  echo "[${script_name}] Wrap Dockerfile     : ${WRAP_DOCKERFILE_FULLPATH}" >&2
  echo "[${script_name}] Build context (root): ${SITE_DIR}" >&2
  echo "[${script_name}] Build platform      : ${platform} (single-platform; buildx disabled)" >&2

  echo "[${script_name}] Pulling vendor image (best-effort)..." >&2
  docker pull "${VENDOR_IMAGE_TAG}" || true

  local vendor_digest=""
  get_local_digest vendor_digest "${VENDOR_IMAGE_TAG}" || true
  if [[ -n "${vendor_digest}" ]]; then
    echo "[${script_name}] Vendor digest (platform-specific): ${VENDOR_IMAGE_TAG}@${vendor_digest}" >&2
  else
    echo "[${script_name}] WARN: Could not determine vendor digest locally; proceeding without a pinned digest." >&2
  fi

  echo "[${script_name}] Building wrapped image (single-platform)..." >&2
  docker build \
    --build-arg "VENDOR_IMAGE_TAG=${VENDOR_IMAGE_TAG}" \
    --build-arg "VENDOR_DIGEST=${vendor_digest}" \
    --build-arg "IMAGE_DESCRIPTION=${IMAGE_DESCRIPTION}" \
    -f "${WRAP_DOCKERFILE_FULLPATH}" \
    -t "${WRAPPED_TAG_VERSIONED}" \
    -t "${WRAPPED_TAG_STABLE}" \
    "${SITE_DIR}"

  local wrapped_digest
  echo "[${script_name}] Pushing versioned tag (capturing digest)..." >&2
  push_and_capture_digest wrapped_digest "${WRAPPED_TAG_VERSIONED}"

  echo "[${script_name}] Pushing stable tag..." >&2
  docker push "${WRAPPED_TAG_STABLE}" >/dev/null

  ensure_oci_defaults "${WRAPPED_TAG_VERSIONED}"

  local wrapped_repo_digest
  wrapped_repo_digest="${WRAPPED_REPO}@${wrapped_digest}"
  echo "[${script_name}] Wrapped image digest: ${wrapped_repo_digest}" >&2

  signing_pass=
  sign_image "${wrapped_repo_digest}"
  unset signing_pass

  echo "[${script_name}] Done." >&2
}

main "$@"

#!/usr/bin/env bash
set -euo pipefail

: "${CUR_SCRIPT:=$(basename "${BASH_SOURCE[0]:-${0:-}}")}"
export CUR_SCRIPT

# Usage:
#   pm-update-docker-env-file [path-to-compose]
#
# Default file is ./docker-compose.yml

main() {
    FILE="${1:-docker-compose.yml}"

    if [[ ! -f "$FILE" ]]; then
    echo "ERROR: $FILE not found." >&2
    exit 1
    fi

    if ! command -v yq >/dev/null 2>&1; then
    echo "ERROR: 'yq' (mikefarah Go version) is required but not found in PATH." >&2
    exit 1
    fi

    # Find env file in current directory:
    # 1) Prefer .env if it exists
    # 2) Otherwise, take the first *.env file (lexicographically)
    ENV_FILE=""

    if [[ -f ".env" ]]; then
    ENV_FILE=".env"
    else
    shopt -s nullglob
    env_candidates=( *.env )
    shopt -u nullglob

    if (( ${#env_candidates[@]} > 0 )); then
        ENV_FILE="${env_candidates[0]}"
    fi
    fi

    if [[ -z "$ENV_FILE" ]]; then
    echo "ERROR: No env file found in current directory." >&2
    echo "Looked for '.env' or '*.env' here: $(pwd)" >&2
    exit 1
    fi

    echo "Using env file: $ENV_FILE"

    # Backup original
    cp "$FILE" "${FILE}.bak"

    # For every service:
    # - overwrite env_file with the detected ENV_FILE
    # This uses Go yq v4 syntax.
    yq eval -i ".services[].env_file = \"$ENV_FILE\"" "$FILE"

    echo "Updated $FILE -> env_file: \"$ENV_FILE\" for all services."
    echo "Backup saved as ${FILE}.bak."
}

main "$@"

===== pm-util-render-env =====
#!/usr/bin/env python3
import os
import re
import sys
import json
import fnmatch
from pathlib import Path

USAGE = """pm-util-render-env [--dotenv|--json] [--print-exports] [--out FILE] [FILE|-]

Renders environment-style expansions in text:
  ${VAR}
  ${VAR:-default}   default if VAR unset OR empty
  ${VAR-default}    default if VAR unset only
  ${VAR:=default}   default if VAR unset OR empty AND records an assignment
  $VAR              simple expansion (VAR must be [A-Za-z_][A-Za-z0-9_]*)

Options:
  --dotenv          Treat input as a .env file: parse KEY=VALUE lines top-to-bottom,
                    updating an internal environment map so later lines can reference
                    earlier assignments in the same file.
  --json            Treat input as a .env file and emit a public-safe JSON object.
                    Secret-bearing keys are excluded by default.
  --print-exports   Print export statements for := assignments to STDERR.
  --out FILE        Write rendered output to FILE instead of STDOUT.
                    (Diagnostics still go to STDERR.)

Notes:
- This utility cannot modify the parent shell's environment.
- With --print-exports, assignments from := are printed as export statements to STDERR.
"""

# Matches ${VAR}, ${VAR:-default}, ${VAR-default}, ${VAR:=default}
PAT_BRACED = re.compile(r"\$\{([A-Za-z_][A-Za-z0-9_]*)(?:(:-|:=|-)(.*?))?\}")

# Matches $VAR (but not \$VAR); does not match $1, $$, $- etc.
PAT_DOLLAR = re.compile(r"(?<!\\)\$([A-Za-z_][A-Za-z0-9_]*)")

# .env assignment lines (supports optional leading "export ")
ENV_LINE = re.compile(r"^\s*(?:export\s+)?([A-Za-z_][A-Za-z0-9_]*)\s*=\s*(.*)$")

PUBLIC_JSON_EXCLUDE_GLOBS = (
    "*_TOKEN",
    "*_SECRET",
    "*_KEY",
    "*_PASSWORD",
    "SECRET_PATHS",
    "ENV_FILE",
    "DEBUG",
    "VERBOSE",
    "CUR_SCRIPT",
    "PATH",
    "HOME",
    "PWD",
    "OLDPWD",
    "SHELL",
    "SHLVL",
    "USER",
    "LOGNAME",
    "LANG",
    "LANGUAGE",
    "LC_*",
    "TERM",
    "XDG_*",
    "PORT",
)

def _is_public_json_key(key: str) -> bool:
    return not any(fnmatch.fnmatchcase(key, pat) for pat in PUBLIC_JSON_EXCLUDE_GLOBS)

def _strip_inline_comment(value: str) -> str:
    # Remove unquoted comments. This is intentionally conservative and only
    # treats # as a comment when it is outside quotes and preceded by whitespace.
    in_single = False
    in_double = False
    escaped = False
    for i, ch in enumerate(value):
        if escaped:
            escaped = False
            continue
        if ch == "\\" and in_double:
            escaped = True
            continue
        if ch == "'" and not in_double:
            in_single = not in_single
            continue
        if ch == '"' and not in_single:
            in_double = not in_double
            continue
        if ch == "#" and not in_single and not in_double and (i == 0 or value[i - 1].isspace()):
            return value[:i].rstrip()
    return value.rstrip()

def shell_quote(value: str) -> str:
    # single-quote safe for bash: ' -> '"'"'
    return "'" + value.replace("'", "'\"'\"'") + "'"

def _strip_outer_quotes(s: str) -> str:
    s = s.strip()
    if len(s) >= 2 and ((s[0] == s[-1] == '"') or (s[0] == s[-1] == "'")):
        return s[1:-1]
    return s

def render_text(text: str, env: dict[str, str], assignments: dict[str, str]) -> str:
    """
    Render Bash-like environment expansions using 'env'. Supported forms:
      ${VAR}
      ${VAR:-default}
      ${VAR-default}
      ${VAR:=default}
      ${VAR:+alternate}
      $VAR

    Defaults/alternates may themselves contain expansions, including nested
    ${...}. Command substitution is intentionally not evaluated.
    """

    def find_matching_brace(src: str, open_index: int) -> int:
        depth = 1
        i = open_index + 1
        while i < len(src):
            if src[i] == "$" and i + 1 < len(src) and src[i + 1] == "{":
                depth += 1
                i += 2
                continue
            if src[i] == "}":
                depth -= 1
                if depth == 0:
                    return i
            i += 1
        return -1

    def expand_expr(expr: str) -> str:
        m = re.match(r"^([A-Za-z_][A-Za-z0-9_]*)(.*)$", expr, re.DOTALL)
        if not m:
            return "${" + expr + "}"

        key = m.group(1)
        rest = m.group(2)
        val = env.get(key)

        if rest == "":
            return "" if val is None else val

        op = None
        default = ""
        for candidate in (":-", ":=", ":+", "-"):
            if rest.startswith(candidate):
                op = candidate
                default = rest[len(candidate):]
                break

        if op is None:
            return "${" + expr + "}"

        if op == ":-":
            if val is None or val == "":
                return render_text(default, env, assignments)
            return val

        if op == "-":
            if val is None:
                return render_text(default, env, assignments)
            return val

        if op == ":=":
            if val is None or val == "":
                assigned = render_text(default, env, assignments)
                env[key] = assigned
                assignments[key] = assigned
                return assigned
            return val

        if op == ":+":
            if val is not None and val != "":
                return render_text(default, env, assignments)
            return ""

        return "${" + expr + "}"

    out_parts: list[str] = []
    i = 0
    while i < len(text):
        if text[i] == "$" and i + 1 < len(text) and text[i + 1] == "{":
            close = find_matching_brace(text, i + 1)
            if close == -1:
                out_parts.append(text[i])
                i += 1
                continue
            out_parts.append(expand_expr(text[i + 2:close]))
            i = close + 1
            continue
        out_parts.append(text[i])
        i += 1

    out = "".join(out_parts)

    def repl_dollar(m: re.Match) -> str:
        key = m.group(1)
        val = env.get(key)
        return "" if val is None else val

    previous = None
    for _ in range(8):
        if out == previous:
            break
        previous = out
        out = PAT_DOLLAR.sub(repl_dollar, out)
        if "${" in out:
            # Resolve expansions created by defaults/alternates.
            nested_assignments: dict[str, str] = assignments
            # Avoid infinite recursion on unchanged malformed input.
            nested = render_text(out, env, nested_assignments) if out != previous else out
            out = nested

    return out

def render_dotenv(text: str, base_env: dict[str, str], print_exports: bool) -> str:
    env = dict(base_env)
    assignments: dict[str, str] = {}

    out_lines: list[str] = []
    for raw_line in text.splitlines(keepends=True):
        m = ENV_LINE.match(raw_line)
        if not m:
            out_lines.append(render_text(raw_line, env, assignments))
            continue

        key = m.group(1)
        rhs = m.group(2)

        rendered_rhs = render_text(rhs, env, assignments)

        new_line = f"{key}={rendered_rhs}"
        if raw_line.endswith("\n") and not new_line.endswith("\n"):
            new_line += "\n"

        out_lines.append(new_line)

        # Update env for later lines
        env[key] = _strip_outer_quotes(rendered_rhs)

    if print_exports and assignments:
        for k, v in assignments.items():
            sys.stderr.write(f"export {k}={shell_quote(v)}\n")

    return "".join(out_lines)

def render_dotenv_map(text: str, base_env: dict[str, str], print_exports: bool) -> dict[str, str]:
    env = dict(base_env)
    assignments: dict[str, str] = {}
    out: dict[str, str] = {}
    in_array_key: str | None = None

    for raw_line in text.splitlines():
        stripped = raw_line.strip()

        # Skip multiline Bash arrays entirely. They are operator-side config,
        # not public browser config.
        if in_array_key is not None:
            if stripped == ")":
                in_array_key = None
            continue

        if not stripped or stripped.startswith("#"):
            continue

        m = ENV_LINE.match(raw_line)
        if not m:
            continue

        key = m.group(1)
        rhs = m.group(2).strip()

        if rhs == "(":
            in_array_key = key
            continue

        rendered_rhs = render_text(_strip_inline_comment(rhs), env, assignments)
        value = _strip_outer_quotes(rendered_rhs)
        env[key] = value

        if _is_public_json_key(key):
            out[key] = value

    if print_exports and assignments:
        for k, v in assignments.items():
            sys.stderr.write(f"export {k}={shell_quote(v)}\n")

    return out

def render_generic(text: str, print_exports: bool) -> str:
    assignments: dict[str, str] = {}
    env = dict(os.environ)
    out = render_text(text, env, assignments)

    if print_exports and assignments:
        for k, v in assignments.items():
            sys.stderr.write(f"export {k}={shell_quote(v)}\n")

    return out

def _parse_args(argv: list[str]):
    dotenv_mode = False
    json_mode = False
    print_exports = False
    out_path: str | None = None

    args = argv[1:]
    if not args or args[0] in ("-h", "--help"):
        return None, None, None, None, None, 2

    i = 0
    while i < len(args):
        a = args[i]

        if a == "--dotenv":
            dotenv_mode = True
            i += 1
            continue

        if a == "--json":
            json_mode = True
            dotenv_mode = True
            i += 1
            continue

        if a == "--print-exports":
            print_exports = True
            i += 1
            continue

        if a == "--out":
            if i + 1 >= len(args):
                sys.stderr.write("ERROR: --out requires a filename\n")
                sys.stderr.write(USAGE)
                return None, None, None, None, None, 2
            out_path = args[i + 1]
            i += 2
            continue

        if a.startswith("--out="):
            out_path = a.split("=", 1)[1]
            if not out_path:
                sys.stderr.write("ERROR: --out requires a filename\n")
                sys.stderr.write(USAGE)
                return None, None, None, None, None, 2
            i += 1
            continue

        if a in ("-h", "--help"):
            sys.stderr.write(USAGE)
            return None, None, None, None, None, 2

        src = a
        rest = args[i + 1 :]
        if rest:
            sys.stderr.write("ERROR: unexpected extra arguments\n")
            sys.stderr.write(USAGE)
            return None, None, None, None, None, 2
        return dotenv_mode, json_mode, print_exports, out_path, src, 0

    sys.stderr.write(USAGE)
    return None, None, None, None, None, 2

def main(argv: list[str]) -> int:
    dotenv_mode, json_mode, print_exports, out_path, src, code = _parse_args(argv)
    if code != 0:
        return code
    assert src is not None
    assert dotenv_mode is not None
    assert json_mode is not None
    assert print_exports is not None

    try:
        if src == "-":
            text = sys.stdin.read()
        else:
            text = Path(src).read_text(encoding="utf-8")
    except Exception as e:
        sys.stderr.write(f"ERROR: failed to read {src!r}: {e}\n")
        return 1

    try:
        if json_mode:
            rendered = json.dumps(render_dotenv_map(text, dict(os.environ), print_exports), indent=2, sort_keys=True) + "\n"
        elif dotenv_mode:
            rendered = render_dotenv(text, dict(os.environ), print_exports)
        else:
            rendered = render_generic(text, print_exports)
    except Exception as e:
        sys.stderr.write(f"ERROR: render failed: {e}\n")
        return 1

    try:
        if out_path:
            Path(out_path).write_text(rendered, encoding="utf-8")
        else:
            sys.stdout.write(rendered)
    except Exception as e:
        target = out_path if out_path else "<stdout>"
        sys.stderr.write(f"ERROR: failed to write output to {target!r}: {e}\n")
        return 1

    return 0

if __name__ == "__main__":
    raise SystemExit(main(sys.argv))
